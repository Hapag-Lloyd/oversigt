16  	public void shouldCreateObject_whenParsingStringArray_givenValidCommandLine() {

18  		final String[] givenCommandLine
19  				= new String[] { "--debug", "--startEventSources", "--ldapBindPassword", "idefix" };
20  
21  		// when
22  		final CommandLineOptions actualStartOptions = CommandLineOptions.parse(givenCommandLine).get();
23  
24  		// then
