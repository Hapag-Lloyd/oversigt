16  /**
17   * A jira client using multiple threads to execute jira queries simultaneously.
18   * The number of threads changes dynamically depending on the current workload.
19   *
20   * @author Olaf Neumann
21   *
22   */
23  class ThreadedJiraClient extends JiraClientFilter {
24  	private static final int TIMEOUT = 3;
25  
26  	private static final TimeUnit TIMEOUT_UNIT = TimeUnit.MINUTES;
27  
28  	private static final GrowingExecutoror.html#GrowingExecutor">GrowingExecutor EXECUTOR = new GrowingExecutor(1,
29  			50,
30  			Duration.ofSeconds(2),
31  			Duration.ofMinutes(4),
32  			CountingThreadFactory.createDaemonThreadFactory("ThreadJiraClient-executor-"));
33  
34  	/**
35  	 * Create a new threaded jira client
36  	 *
37  	 * @param connection  the server where to connect to
38  	 * @param credentials the credentials to be used
39  	 * @throws JiraClientException if something fails while creating the client
40  	 */
41  	ThreadedJiraClient(final ServerConnection connection, final Credentials credentials) throws JiraClientException {
42  		super(JiraClientFactory.createUnlimitedJiraClient(connection, credentials));
43  	}
44  
45  	/**
46  	 * Perform the JIRA search using a separate thread. The executing thread is
47  	 * handled by an executor service.
48  	 *
49  	 * @see com.hlag.oversigt.connect.jira.JiraClient#search(java.lang.String, int,
50  	 *      int)
51  	 */
52  	@Override
53  	public List<Issue> search(final String jql, final int maxResults, final int startAt)
54  			throws JiraClientException, TimeoutException {
55  		try {
56  			final Callable<List<Issue>> callable = () -> getJiraClient().search(jql, maxResults, startAt);
57  			return EXECUTOR.execute(callable).get(TIMEOUT, TIMEOUT_UNIT);
58  		} catch (final InterruptedException | ExecutionException e) {
59  			if (e.getCause() != null && e.getCause() instanceof JiraClientException) {
60  				throw (JiraClientException) e.getCause();
61  			}
62  			throw new JiraClientException("Unable to execute JIRA query", e);
63  		}
64  	}
65  }
