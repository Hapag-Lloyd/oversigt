3   import com.google.common.collect.HashBasedTable;
4   import com.google.common.collect.Table;
5   import com.hlag.oversigt.properties.Credentials;
6   import com.hlag.oversigt.properties.ServerConnection;
7   
8   /**
9    * Factory for creating {@link JiraClient}s without knowing the implementation
10   *
11   * @author Olaf Neumann
12   *
13   */
14  public final class JiraClientFactory {
15  	private JiraClientFactory() {
16  		throw new RuntimeException("Do not instantiate this class.");
17  	}
18  
19  	/**
20  	 * Create a new {@link JiraClient} for the given parameters.
21  	 *
22  	 * @param connection  the server details to be used to create the server
23  	 *                    connection
24  	 * @param credentials the credentials to be used for authenticating the JIRA
25  	 *                    requests
26  	 * @return created {@link JiraClient}
27  	 * @throws JiraClientException in case of error while creating the client
28  	 */
29  	public static JiraClient createJiraClient(final ServerConnection connection, final Credentials credentials)
30  			throws JiraClientException {
31  		return new ThreadedJiraClient(connection, credentials);
32  	}
33  
34  	private static final Table<String, String, UnlimitedJiraClient> CLIENT_CACHE = HashBasedTable.create();
35  
36  	/**
37  	 * Create a new jira client for the givon connection and credentials. Only one
38  	 * client per connection and credentials will be created and then cached.
39  	 *
40  	 * @param connection  where to connect the jira client to
41  	 * @param credentials the credentials to use for the jira connection
42  	 * @return the newly create jira client
43  	 * @throws JiraClientException if something fails while creating the connection
44  	 */
45  	static synchronized JiraClient createUnlimitedJiraClient(final ServerConnection connection,
46  			final Credentials credentials) throws JiraClientException {
47  		if (!CLIENT_CACHE.contains(connection.getUrl(), credentials.getUsername())) {
48  			CLIENT_CACHE.put(connection.getUrl(),
49  					credentials.getUsername(),
50  					new UnlimitedJiraClient(connection, credentials));
51  		}
52  		return CLIENT_CACHE.get(connection.getUrl(), credentials.getUsername());
53  	}
54  }
