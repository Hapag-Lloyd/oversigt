255 	public int getComputedTileWidth() {
256 		return (getScreenWidth() - TILE_DISTANCE * getColumns()) / getColumns();
257 	}
258 
259 	@JsonIgnore
260 	public int getComputedTileHeight() {
261 		int height = getComputedTileWidth() - 2;
262 		height = getScreenHeight() / height;
263 		height = (getScreenHeight() - TILE_DISTANCE * height) / height;
264 		return height;
265 	}

272 	public Dashboard copy() {
273 		final Dashboard dashboard = new Dashboard(getId(),
274 				getTitle(),
275 				isEnabled(),
276 				getScreenWidth(),
277 				getScreenHeight(),
278 				getColumns(),
279 				getBackgroundColor(),
280 				getColorScheme(),
281 				getForegroundColorStart(),
282 				getForegroundColorEnd(),
283 				getOwners(),
284 				getEditors());
285 		dashboard.getModifiableWidgets().addAll(getWidgets());
286 		return dashboard;
