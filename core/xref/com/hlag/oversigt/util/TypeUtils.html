111 	public static <T> Stream<Class<T>> findClasses(final Package packageToSearch, final Predicate<Class<?>> filter) {
112 		LOGGER.info("Inspecting classes using ClassLoader[{}] and package[{}]",
113 				Thread.currentThread().getContextClassLoader(),
114 				packageToSearch);
115 		final Set<ClassInfo> classes;
116 		try {
117 			classes = ClassPath.from(Thread.currentThread().getContextClassLoader())
118 					.getTopLevelClassesRecursive(packageToSearch.getName());
119 		} catch (final IOException e) {
120 			throw new RuntimeException("Unable to search classes in class path", e);
121 		}
122 
123 		// iterates over all classes, filter by HandlesEvent annotation and transforms
124 		// stream to needed form
125 		@SuppressWarnings("unchecked")
126 		final Stream<Class<T>> stream = classes.stream()
127 				.map(TypeUtils::loadClassInfo)
128 				.filter(Optional::isPresent)
129 				.map(Optional::get)
130 				.filter(c -> !c.isInterface())
131 				.filter(filter)
132 				.filter(c -> !Modifier.isAbstract(c.getModifiers()))
133 				.map(c -> (Class<T>) c);
134 		return stream;
135 	}

138 			final Collection<String> classNames,
139 			final Class<T> assignableTo,
140 			final Class<? extends Annotation> annotationToBePresent) {
141 		final Predicate<Class<?>> predicate
142 				= c -> assignableTo.isAssignableFrom(c) && c.isAnnotationPresent(annotationToBePresent);
143 		@SuppressWarnings("unchecked")
144 		final Stream<Class<T>> stream = classNames.stream()
145 				.map(s -> loadClassInfo(classLoader, s))

149 				.filter(predicate)
150 				.filter(c -> !Modifier.isAbstract(c.getModifiers()))
151 				.map(c -> (Class<T>) c);
152 		return stream;
153 	}

186 	public static List<String> listClassesInJarFiles(final URL[] jarFileUrls) throws IOException {
187 		final List<String> classNames = new ArrayList<>();
188 		for (final URL jarFileUrl : jarFileUrls) {
189 			try (JarInputStream in = new JarInputStream(jarFileUrl.openStream())) {
190 				JarEntry jarEntry;
191 				while ((jarEntry = in.getNextJarEntry()) != null) {
192 					if (!jarEntry.isDirectory()) {
193 						String name = jarEntry.getName();
194 						if (name.toLowerCase().endsWith(".class")) {
195 							name = name.substring(0, name.length() - ".class".length());
196 							name = name.replace("/", ".");
197 							classNames.add(name);
198 						}
199 					}
200 				}
201 			}
202 		}
203 		return classNames;
204 	}

213 	 * Inspects the given value and returns a {@link Map} containing all members
214 	 * with their values. If the given object is an {@link Optional} it will be
215 	 * unwrapped - if the {@link Optional} is empty an empty {@link Map} will be
216 	 * returned.
217 	 *
218 	 * @param object the object to inspect
219 	 * @return a map containing all members of the given object
220 	 */
221 	public static Map<String, Object> toMemberMap(final Object object) {
222 		if (object instanceof Optional) {
223 			if (!((Optional<?>) object).isPresent()) {
224 				return new HashMap<>();
225 			}
226 			return toMemberMap(((Optional<?>) object).get());
227 		}
228 		try {
229 			return Stream.of(Introspector.getBeanInfo(object.getClass(), Object.class).getPropertyDescriptors())
230 					.filter(i -> i.getReadMethod() != null)
231 					.filter(i -> !i.getReadMethod().isAnnotationPresent(JsonIgnore.class))
232 					.filter(i -> i.getWriteMethod() == null
233 							|| !i.getWriteMethod().isAnnotationPresent(JsonIgnore.class))
234 					.collect(Collectors.toMap(PropertyDescriptor::getName, pd -> get(pd.getReadMethod(), object)));
235 		} catch (final Exception e) {
236 			throw new RuntimeException(e);
237 		}
238 	}

249 			final Class<T> clazz) {
250 		Collection<SerializablePropertyMember> members = new LinkedHashSet<>();
251 		try {
252 			members.add(new SerializablePropertyMember(SerializableProperty.class.getDeclaredField("id")));
253 			members.add(new SerializablePropertyMember(SerializableProperty.class.getDeclaredField("name")));
254 		} catch (final NoSuchFieldException | SecurityException e) {
255 			throw new RuntimeException("Unable to find standard values", e);
256 		}
257 		members.addAll(streamFields(clazz).filter(f -> !Modifier.isTransient(f.getModifiers()))
258 				.filter(f -> !Modifier.isStatic(f.getModifiers()))
259 				.map(SerializablePropertyMember::new)
260 				.collect(Collectors.toList()));
261 		members = new ArrayList<>(members);
262 		((List<?>) members).remove(0);
263 		return members;
264 	}

274 	public static <C> Constructor<C> getAppropriateConstructor(final Class<C> c, final Object[] initArguments) {
275 		final Object[] initArgs = Nullables.orElseGet(initArguments, () -> new Object[0]);
276 		for (final Constructor<?> constructor : c.getDeclaredConstructors()) {
277 			final Class<?>[] types = constructor.getParameterTypes();
278 			if (types.length != initArgs.length) {
279 				continue;
280 			}
281 			boolean match = true;
282 			for (int i = 0; i < types.length; i += 1) {
283 				final Class<?> need = types[i];
284 				final Class<?> got = initArgs[i].getClass();
285 				if (!need.isAssignableFrom(got)) {
286 					if (need.isPrimitive()) {
287 						match = int.class.equals(need) && Integer.class.equals(got)
288 								|| long.class.equals(need) && Long.class.equals(got)
289 								|| char.class.equals(need) && Character.class.equals(got)
290 								|| short.class.equals(need) && Short.class.equals(got)
291 								|| boolean.class.equals(need) && Boolean.class.equals(got)
292 								|| byte.class.equals(need) && Byte.class.equals(got);
293 					} else {
294 						match = false;
295 					}
296 				}
297 				if (!match) {
298 					break;
299 				}
300 			}
301 			if (match) {
302 				return (Constructor<C>) constructor;
303 			}
304 		}
305 		throw new IllegalArgumentException("Cannot find an appropriate constructor for class "
306 				+ c
307 				+ " and arguments "
308 				+ Arrays.toString(initArgs));
309 	}

324 	public static <T> T createInstance(final Class<? extends T> clazz, final Object... parameters) {
325 		Objects.requireNonNull(parameters);
326 		for (final Object param : parameters) {
327 			Objects.requireNonNull(param);
328 		}
329 		if (parameters.length == 1 && parameters[0].getClass() == clazz) {
330 			return (T) parameters[0];
331 		} else if (clazz.isPrimitive()) {
332 			if (int.class == clazz) {
333 				final int value = Integer.parseInt((String) parameters[0]);
334 				final Object[] values = { value };
335 				return (T) values[0];
336 			} else if (long.class == clazz) {
337 				final Object[] values = { Long.parseLong((String) parameters[0]) };
338 				return (T) values[0];
339 			} else if (double.class == clazz) {
340 				final Object[] values = { Double.parseDouble((String) parameters[0]) };
341 				return (T) values[0];
342 			} else if (float.class == clazz) {
343 				final Object[] values = { Float.parseFloat((String) parameters[0]) };
344 				return (T) values[0];
345 			} else if (byte.class == clazz) {
346 				final Object[] values = { Byte.parseByte((String) parameters[0]) };
347 				return (T) values[0];
348 			} else if (short.class == clazz) {
349 				final Object[] values = { Short.parseShort((String) parameters[0]) };
350 				return (T) values[0];
351 			} else if (char.class == clazz) {
352 				final Object[] values = { ((String) parameters[0]).charAt(0) };
353 				return (T) values[0];
354 			} else {
355 				throw new RuntimeException("Unknown primitive type: " + clazz.getName());
356 			}
357 		} else {
358 			final List<Constructor<?>> constructors = Arrays.stream(clazz.getDeclaredConstructors())
359 					.filter(c -> c.getParameterCount() == parameters.length)
360 					.filter(c -> {
361 						for (int i = 0; i < parameters.length; i += 1) {
362 							if (!c.getParameterTypes()[i].isAssignableFrom(parameters[i].getClass())) {
363 								return false;
364 							}
365 						}
366 						return true;
367 					})
368 					.limit(2)
369 					.collect(Collectors.toList());
370 			if (constructors.size() == 1) {
371 				constructors.get(0).setAccessible(true);
372 				try {
373 					return (T) constructors.get(0).newInstance(parameters);
374 				} catch (final InstantiationException
375 						| IllegalAccessException
376 						| IllegalArgumentException
377 						| InvocationTargetException e) {
378 					throw new RuntimeException("Unable to create instance of class: " + clazz.getName(), e);
379 				}
380 			}
381 
382 			final List<Method> fabricMethods = Arrays.stream(clazz.getDeclaredMethods())
383 					.filter(m -> Modifier.isStatic(m.getModifiers()))
384 					.filter(m -> m.getParameterTypes().length == parameters.length)
385 					.filter(m -> clazz.isAssignableFrom(m.getReturnType()))
386 					.filter(m -> {
387 						for (int i = 0; i < parameters.length; i += 1) {
388 							if (!m.getParameterTypes()[i].isAssignableFrom(parameters[i].getClass())) {
389 								return false;
390 							}
391 						}
392 						return true;
393 					})
394 					.collect(Collectors.toList());
395 			if (fabricMethods.size() == 1) {
396 				return invokeStaticMethod(fabricMethods.get(0), parameters);
397 			}
398 			if (fabricMethods.size() > 1) {
399 				final Optional<T> maybe = tryStaticMethods(fabricMethods,
400 						new String[] {
401 								"parse",
402 								"create",
403 								"compile",
404 								"get",
405 								"createInstance",
406 								"getInstance",
407 								"construct" },
408 						parameters);
409 				return maybe.get();
410 			}
411 			throw new RuntimeException("No fitting constructor or method found for class: " + clazz.getName());
412 		}
413 	}

417 			final String[] names,
418 			final Object... values) {
419 		return (Optional<T>) Arrays.stream(names)

441 	public static Optional<Method> getMethod(final Class<?> clazz,
442 			final List<String> methodNames,
443 			final Class<?>[] parameterTypes) {
444 		if (methodNames.isEmpty()) {
445 			return Optional.empty();
446 		}
447 
448 		Class<?> currentClazz = clazz;
449 		while (currentClazz != Object.class) {
450 			final Optional<Method> method = Stream.of(currentClazz.getDeclaredMethods())
451 					.filter(m -> methodNames.contains(m.getName()))
452 					.filter(m -> Arrays.deepEquals(m.getParameterTypes(), parameterTypes))
453 					.sorted(new IndexOfComparator(methodNames).thenComparing(CLASS_DEPTH_COMPARATOR))
454 					.findFirst();
455 			if (method.isPresent()) {
456 				return method;
457 			}
458 			currentClazz = currentClazz.getSuperclass();
459 		}
460 		return Optional.empty();
461 	}

464 		Class<?> currentClass = clazz;
465 		int i = 1;
466 		while (currentClass != null && currentClass != Object.class) {
467 			i += 1;
468 			currentClass = currentClass.getSuperclass();
469 		}
470 		return i;
471 	}

685 	public static final class ClassProxy implements InvocationHandler {
686 		@SuppressWarnings("unchecked")
687 		public static <T> T create(final Class<T> clazz, final ReturnValue... returns) {
688 			return (T) Proxy.newProxyInstance(ClassProxy.class.getClassLoader(),
689 					new Class[] { clazz },
690 					new ClassProxy(returns));
691 		}
692 
693 		private final List<ReturnValue> returns;
694 
695 		private ClassProxy(final ReturnValue[] returns) {
696 			this.returns = Arrays.asList(returns);
697 		}
698 
699 		@Override
700 		public Object invoke(@SuppressWarnings("unused") @Nullable final Object object,
701 				@Nullable final Method method,
702 				@SuppressWarnings("unused") @Nullable final Object[] args) throws Throwable {
703 			return returns.stream()
704 					.filter(r -> r.getMethod().equals(method))
705 					.map(ReturnValue::getReturnValue)
706 					.findFirst()
707 					.orElse(null);
708 		}
709 	}

712 		public static ReturnValue find(final Class<?> clazz, final String methodName, final Object returnValue) {
713 			return new ReturnValue(
714 					Arrays.stream(clazz.getMethods()).filter(m -> m.getName().equals(methodName)).findFirst().get(),
715 					returnValue);
716 		}
