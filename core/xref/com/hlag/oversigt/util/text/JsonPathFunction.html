3   import java.util.Collection;
4   import java.util.Objects;
5   import java.util.function.Function;
6   
7   import com.hlag.oversigt.util.JsonUtils;
8   import com.jayway.jsonpath.InvalidJsonException;
9   import com.jayway.jsonpath.JsonPath;
10  
11  import edu.umd.cs.findbugs.annotations.Nullable;
12  
13  class JsonPathFunction implements Function<String, String> {
14  	private final String probablyJson;
15  
16  	JsonPathFunction(final String probablyJson) {
17  		this.probablyJson = probablyJson;
18  	}
19  
20  	@Nullable
21  	@Override
22  	public String apply(@Nullable final String jsonPathString) {
23  		Objects.requireNonNull(jsonPathString, "Input for a JsonPath must be non null");
24  		final JsonPath jsonPath = JsonPath.compile(jsonPathString);
25  
26  		final Object result;
27  		try {
28  			result = JsonUtils.extractValueUsingJsonPath(jsonPath, probablyJson);
29  		} catch (@SuppressWarnings("unused") final InvalidJsonException e) {
30  			return null;
31  		}
32  
33  		if (result == null) {
34  			return "";
35  		} else if (result instanceof Collection && ((Collection<?>) result).size() == 1) {
36  			return ((Collection<?>) result).iterator().next().toString();
37  		} else {
38  			return result.toString();
39  		}
40  	}
41  }
