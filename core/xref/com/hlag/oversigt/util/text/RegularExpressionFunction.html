3   import java.util.Objects;
4   import java.util.function.Function;
5   import java.util.regex.Matcher;
6   import java.util.regex.Pattern;
7   
8   import com.google.common.base.Strings;
9   
10  import edu.umd.cs.findbugs.annotations.Nullable;
11  
12  class RegularExpressionFunction implements Function<String, String> {
13  	private final String value;
14  
15  	RegularExpressionFunction(final String value) {
16  		this.value = value;
17  	}
18  
19  	@Override
20  	public String apply(@Nullable final String regex) {
21  		Objects.requireNonNull(regex, "Input for a Regex must be non null");
22  		final Matcher matcher = Pattern.compile(regex).matcher(value); // TODO: Cache pattern statically
23  
24  		// Return empty string if not found
25  		if (!matcher.find()) {
26  			return "";
27  		}
28  
29  		// Return either the group "return" or the matched string
30  		String result;
31  		try {
32  			result = matcher.group("return");
33  		} catch (@SuppressWarnings("unused") final IllegalArgumentException e) {
34  			result = matcher.group();
35  		}
36  
37  		// Making sure to differentiate between no result and empty result.
38  		return Strings.isNullOrEmpty(result) ? " " : result;
39  	}
40  }
