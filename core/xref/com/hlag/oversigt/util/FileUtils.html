55  	public static void deleteFolder(final Path root) {
56  		try {
57  			LOGGER.info("Deleting folder [{}]", root.toAbsolutePath().toString());
58  			Files.walk(root)
59  					.sorted(Comparator.reverseOrder())
60  					.forEach(ThrowingConsumer.throwing(Files::deleteIfExists));
61  		} catch (final IOException e) {
62  			throw new SneakyException(e);
63  		}
64  	}

66  	public static Path getPath(final URI uri) {
67  		if (Ascii.equalsIgnoreCase(uri.getScheme(), "jar")) {
68  			final String uriString = uri.toString();
69  			final List<String> jarPathParts = Splitter.on('!').limit(2).splitToList(uriString);
70  			if (jarPathParts.size() == 2) {
71  				return getFileSystem(URI.create(jarPathParts.get(0))).getPath(jarPathParts.get(1));
72  			}
73  			throw new RuntimeException("Unable to interpret path: " + uri);
74  		}
75  		return Paths.get(uri);
76  	}

102 	private static Stream<Path> streamResources(final Path classpathEntry) {
103 		if (!Files.isRegularFile(classpathEntry)) {
104 			return streamResourcesFromDirectory(classpathEntry);
105 		}
106 		final Optional<String> extension = getExtension(classpathEntry);
107 		if (extension.isPresent()) {
108 			switch (Ascii.toLowerCase(extension.get())) {
109 			case "jar":
110 			case "zip":
111 				return listResourcesFromJar(classpathEntry).stream();
112 			default:
113 				// nothing
114 			}
115 		}
116 		throw new RuntimeException("Unable to handle file for resource scanning: " + classpathEntry);
117 	}

119 	@SuppressWarnings("resource")
120 	private static List<Path> listResourcesFromJar(final Path zip) {
121 		final String uriString = "jar:" + zip.toUri().toString();
122 
123 		final FileSystem fileSystem = getFileSystem(URI.create(uriString));
124 		final LinkedList<Path> paths = new LinkedList<>();
125 		try (JarInputStream jarInputStream = new JarInputStream(Files.newInputStream(zip))) {
126 			JarEntry entry = null;
127 			while ((entry = jarInputStream.getNextJarEntry()) != null) {
128 				paths.add(fileSystem.getPath(entry.getName()));
129 			}
130 
131 			// take the JAR entry...
132 			Optional.of(jarInputStream)
133 					// ... and extract the manifest entry...
134 					.map(JarInputStream::getManifest)
135 					// ... read its main attributes ...
136 					.map(Manifest::getMainAttributes)
137 					// ... take the ClassPath entry ...
138 					.map(attributes -> attributes.getValue("Class-Path"))
139 					// ... split it on whitespaces. Then you have a list of new relative class path
140 					// entries.
141 					.map(classpath -> Splitter.on(CharMatcher.whitespace()).omitEmptyStrings().splitToList(classpath))
142 					// take them, resolve the corresponding files and recursivly call this method
143 					.map(jarClasspathEntries -> jarClasspathEntries.stream()
144 							.map(zip.toAbsolutePath().getParent()::resolve)
145 							.filter(Files::exists)
146 							.map(FileUtils::listResourcesFromJar)
147 							.flatMap(Collection::stream)
148 							.collect(toList()))
149 					.ifPresent(paths::addAll);
150 		} catch (final IOException e) {
151 			throw new UncheckedIOException(e);
152 		}
153 
154 		return paths;
155 	}

179 	public static Optional<String> getExtension(final String filename) {
180 		final int lastIndex = filename.lastIndexOf('.');
181 		if (lastIndex < 0) {
182 			return Optional.empty();
183 		}
184 		return Optional.of(filename.substring(lastIndex + 1));
185 	}

202 		if (i < glob.length()) {
203 			return glob.charAt(i);

208 	/**
209 	 * Creates a regex pattern from the given glob expression.
210 	 *
211 	 * @param globPattern the glob to turn into a regex
212 	 * @param isDos       whether to use DOS/ windows specific syntax or not
213 	 * @return the converted regex
214 	 * @throws PatternSyntaxException if the given syntax has errors
215 	 * @see <a href=
216 	 *      "https://github.com/rtyley/globs-for-java/blob/master/src/main/java/com/madgag/globs/openjdk/Globs.java">OpenJDK</a>
217 	 */
218 	@SuppressWarnings({ "checkstyle:DescendantToken", "checkstyle:InnerAssignment" })
219 	@SuppressFBWarnings(value = "CC_CYCLOMATIC_COMPLEXITY", justification = "copied from OpenJDK as is")
220 	private static String toRegexPattern(final String globPattern, final boolean isDos) {
221 		boolean inGroup = false;
222 		// final StringBuilder regex = new StringBuilder("^");
223 		final StringBuilder regex = new StringBuilder();
224 
225 		int i = 0;
226 		while (i < globPattern.length()) {
227 			char c = globPattern.charAt(i++);
228 			switch (c) {
229 			case '\\':
230 				// escape special characters
231 				if (i == globPattern.length()) {
232 					throw new PatternSyntaxException("No character to escape", globPattern, i - 1);
233 				}
234 				final char next = globPattern.charAt(i++);
235 				if (isGlobMeta(next) || isRegexMeta(next)) {
236 					regex.append('\\');
237 				}
238 				regex.append(next);
239 				break;
240 			case '/':
241 				if (isDos) {
242 					regex.append("\\\\");
243 				} else {
244 					regex.append(c);
245 				}
246 				break;
247 			case '[':
248 				// don't match name separator in class
249 				if (isDos) {
250 					regex.append("[[^\\\\]&&[");
251 				} else {
252 					regex.append("[[^/]&&[");
253 				}
254 				if (next(globPattern, i) == '^') {
255 					// escape the regex negation char if it appears
256 					regex.append("\\^");
257 					i += 1;
258 				} else {
259 					// negation
260 					if (next(globPattern, i) == '!') {
261 						regex.append('^');
262 						i += 1;
263 					}
264 					// hyphen allowed at start
265 					if (next(globPattern, i) == '-') {
266 						regex.append('-');
267 						i += 1;
268 					}
269 				}
270 				boolean hasRangeStart = false;
271 				char last = 0;
272 				while (i < globPattern.length()) {
273 					c = globPattern.charAt(i++);
274 					if (c == ']') {
275 						break;
276 					}
277 					if (c == '/' || isDos && c == '\\') {
278 						throw new PatternSyntaxException("Explicit 'name separator' in class", globPattern, i - 1);
279 					}
280 					// TBD: how to specify ']' in a class?
281 					if (c == '\\' || c == '[' || c == '&' && next(globPattern, i) == '&') {
282 						// escape '\', '[' or "&&" for regex class
283 						regex.append('\\');
284 					}
285 					regex.append(c);
286 
287 					if (c == '-') {
288 						if (!hasRangeStart) {
289 							throw new PatternSyntaxException("Invalid range", globPattern, i - 1);
290 						}
291 						if ((c = next(globPattern, i++)) == EOL || c == ']') {
292 							break;
293 						}
294 						if (c < last) {
295 							throw new PatternSyntaxException("Invalid range", globPattern, i - 3);
296 						}
297 						regex.append(c);
298 						hasRangeStart = false;
299 					} else {
300 						hasRangeStart = true;
301 						last = c;
302 					}
303 				}
304 				if (c != ']') {
305 					throw new PatternSyntaxException("Missing ']", globPattern, i - 1);
306 				}
307 				regex.append("]]");
308 				break;
309 			case '{':
310 				if (inGroup) {
311 					throw new PatternSyntaxException("Cannot nest groups", globPattern, i - 1);
312 				}
313 				regex.append("(?:(?:");
314 				inGroup = true;
315 				break;
316 			case '}':
317 				if (inGroup) {
318 					regex.append("))");
319 					inGroup = false;
320 				} else {
321 					regex.append('}');
322 				}
323 				break;
324 			case ',':
325 				if (inGroup) {
326 					regex.append(")|(?:");
327 				} else {
328 					regex.append(',');
329 				}
330 				break;
331 			case '*':
332 				if (next(globPattern, i) == '*') {
333 					// crosses directory boundaries
334 					regex.append(".*");
335 					i += 1;
336 				} else {
337 					// within directory boundary
338 					if (isDos) {
339 						regex.append("[^\\\\]*");
340 					} else {
341 						regex.append("[^/]*");
342 					}
343 				}
344 				break;
345 			case '?':
346 				if (isDos) {
347 					regex.append("[^\\\\]");
348 				} else {
349 					regex.append("[^/]");
350 				}
351 				break;
352 			default:
353 				if (isRegexMeta(c)) {
354 					regex.append('\\');
355 				}
356 				regex.append(c);
357 				break;
358 			}
359 		}
360 
361 		if (inGroup) {
362 			throw new PatternSyntaxException("Missing '}", globPattern, i - 1);
363 		}
364 
365 		return regex.append('$').toString();
366 	}
