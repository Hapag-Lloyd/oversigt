1   package com.hlag.oversigt.util;
2   
3   import java.util.Optional;
4   import java.util.concurrent.Callable;
5   import java.util.concurrent.ExecutionException;
6   import java.util.concurrent.atomic.AtomicReference;
7   
8   /**
9    * https://stackoverflow.com/questions/20087173/how-to-do-a-lazy-create-and-set-with-atomicreference-in-a-safe-and-efficient-man
10   *
11   * @param <V> the type to hold in this class
12   */
13  public class LazyInitializedReference<V> {
14  	private final AtomicReference<V> cachedValue = new AtomicReference<>();
15  
16  	private final Callable<V> callable;
17  
18  	public LazyInitializedReference(final Callable<V> callable) {
19  		this.callable = callable;
20  	}
21  
22  	public synchronized V get() {
23  		try {
24  			if (cachedValue.get() == null) {
25  				cachedValue.set(callable.call());
26  			}
27  			return cachedValue.get();
28  		} catch (final ExecutionException e) {
29  			/* Re-set the reference. */
30  			cachedValue.set(null);
31  
32  			final Throwable cause = e.getCause();
33  			if (cause instanceof Error) {
34  				throw (Error) cause;
35  			}
36  			throw toUnchecked(cause);
37  		} catch (final InterruptedException e) {
38  			/* Re-set the reference. */
39  			cachedValue.set(null);
40  
41  			/* It's the client's responsibility to check the cause. */
42  			throw new RuntimeException("Interrupted", e);
43  		} catch (final Exception e) {
44  			throw toUnchecked(e);
45  		}
46  	}
47  
48  	public Optional<V> peek() {
49  		return Optional.ofNullable(cachedValue.get());
50  	}
51  
52  	public V reset() {
53  		return cachedValue.getAndSet(null);
54  	}
55  
56  	private static RuntimeException toUnchecked(final Throwable t) {
57  		return t instanceof RuntimeException ? (RuntimeException) t : new RuntimeException("Unchecked", t);
58  	}
59  }
