73  		try {

96  	 * Remove keys from the JSON string
97  	 *
98  	 * @param json   the JSON to work on
99  	 * @param filter determine whether a key should stay in the JSON or not. If the
100 	 *               {@link Predicate} returns <code>false</code> the key will be
101 	 *               removed.
102 	 * @return the stripped JSON
103 	 */
104 	public static String removeKeysFromJson(final String json, final Predicate<String> filter) {
105 		return toJson(removeKeys(Objects.requireNonNull(fromJson(json, Object.class)),
106 				filter,
107 				obj -> obj instanceof Double ? ((Double) obj).longValue() : obj));
108 	}

126 	@SuppressWarnings("unchecked")
127 	private static Object removeKeys(final Object object,
128 			final Predicate<String> filter,
129 			final Function<Object, Object> customConverter) {
130 		if (object instanceof Collection) {
131 			removeKeys((Collection<?>) object, filter, customConverter);
132 		} else if (object instanceof Map) {
133 			removeKeys((Map<String, Object>) object, filter, customConverter);
134 		}
135 		return customConverter.apply(object);
136 	}

138 	private static void removeKeys(final Collection<?> collection,
139 			final Predicate<String> filter,
140 			final Function<Object, Object> customConverter) {
141 		for (final Object e : collection) {
142 			removeKeys(e, filter, customConverter);
143 		}
144 	}

146 	private static void removeKeys(final Map<String, Object> map,
147 			final Predicate<String> filter,
148 			final Function<Object, Object> customConverter) {
149 		for (final Entry<String, Object> e : map.entrySet()) {
150 			if (!filter.test(e.getKey())) {
151 				e.setValue(null);
152 			} else {
153 				e.setValue(removeKeys(e.getValue(), filter, customConverter));
154 			}
155 		}
156 	}

162 	@SuppressWarnings("unchecked")
163 	public static String toJsonSchema(final Class<?> clazz, final Optional<JsonHint> hint) {
164 		LOGGER.debug("Creating JSONSchema for class: " + clazz.getName());
165 		final Map<String, Object> schema;
166 		if (SerializableProperty.class.isAssignableFrom(clazz)) {
167 			// TODO move this conversion to another class
168 			schema = Maps.builder(new LinkedHashMap<String, Object>())
169 					.put("$schema", "http://json-schema.org/schema#")
170 					.put("$id", "http://schema.hlag.com/oversigt/property/" + clazz.getName())
171 					.put("title", clazz.getSimpleName())
172 					.get();
173 			schema.putAll(toJsonSchemaFromProperty((Class<? extends SerializableProperty>) clazz));
174 		} else {
175 			schema = Maps.builder(new LinkedHashMap<String, Object>())
176 					.put("$schema", "http://json-schema.org/schema#")
177 					.put("$id", "http://schema.hlag.com/oversigt/type/" + clazz.getName())
178 					.put("title", clazz.getSimpleName())
179 					.get();
180 			schema.putAll(toJsonSchemaFromType(clazz, hint));
181 		}
182 		return toJson(schema);
183 	}

185 	private static Map<String, Object> toJsonSchemaFromProperty(final Class<? extends SerializableProperty> clazz) {
186 		final List<? extends SerializableProperty> propertyValues
187 				= Objects.requireNonNull(storage).listProperties(clazz);
188 		final List<String> names = new ArrayList<>(
189 				propertyValues.stream().map(SerializableProperty::getName).collect(Collectors.toList()));
190 		final List<Integer> ids = new ArrayList<>(
191 				propertyValues.stream().map(SerializableProperty::getId).collect(Collectors.toList()));
192 		final List<Map<String, Object>> mapping = propertyValues.stream()
193 				.map(p -> Maps.builder(new LinkedHashMap<String, Object>()) //
194 						.put("value", p.getId())
195 						.put("title", p.getName())
196 						.get())
197 				.collect(Collectors.toList());
198 		try {
199 			if (clazz.getDeclaredField("EMPTY") != null) {
200 				names.add(0, "\u00a0");
201 				ids.add(0, 0);
202 				mapping.add(0,
203 						Maps.builder(new LinkedHashMap<String, Object>()) //
204 								.put("value", 0)
205 								.put("title", "\u00a0")
206 								.get());
207 			}
208 		} catch (@SuppressWarnings("unused") final NoSuchFieldException | SecurityException ignore) {
209 			// continue if EMPTY is not found
210 		}
211 		return Maps.builder(new LinkedHashMap<String, Object>())
212 				.put("type", SERIALIZABLE_PROPERTY_TYPE)
213 				.put("uniqueItems", true)
214 				.put("enum", ids)
215 				.put("oversigt-ids", ids)
216 				.put("enumSource",
217 						Arrays.asList(Maps.builder(new LinkedHashMap<String, Object>())
218 								.put("title", "{{item.title}}")
219 								.put("value", "{{item.value}}")
220 								.put("source", mapping)
221 								.get()))
222 				.get();
223 	}

225 	private static Map<String, Object> toJsonSchemaFromType(final Type type, final Optional<JsonHint> hint) {
226 		if (type instanceof Class) {
227 			Optional<JsonHint> jsonHint = hint;
228 			if (((Class<?>) type).isAnnotationPresent(JsonHint.class)) {
229 				jsonHint = Optional.of(((Class<?>) type).getAnnotation(JsonHint.class));
230 			}
231 			return toJsonSchemaFromClass((Class<?>) type, jsonHint);
232 		} else if (type instanceof ParameterizedType) {
233 			return toJsonSchemaFromType(((ParameterizedType) type).getActualTypeArguments()[0], hint);
234 		} else if (type instanceof TypeVariable) {
235 			// TypeVariable<?> tv = (TypeVariable<?>) type;
236 			// AnnotatedType[] at = tv.getAnnotatedBounds();
237 			// Type[] b = tv.getBounds();
238 			// GenericDeclaration d = tv.getGenericDeclaration();
239 			return emptyMap();
240 		} else {
241 			throw new RuntimeException("Unknown type: " + type);
242 		}
243 	}

245 	@SuppressWarnings("unchecked")
246 	private static Map<String, Object> toJsonSchemaFromClass(final Class<?> clazz, final Optional<JsonHint> hint) {
247 		Optional<JsonHint> jsonHint = hint;
248 		if (clazz == String.class) {
249 			return Maps.builder(new LinkedHashMap<String, Object>()) //
250 					.put("type", "string")
251 					.get();
252 		} else if (clazz == boolean.class || clazz == Boolean.class) {
253 			return Maps.builder(new LinkedHashMap<String, Object>()) //
254 					.put("type", "boolean")
255 					.get();
256 		} else if (clazz == int.class || clazz == Integer.class) {
257 			return makeNumber(true, Optional.of(Integer.MIN_VALUE), Optional.of(Integer.MAX_VALUE));
258 		} else if (clazz == long.class || clazz == Long.class) {
259 			return makeNumber(true, Optional.of(Long.MIN_VALUE), Optional.of(Long.MAX_VALUE));
260 		} else if (clazz == double.class || clazz == Double.class) {
261 			return makeNumber(true, Optional.of(Double.MIN_VALUE), Optional.of(Double.MAX_VALUE));
262 		} else if (clazz == float.class || clazz == Float.class) {
263 			return makeNumber(true, Optional.of(Float.MIN_VALUE), Optional.of(Float.MAX_VALUE));
264 		} else if (clazz == byte.class || clazz == Byte.class) {
265 			return makeNumber(true, Optional.of(Byte.MIN_VALUE), Optional.of(Byte.MAX_VALUE));
266 		} else if (clazz == short.class || clazz == Short.class) {
267 			return makeNumber(true, Optional.of(Short.MIN_VALUE), Optional.of(Short.MAX_VALUE));
268 		} else if (clazz == char.class || clazz == Character.class) {
269 			return makeNumber(true, Optional.of(Character.MIN_VALUE), Optional.of(Character.MAX_VALUE));
270 		} else if (clazz == BigDecimal.class) {
271 			return makeNumber(false, Optional.empty(), Optional.empty());
272 		} else if (Enum.class.isAssignableFrom(clazz)) {
273 			final List<Map<String, Object>> abc = Arrays.stream(clazz.getEnumConstants())
274 					.map(e -> Maps.builder(new LinkedHashMap<String, Object>())
275 							.put("value", ((Enum<?>) e).name())
276 							.put("title", e.toString())
277 							.get())
278 					.collect(Collectors.toList());
279 			return Maps.builder(new LinkedHashMap<String, Object>())
280 					.put("type", "string")
281 					.put("uniqueItems", true) // TODO check whether this should always be true
282 					.put("oversigt-ids", getEnumConstants((Class<Enum<?>>) clazz))
283 					.put("enumSource",
284 							Arrays.asList(Maps.builder(new LinkedHashMap<String, Object>())
285 									.put("title", "{{item.title}}")
286 									.put("value", "{{item.value}}")
287 									.put("source", abc)
288 									.get()))
289 					.get();
290 		} else if (SerializableProperty.class.isAssignableFrom(clazz)) {
291 			return Maps.builder(new LinkedHashMap<String, Object>())
292 					.put("type", SERIALIZABLE_PROPERTY_TYPE)
293 					.put("$ref", "/schema/" + clazz.getName())
294 					.put("oversigt-property", clazz.getName())
295 					.put("$class", clazz.getName())
296 					.put("serializable-property", clazz.getSimpleName())
297 					.get();
298 		} else if (clazz.isArray() || Collection.class.isAssignableFrom(clazz)) {
299 			final Type componentType = clazz.isArray() ? clazz.getComponentType() : clazz.getGenericInterfaces()[0];
300 			final Map<String, Object> map = Maps.builder(new LinkedHashMap<String, Object>())
301 					.put("type", "array")
302 					.put("items", toJsonSchemaFromType(componentType, Optional.empty()))
303 					.put("additionalItems", false)
304 					.get();
305 			if (Set.class.isAssignableFrom(clazz)) {
306 				map.put("unique", true);
307 			}
308 			if (componentType instanceof Class && ((Class<?>) componentType).isAnnotationPresent(JsonHint.class)) {
309 				jsonHint = Optional.of(((Class<?>) componentType).getAnnotation(JsonHint.class));
310 			}
311 			if (jsonHint.isPresent() && jsonHint.get().arrayStyle().value().isPresent()) {
312 				map.put("format", jsonHint.map(JsonHint::arrayStyle).flatMap(ArrayStyle::value).get());
313 			} else {
314 				final Map<String, Object> items = (Map<String, Object>) map.get("items");
315 				if (items.containsKey("properties") && ((Map<?, ?>) items.get("properties")).size() <= 3) {
316 					map.put("format", "table");
317 				}
318 			}
319 			return map;
320 		} else if (clazz == Color.class) {
321 			return Maps.builder(new LinkedHashMap<String, Object>()) //
322 					.put("type", "string")
323 					.put("format", "color")
324 					.get();
325 		} else if (clazz == LocalDate.class) {
326 			return Maps.builder(new LinkedHashMap<String, Object>()) //
327 					.put("type", "string")
328 					.put("format", "date")
329 					.get();
330 		} else if (clazz == LocalTime.class) {
331 			return Maps.builder(new LinkedHashMap<String, Object>()) //
332 					.put("type", "string")
333 					.put("format", "time")
334 					.get();
335 		} else {
336 			// TODO check for notnull ???
337 			final List<Field> fields = TypeUtils.streamFields(clazz)
338 					.filter(f -> !Modifier.isTransient(f.getModifiers()))
339 					.filter(f -> !Modifier.isStatic(f.getModifiers()))
340 					.collect(Collectors.toList());
341 			final Map<String, Map<String, Object>> fieldsMap = new LinkedHashMap<>();
342 			for (final Field field : fields) {
343 				Optional<JsonHint> fieldHint = Optional.empty();
344 				if (field.isAnnotationPresent(JsonHint.class)) {
345 					fieldHint = Optional.of(field.getAnnotation(JsonHint.class));
346 				}
347 				final Map<String, Object> map = JsonUtils.toJsonSchemaFromClass(field.getType(), fieldHint);
348 				map.put("title", makeFirstCharacterCapital(field.getName()));
349 				getFormat(field).ifPresent(format -> map.put("format", format));
350 				fieldsMap.put(field.getName(), map);
351 			}
352 			final Map<String, Object> map = Maps.builder(new LinkedHashMap<String, Object>())
353 					.put("type", "object")
354 					.put("title", makeFirstCharacterCapital(clazz.getSimpleName()))
355 					.put("properties", fieldsMap)
356 					.put("required", fields.stream().filter(JsonUtils::isRequired).map(Field::getName).toArray())
357 					.put("additionalProperties", false)
358 					.get();
359 
360 			if (clazz.isAnnotationPresent(JsonHint.class)) {
361 				jsonHint = Optional.of(clazz.getAnnotation(JsonHint.class));
362 			}
363 			if (jsonHint.isPresent() && !Strings.isNullOrEmpty(jsonHint.get().headerTemplate())) {
364 				map.put("headerTemplate", jsonHint.get().headerTemplate());
365 			}
366 			return map;
367 		}
368 	}

370 	private static boolean isRequired(final Field field) {
371 		if (field.isAnnotationPresent(javax.annotation.Nullable.class)) {
372 			return false;
373 		} else if (field.isAnnotationPresent(com.fasterxml.jackson.annotation.JsonProperty.class)) {
374 			return field.getAnnotation(com.fasterxml.jackson.annotation.JsonProperty.class).required();
375 		} else {
376 			return true;
377 		}
378 	}

380 	private static Map<String, Object> makeNumber(final boolean wholeNumbers,
381 			final Optional<Object> min,
382 			final Optional<Object> max) {
383 		final Map<String, Object> map = Maps.builder(new LinkedHashMap<String, Object>()) //
384 				.put("type", "number")
385 				.get();
386 		min.ifPresent(v -> map.put("minimum", v));
387 		max.ifPresent(v -> map.put("maximum", v));
388 		if (wholeNumbers) {
389 			map.put("multipleOf", 1.0);
390 		}
391 		return map;
392 	}

398 	private static <E extends Enum<?>> String[] getEnumConstants(final Class<E> clazz) {
399 		return Stream.of((Object[]) clazz.getEnumConstants())

405 	private static Optional<String> getFormat(final Field field) {
406 		final String name = field.getName().toLowerCase();
407 		if (name.contains("password")) {
408 			return Optional.of("password");
409 		} else if (name.contains("mail")) {
410 			return Optional.of("email");
411 		} else if (name.endsWith("hostname")) {
412 			return Optional.of("hostname");
413 		} else if (name.endsWith("url")) {
414 			return Optional.of("url");
415 		}
416 		return Optional.empty();
417 	}
