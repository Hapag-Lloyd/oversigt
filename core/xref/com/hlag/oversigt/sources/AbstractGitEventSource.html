104 	private synchronized Stream<RevCommit> streamLog() throws GitAPIException, IOException {
105 		try (StartedAction action = getStatisticsCollector().startAction("GIT fetch", getRepositoryUrl())) {
106 			fetch();
107 		}
108 		return StreamSupport.stream(getGit().log().all().call().spliterator(), true)
109 				.filter(isCommitAfter(getEarliestPointToTakeIntoAccount()));
110 	}

129 	protected <T> Predicate<T> between(final Function<T, Instant> getTimeFunction,
130 			final Instant from,
131 			final Optional<Instant> to) {
132 		return r -> {
133 			final Instant time = getTimeFunction.apply(r);
134 			return to.map(time::isBefore).orElse(true) && time.isAfter(from);
135 		};
136 	}

138 	private synchronized void fetch() {
139 		try {
140 			getLogger().info("Pulling updates of GIT repository [{}]",
141 					getGit().getRepository().getDirectory().getAbsolutePath());
142 			withCredentials(getGit().fetch()).call();
143 			final PullResult pullResult = withCredentials(getGit().pull()).call();
144 			getLogger().info("Fetched from [{}]. Result [{}]", pullResult.getFetchedFrom(), pullResult.isSuccessful());
145 		} catch (final GitAPIException e) {
146 			getLogger().error(String.format("Cannot PULL repository [%s]",
147 					getGit().getRepository().getDirectory().getAbsolutePath()), e);
148 		}
149 	}

151 	@SuppressFBWarnings(value = "IMPROPER_UNICODE",
152 			justification = "comparing lower cased string with ASCII characters only")
153 	private Git createGitRepository() throws IOException, GitAPIException {
154 		if (getLogger().isDebugEnabled()) {
155 			getLogger().debug("Creating Git object for: " + getRepositoryUrl());
156 		}
157 		if (getRepositoryUrl().contains(":")) {
158 			final String protocol = getRepositoryUrl().substring(0, getRepositoryUrl().indexOf(":")).toLowerCase();
159 			switch (protocol) {
160 			case "http":
161 			case "https":
162 			case "git":
163 				return createRemoteGitRepository(getRepositoryUrl());
164 			case "file":
165 				return createLocalGitRepository(getRepositoryUrl());
166 			default:
167 			}
168 		}
169 		throw new RuntimeException("Cannot recognize url: " + getRepositoryUrl());
170 	}

180 	private Git createRemoteGitRepository(final String repositoryUrl) throws GitAPIException, IOException {
181 		getLogger().info("Creating remote repository [{}] ", repositoryUrl);
182 		// Create temp-directory
183 		final Path tempDir = Files.createTempDirectory("oversigt-temp-git");
184 		FileUtils.deleteFolderOnExit(tempDir);
185 
186 		// Clone repo to temp dir
187 		getLogger().debug("Start cloning");
188 		final CloneCommand cloneCommand = Git.cloneRepository();
189 		cloneCommand.setDirectory(tempDir.toFile());
190 		cloneCommand.setURI(repositoryUrl);
191 		cloneCommand.setCredentialsProvider(createCredentialsProvider(tempDir));
192 
193 		// Create repo
194 		final Git git = cloneCommand.call();
195 		getLogger().debug("Cloning done.");
196 		return git;
197 	}

213 	private final class GitCredentialsProvider extends CredentialsProvider {
214 		private final Path gitPath;
215 
216 		private GitCredentialsProvider(final Path gitPath) {
217 			this.gitPath = gitPath;
218 		}
219 
220 		@Override
221 		public boolean supports(@Nullable final CredentialItem... items) {
222 			if (items == null) {
223 				return false;
224 			}
225 			return isUsernamePassword(items) || isSkipSslTrust(items);
226 		}
227 
228 		@Override
229 		public boolean isInteractive() {
230 			return false;
231 		}
232 
233 		@Override
234 		public boolean get(@SuppressWarnings("unused") @Nullable final URIish uri,
235 				@Nullable final CredentialItem... items) throws UnsupportedCredentialItem {
236 			if (items == null) {
237 				return false;
238 			}
239 			if (isUsernamePassword(items)) {
240 				for (final CredentialItem item : items) {
241 					if (item instanceof Username) {
242 						((Username) item).setValue(getCredentials().getUsername());
243 					} else if (item instanceof Password) {
244 						((Password) item).setValue(getCredentials().getPassword().toCharArray());
245 					}
246 				}
247 				return true;
248 			} else if (!isCheckSSL() && isSkipSslTrust(items)) {
249 				// JGitText.get().sslTrustForNow
250 				final String message
251 						= MessageFormat.format(JGitText.get().sslTrustForRepo, gitPath.toAbsolutePath().toString());
252 				for (final CredentialItem item : items) {
253 					if (item instanceof YesNoType && item.getPromptText().equals(message)) {
254 						((YesNoType) item).setValue(true);
255 					}
256 				}
257 				return true;
258 			}
259 			return false;
260 		}
261 
262 		private boolean isSkipSslTrust(final CredentialItem[] items) {
263 			return items.length == 4
264 					&& items[0] instanceof InformationalMessage
265 					&& items[1] instanceof YesNoType
266 					&& items[2] instanceof YesNoType
267 					&& items[3] instanceof YesNoType
268 					&& items[0].getPromptText().endsWith(JGitText.get().sslFailureTrustExplanation);
269 		}
270 
271 		private boolean isUsernamePassword(final CredentialItem[] items) {
272 			return items.length == 2
273 					&& (items[0] instanceof Username && items[1] instanceof Password
274 							|| items[0] instanceof Password && items[1] instanceof Username);
275 		}
276 	}
