12

14  
15  	public PieChartEvent() {
16  		// no fields to be initialized
17  	}
18  
19  	public void addData(final String label, final double value, final String color) {
20  		addData(label, value, color, color);
21  	}
22  
23  	public void addData(final String label,
24  			final double value,
25  			final String backgroundColor,
26  			final String hoverBackgroundColor) {
27  		labels.add(label);
28  		getDataset().add(value, backgroundColor, hoverBackgroundColor);
29  	}
30  
31  	private Dataset getDataset() {

33  	}

37  
38  		private final Collection<String> backgroundColor = new ArrayList<>();
39  
40  		private final Collection<String> hoverBackgroundColor = new ArrayList<>();
41  
42  		private Dataset() {
43  			// no fields to be initialized
44  		}
45  
46  		private void add(final double value, final String backgroundColor, final String hoverBackgroundColor) {
47  			data.add(value);
48  			this.backgroundColor.add(backgroundColor);
49  			this.hoverBackgroundColor.add(hoverBackgroundColor);
50  		}
51  	}
