22  @EventSource(view = "List", displayName = "Microsoft Exchange Tasks Viewer", hiddenDataItems = { "moreinfo" })
23  public class ExchangeTasksEventSource extends AbstractExchangeEventSource<TwoColumnListEvent<String>> {
24  	private boolean showDateColumn = true;
25  
26  	public ExchangeTasksEventSource() {
27  		// no fields to be initialized
28  	}
29  
30  	@Property(name = "Show Date Column",
31  			description = "Determines whether the widget should show the date column or not")
32  	public boolean isShowDateColumn() {
33  		return showDateColumn;
34  	}
35  
36  	public void setShowDateColumn(final boolean showDateColumn) {
37  		this.showDateColumn = showDateColumn;
38  	}
39  
40  	@Override
41  	protected Optional<TwoColumnListEvent<String>> produceExchangeEvent() throws Exception {
42  		final List<Task> tasks;
43  		try (StartedAction action = getStatisticsCollector().startAction("Exchange read tasks", "")) {
44  			tasks = getExchangeClient().loadTasks();
45  		}
46  		final TwoColumnListEvent<String> event = createEvent(tasks);
47  		return Optional.of(event);
48  	}
49  
50  	private TwoColumnListEvent<String> createEvent(final List<Task> tasks) {
51  		return new TwoColumnListEvent<>(tasks.stream().map(t -> createItem(t)).collect(Collectors.toList()));
52  	}
53  
54  	private ListEventItem<String> createItem(final Task task) {
55  		try {
56  			String subject = task.getSubject();
57  			String labelStyle = "";
58  			String valueStyle = "";
59  			for (final Style style : Style.values()) {
60  				final String searchTerm = "[" + style.name() + "]";
61  				if (subject.contains(searchTerm)) {
62  					subject = subject.replace(searchTerm, "");
63  					labelStyle += style.getLabelStyle();
64  					valueStyle += style.getValueStyle();
65  				}
66  			}
67  
68  			switch (task.getImportance()) {
69  			case Low:
70  				labelStyle += "opacity: 0.75;";
71  				break;
72  			case High:
73  				labelStyle += "font-weight: bold;";
74  				break;
75  			case Normal:
76  			default:
77  				break;
78  			}
79  
80  			String createdAtStr = null;
81  			if (isShowDateColumn()) {
82  				final ZonedDateTime createdAt = task.getDateTimeCreated().toInstant().atZone(getZoneId());
83  				if (createdAt.toLocalDate().isBefore(LocalDate.now(getZoneId()))) {
84  					createdAtStr
85  							= createdAt.toLocalDate().format(DateTimeFormatter.ofPattern("dd.MM.yyyy", Locale.GERMANY));
86  				} else {
87  					createdAtStr = createdAt.format(DateTimeFormatter.ofPattern("HH:mm", Locale.GERMANY));
88  				}
89  			}
90  
91  			return new ListEventItem<>(subject, createdAtStr, labelStyle, valueStyle);
92  		} catch (final Exception e) {
93  			throw new RuntimeException("Unable to read task details", e);
94  		}
95  	}
96  
97  	private enum Style {
98  		green("color: green;", ""),
99  		yellow("color: yellow;", ""),
100 		red("color: red;", "");
101 
102 		private final String labelStyle;
103 
104 		private final String valueStyle;
105 
106 		Style(final String labelStyle, final String valueStyle) {
107 			this.labelStyle = labelStyle;
108 			this.valueStyle = valueStyle;
109 		}
110 
111 		public String getLabelStyle() {
112 			return labelStyle;
113 		}
114 
115 		public String getValueStyle() {
116 			return valueStyle;
117 		}
118 	}
119 }
