57  	@Override
58  	protected Optional<JenkinsPipelineStatusEvent> produceEvent() throws EventSourceException {
59  		final List<JenkinsPipelineEventStatusItem> list = new ArrayList<>();
60  
61  		// Requests pipeline status
62  		for (final Pipeline pipeline : pipelines) {
63  			for (final Branch branch : pipeline.branch) {
64  				list.addAll(getJenkinsDataByID(pipeline, branch, branch.times));
65  			}
66  		}
67  
68  		// Compare Pipelines
69  		final Comparator<JenkinsPipelineEventStatusItem> comparator
70  				= Comparator.comparing(JenkinsPipelineEventStatusItem::getStatus)
71  						.thenComparing(JenkinsPipelineEventStatusItem::getPipeline);
72  		list.sort(comparator);
73  
74  		return Optional.of(new JenkinsPipelineStatusEvent(list));
75  	}

77  	@SuppressWarnings("resource")
78  	public List<JenkinsPipelineEventStatusItem> getJenkinsDataByID(final Pipeline pipe,
79  			final Branch branch,
80  			final int times) throws EventSourceException {
81  		try (JenkinsServer jenkins = new JenkinsServer(new JenkinsHttpClient(getJenkinsUri(),
82  				createHttpClientBuilder(),
83  				credentials.getUsername(),
84  				credentials.getPassword()))) {
85  			final Map<String, Job> jobs = jenkins.getJobs();
86  			// request Pipeline information
87  			final JobWithDetails job = jobs.get(pipe.pipelineName).details();
88  			final List<JenkinsPipelineEventStatusItem> returns = new ArrayList<>();
89  			for (final Build buildfromjob : job.getBuilds()) {
90  				if (buildfromjob.details().getDescription() != null
91  						&& buildfromjob.details().getDescription().contains(branch.branchName)) {
92  					final BuildResult status = buildfromjob.details().getResult();
93  					if (status != null) {
94  						String userID = "";
95  						String ticket = "";
96  						final Matcher m = Pattern.compile(regEx).matcher(buildfromjob.details().getDescription());
97  						m.reset(buildfromjob.details().getDescription());
98  						if (m.find()) {
99  							// Use group 2 and 3 from RegEx
100 							userID = m.group(2);
101 							ticket = m.group(3);
102 						}
103 						if (status.equals(BuildResult.UNKNOWN)) {
104 							returns.add(
105 									new JenkinsPipelineEventStatusItem(pipe.pipelineName.replace("_", " ").substring(4),
106 											branch.branchName,
107 											ticket,
108 											userID,
109 											Integer.toString(buildfromjob.getNumber()),
110 											"No Build available",
111 											"font-weight: bold",
112 											"",
113 											"font-style: italic",
114 											"font-style: italic",
115 											"font-size: 80%",
116 											""));
117 						} else if (status.equals(BuildResult.SUCCESS)) {
118 							returns.add(
119 									new JenkinsPipelineEventStatusItem(pipe.pipelineName.replace("_", " ").substring(4),
120 											branch.branchName,
121 											ticket,
122 											userID,
123 											Integer.toString(buildfromjob.getNumber()),
124 											"Build successful",
125 											"font-weight: bold",
126 											"",
127 											"font-style: italic",
128 											"font-style: italic",
129 											"font-size: 80%",
130 											""));
131 						} else if (status.equals(BuildResult.UNSTABLE)) {
132 							returns.add(
133 									new JenkinsPipelineEventStatusItem(pipe.pipelineName.replace("_", " ").substring(4),
134 											branch.branchName,
135 											ticket,
136 											userID,
137 											Integer.toString(buildfromjob.getNumber()),
138 											"Build unstable",
139 											"font-weight: bold",
140 											"",
141 											"font-style: italic",
142 											"font-style: italic",
143 											"font-size: 80%",
144 											""));
145 						} else if (status.equals(BuildResult.FAILURE)) {
146 							returns.add(
147 									new JenkinsPipelineEventStatusItem(pipe.pipelineName.replace("_", " ").substring(4),
148 											branch.branchName,
149 											ticket,
150 											userID,
151 											Integer.toString(buildfromjob.getNumber()),
152 											"Build failed",
153 											"font-weight: bold",
154 											"",
155 											"font-style: italic",
156 											"font-style: italic",
157 											"font-size: 80%",
158 											"color:red"));
159 						} else if (status.equals(BuildResult.NOT_BUILT)) {
160 							returns.add(
161 									new JenkinsPipelineEventStatusItem(pipe.pipelineName.replace("_", " ").substring(4),
162 											branch.branchName,
163 											ticket,
164 											userID,
165 											Integer.toString(buildfromjob.getNumber()),
166 											"Has never run",
167 											"font-weight: bold",
168 											"",
169 											"font-style: italic",
170 											"font-style: italic",
171 											"font-size: 80%",
172 											""));
173 						} else if (status.equals(BuildResult.BUILDING)) {
174 							returns.add(
175 									new JenkinsPipelineEventStatusItem(pipe.pipelineName.replace("_", " ").substring(4),
176 											branch.branchName,
177 											ticket,
178 											userID,
179 											Integer.toString(buildfromjob.getNumber()),
180 											"Is currently building",
181 											"font-weight: bold",
182 											"",
183 											"font-style: italic",
184 											"font-style: italic",
185 											"font-size: 80%",
186 											""));
187 						} else if (status.equals(BuildResult.ABORTED)) {
188 							returns.add(
189 									new JenkinsPipelineEventStatusItem(pipe.pipelineName.replace("_", " ").substring(4),
190 											branch.branchName,
191 											ticket,
192 											userID,
193 											Integer.toString(buildfromjob.getNumber()),
194 											"Build aborted",
195 											"font-weight: bold",
196 											"",
197 											"font-style: italic",
198 											"font-style: italic",
199 											"font-size: 80%",
200 											""));
201 						} else {
202 							returns.add(
203 									new JenkinsPipelineEventStatusItem(pipe.pipelineName.replace("_", " ").substring(4),
204 											branch.branchName,
205 											ticket,
206 											userID,
207 											Integer.toString(buildfromjob.getNumber()),
208 											"no status available",
209 											"font-weight: bold",
210 											"",
211 											"font-style: italic",
212 											"font-style: italic",
213 											"font-size: 80%",
214 											""));
215 						}
216 						if (returns.size() == times) {
217 							return returns;
218 						}
219 					}
220 				}
221 			}
222 
223 			return returns;
224 		} catch (final IOException e) {
225 			throw new EventSourceException("Failed accessing Jenkins for " + pipe.pipelineName, e);
226 		}
227 	}

229 	private HttpClientBuilder createHttpClientBuilder() {
230 		final HttpClientBuilder builder = HttpClientBuilder.create();
231 
232 		final HttpProxy proxy = getHttpProxy();
233 		if (proxy != HttpProxy.EMPTY) {
234 			builder.setProxy(new HttpHost(proxy.getHostname(), proxy.getPort()));
235 		}
236 
237 		return builder;
238 	}
