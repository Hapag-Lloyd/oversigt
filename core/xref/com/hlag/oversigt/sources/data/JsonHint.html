12  	String headerTemplate() default "";
13  
14  	ArrayStyle arrayStyle() default ArrayStyle.DEFAULT;
15  
16  	enum ArrayStyle {
17  		DEFAULT,
18  		GRID,
19  		TABLE("table"),
20  		TABS("tabs");
21  
22  		private final Optional<String> value;
23  
24  		ArrayStyle() {
25  			value = Optional.empty();
26  		}
27  
28  		ArrayStyle(final String value) {
29  			this.value = Optional.of(value);
30  		}
31  
32  		public Optional<String> value() {
33  			return value;
34  		}
35  	}
36  }
