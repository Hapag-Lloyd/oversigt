58  	@Override
59  	protected Optional<TextEvent> produceEvent() {
60  		logTrace(getLogger(), "Starting event creation");
61  
62  		final String body;
63  		try {
64  			body = downloadString(createConfiguredConnection());
65  			logDebug(getLogger(), "Downloaded body");
66  			logTrace(getLogger(), "Body content %s", body);
67  		} catch (final IOException e) {
68  			logError(getLogger(), "Unable to download content: %s", e.getMessage());
69  			throw new RuntimeException(e);
70  		}
71  
72  		final List<?> elements = extractElements(body);
73  		logDebug(getLogger(), "Extracted text: %s", elements.toString());
74  
75  		final String output;
76  		if (elements.isEmpty()) {
77  			output = getDefaultValue();
78  		} else if (hasStringFormat()) {
79  			output = elements.stream()
80  					.map(JsonUtils::toJson)
81  					.map(this::useStringFormat)
82  					.collect(Collectors.joining("\n"));
83  		} else {
84  			output = getMultiResultBehaviour().extractElement(elements).toString();
85  		}
86  
87  		return Optional.of(new TextEvent(output));
88  	}

90  	private List<?> extractElements(final String json) {
91  		return getJsonPaths().stream()
92  				.map(jp -> extractElement(jp, json))
93  				.flatMap(Collection::stream)
94  				.collect(Collectors.toList());
95  	}
96  
97  	private Collection<?> extractElement(final JsonPath jsonPath, final String json) {
98  		final Object result = jsonPath.read(json, JSON_PATH_CONFIGURATION);
99  		if (getLogger().isDebugEnabled()) {
100 			getLogger().debug("Result: " + result);
101 		}
102 
103 		if (result == null) {
104 			return Collections.emptyList();
105 		} else if (result instanceof Collection) {
106 			return (Collection<?>) result;
107 		} else {
108 			return Arrays.asList(result);
109 		}
110 	}
111 
112 	private String extractSingleElement(final JsonPath jsonPath, final String json) {
113 		return extractElement(jsonPath, json).stream().findFirst().map(Object::toString).orElse("");
114 	}

175 	}

177 	private String useStringFormat(final String json) {
178 		@SuppressWarnings({ "checkstyle:IllegalInstantiation", "checkstyle:XIllegalTypeCustom" })
179 		final StringBuffer builder = new StringBuffer();
180 
181 		final Matcher matcher = JSONPATH_EXTRACTION_PATTERN.matcher(getStringFormat());
182 		while (matcher.find()) {
183 			final String pathString = matcher.group(1);
184 			final JsonPath path = JsonPath.compile(pathString);
185 			matcher.appendReplacement(builder, extractSingleElement(path, json));
186 		}
187 		matcher.appendTail(builder);
188 		return builder.toString();
189 	}
190 
191 	public enum MultiResultBehaviour {
192 		First(c -> c.iterator().next()),
193 		Last(c -> c.stream().skip(c.size() - 1).findFirst().get()),
194 		Random(c -> c.stream().skip((long) (Math.random() * c.size())).findFirst().get());
195 
196 		private final Function<Collection<?>, ?> extractor;
197 
198 		MultiResultBehaviour(final Function<Collection<?>, ?> extractor) {
199 			this.extractor = extractor;
200 		}
201 
202 		private Object extractElement(final Collection<?> objects) {
203 			return extractor.apply(objects);
204 		}
205 	}
