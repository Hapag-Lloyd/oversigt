1   package com.hlag.oversigt.sources;
2   
3   import java.io.IOException;
4   import java.util.Optional;
5   
6   import com.hlag.oversigt.core.eventsource.EventSource;
7   import com.hlag.oversigt.core.eventsource.EventSourceException;
8   import com.hlag.oversigt.sources.event.StyleEvent;
9   
10  @EventSource(view = "ImageView",
11  		displayName = "Internet Image",
12  		description = "Load an image from the internet by parsing a web site and show the image in the dashboard.",
13  		dataItems = { "width" },
14  		hiddenDataItems = { "more-info" })
15  public class InternetImageEventSource extends AbstractDownloadEventSource<StyleEvent> {
16  	public InternetImageEventSource() {
17  		// no fields to be initialized
18  	}
19  
20  	@Override
21  	protected Optional<StyleEvent> produceEvent() throws EventSourceException {
22  		try {
23  			final DownloadData image = downloadBytes(createConfiguredConnection());
24  			return Optional.of(StyleEvent.backgroundImage(image.getContentType(), image.getData()));
25  		} catch (final IOException e) {
26  			throw new EventSourceException("Unable to download image", e);
27  		}
28  	}
29  }
