31  @EventSource(displayName = "SQL Graph", view = "Rickshawgraph")
32  public class SqlGraphEventSource extends AbstractJdbcEventSource<ComplexGraphEvent> {
33  	private int historyLength = 10;
34  
35  	private String labelFormat = "";
36  
37  	private String sqlStatement = "";
38  
39  	private String[] titles = new String[0];
40  
41  	private int columnCount = 0;
42  
43  	private final SortedMap<ZonedDateTime, List<Long>> values = new TreeMap<>();
44  
45  	public SqlGraphEventSource() {
46  		// no fields to be initialized
47  	}

49  	@Override
50  	protected Optional<ComplexGraphEvent> produceEventFromData() {
51  		final ZonedDateTime now = ZonedDateTime.now(ZoneOffset.UTC);
52  
53  		// Remove too old history
54  		values.keySet().removeIf(now.minusMinutes(getHistoryLength())::isAfter);
55  
56  		// Calculate graph values
57  		final long secondsOffset = now.withHour(0).withMinute(0).withSecond(0).toEpochSecond();
58  		final List<Series> series = new ArrayList<>();
59  		IntStream.range(0, columnCount).forEach(column -> {
60  			final List<Point> points = new ArrayList<>();
61  
62  			if (!values.isEmpty()) {
63  				// Get maximum value of each series to normalize
64  				final OptionalLong optionalMaxValue = values.values().stream().mapToLong(v -> v.get(column)).max();
65  				final long maxValue = optionalMaxValue.getAsLong() == 0 ? 1 : optionalMaxValue.getAsLong();
66  
67  				points.addAll(values.entrySet()
68  						.stream()
69  						.map(e -> new Point(e.getKey().toEpochSecond() - secondsOffset,
70  								100 * e.getValue().get(column) / maxValue))
71  						.collect(Collectors.toList()));
72  			}
73  
74  			series.add(new Series(titles.length > column ? titles[column] : "", points));
75  		});
76  
77  		Collections.reverse(series);
78  
79  		final List<Long> displayValues = new ArrayList<>();
80  		if (!values.isEmpty()) {
81  			displayValues.addAll(values.get(values.lastKey()));
82  		}
83  		String text = "";
84  		try {
85  			text = String.format(labelFormat, displayValues.toArray());
86  		} catch (final IllegalFormatException e) {
87  			getLogger().warn("Unable to fill format: " + getLabelFormat(), e);
88  		}
89  		return Optional.of(new ComplexGraphEvent(series, text));
90  	}
91  
92  	@Override
93  	protected void gatherDatabaseInfo(final Connection connection) throws SQLException {
94  		final List<List<Long>> data = readFromDatabase(connection,
95  				rs -> IntStream.range(1, rs.getMetaData().getColumnCount() + 1)
96  						.mapToObj(ThrowingIntFunction.sneakyInt(rs::getLong))
97  						.collect(toList()),
98  				getSqlStatement());
99  
100 		if (!data.isEmpty()) {
101 			values.put(ZonedDateTime.now(ZoneOffset.UTC), data.get(0));
102 			columnCount = data.get(0).size();
103 			Utils.logInfo(getSpecialLogger(),
104 					"%s",
105 					(Supplier<String>) () -> IntStream.range(0, columnCount)
106 							.mapToObj((final int column) -> String
107 									.format("%s: %d", titles[column], data.get(0).get(column)))
108 							.collect(Collectors.joining(", ")));
109 		} else {
110 			getLogger().error("Failed with empty result set.");
111 		}
112 	}
