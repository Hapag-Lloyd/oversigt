84  	@Override
85  	protected Optional<TimelineEvent> produceExchangeEvent() {
86  		final TimelineEvent event = new TimelineEvent(maximumPointInFuture, getZoneId(), getLocale());
87  		final LocalDate now = LocalDate.now(getZoneId());
88  
89  		fillTimelineEvent(now, event);
90  
91  		event.removeEvents(getMaximumPointInFuture(), getMinimumEventCount());
92  
93  		// maybe add today
94  		if (!event.hasEventAt(now)) {
95  			event.addEvent(TODAY, now, now, false, colorToday);
96  		}
97  
98  		return Optional.of(event);
99  	}

101 	protected void fillTimelineEvent(final LocalDate now, final TimelineEvent event) {
102 		addBirthdays(event, now);
103 		addHolidays(event, now);
104 		addExchangeCalendar(event, now);
105 	}

107 	private void addExchangeCalendar(final TimelineEvent event, final LocalDate now) {
108 		final ZonedDateTime from = now.minusMonths(5).atStartOfDay(getZoneId());
109 		final ZonedDateTime until = now.plus(getMaximumPointInFuture())
110 				.plus(getMaximumPointInFuture())
111 				.plusDays(1)
112 				.atStartOfDay(getZoneId());
113 		final List<Appointment> appointments;
114 		try (StartedAction action
115 				= getStatisticsCollector().startAction("Exchange read appointments", from + " - " + until)) {
116 			appointments = getExchangeClient().loadAppointments(from, until);
117 		}
118 		for (final Appointment appointment : appointments) {
119 			addAppointment(event, appointment);
120 		}
121 	}

123 	private void addAppointment(final TimelineEvent event, final Appointment appointment) {
124 		try {
125 			final ZonedDateTime start = appointment.getStart().toInstant().atZone(getZoneId());
126 			final ZonedDateTime end = appointment.getEnd().toInstant().atZone(getZoneId());
127 			final LocalDate now = LocalDate.now(getZoneId());
128 
129 			// for all day appointments, write "ganzt√§gig"
130 			final String duration;
131 
132 			if (appointment.getIsAllDayEvent()) {
133 				final LocalDate endDate = end.toLocalDate().minus(1, ChronoUnit.DAYS);
134 				final boolean endSameDay = endDate.isEqual(start.toLocalDate());
135 				final boolean endsToday = endDate.isEqual(now);
136 
137 				if (endSameDay || endsToday) {
138 					duration = ALL_DAY;
139 				} else {
140 					duration = ALL_DAY_UNTIL + endDate.format(DateTimeFormatter.ofPattern("d. MMM", getLocale()));
141 				}
142 			} else {
143 				// not an all day event
144 				final boolean startInPast = start.toLocalDate().isBefore(now);
145 
146 				if (startInPast) {
147 					duration = UNTIL + end.format(DateTimeFormatter.ofPattern("d. MMM, HH:mm", getLocale()));
148 				} else {
149 					// begins today or in future: show start time (date is always shown)
150 					duration = start.format(DateTimeFormatter.ofPattern("HH:mm", getLocale()));
151 				}
152 			}
153 
154 			final String title = String.format("%s (%s)", appointment.getSubject(), duration);
155 
156 			// for out of office and mailbox handling use special colors
157 			Color color = colorAppointment;
158 
159 			if (LegacyFreeBusyStatus.OOF.equals(appointment.getLegacyFreeBusyStatus())) {
160 				color = colorOutOfOffice;
161 			} else if (appointment.getSubject() != null && appointment.getSubject().startsWith("Mailbox")) {
162 				// TODO Farben auf regel-basis erstellen, nicht hart-kodiert
163 				color = colorMailbox;
164 			}
165 
166 			event.addEvent(title,
167 					appointment.getStart().toInstant().atZone(getZoneId()).toLocalDate(),
168 					appointment.getEnd().toInstant().atZone(getZoneId()).toLocalDate(),
169 					appointment.getIsAllDayEvent(),
170 					color);
171 		} catch (@SuppressWarnings("unused") final ServiceLocalException ignore) {
172 			// in case of exception ignore event
173 		}
174 	}

176 	private void addBirthdays(final TimelineEvent event, final LocalDate now) {
177 		for (final Birthday birthday : getBirthdays()) {
178 			LocalDate date = birthday.getDate().withYear(now.getYear());
179 			if (date.isBefore(now)) {
180 				date = date.plusYears(1);
181 			}
182 			event.addEvent(birthday.getName() + " (" + birthday.getDate().until(date).getYears() + ")",
183 					date,
184 					date,
185 					true,
186 					colorBirthday);
187 		}
188 	}

190 	private void addHolidays(final TimelineEvent event, final LocalDate now) {
191 		final Color fontColor = Color.LIGHT_GRAY;
192 
193 		final HolidayManager m = HolidayManager.getInstance(ManagerParameters.create(getHolidayCalendar()));
194 		final String area = Strings.emptyToNull(getHolidayArea());
195 		final Set<Holiday> holidays;
196 		if (area != null) {
197 			holidays = m.getHolidays(now,
198 					now.plus(getMaximumPointInFuture()).plus(getMaximumPointInFuture()),
199 					new String[] { area });
200 		} else {
201 			holidays = m.getHolidays(now, now.plus(getMaximumPointInFuture()).plus(getMaximumPointInFuture()));
202 		}
203 		holidays.forEach(h -> event.addEvent(getCorrectHolidayName(
204 				h.getDescription(getLocale())), h.getDate(), h.getDate(), true, colorHoliday, fontColor));
205 	}

207 	private String getCorrectHolidayName(final String apiName) {
208 		return Arrays.stream(getHolidayNameCorrections())
209 				.filter(c -> c.apiName.equals(apiName))
210 				.findAny()
211 				.map(c -> c.correctName)
212 				.orElse(apiName);
213 	}
