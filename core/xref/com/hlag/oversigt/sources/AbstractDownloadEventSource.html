79  	private URLConnection createConnection(final String urlString,
80  			final Optional<String> cookie,
81  			final Collection<LoginData> loginData) throws IOException {
82  		final URL url = new URL(urlString);
83  		final URLConnection con = url.openConnection(getHttpProxy().getProxy());
84  		con.setConnectTimeout(30 * 1000);
85  		con.setReadTimeout(30 * 1000);
86  		con.setRequestProperty("User-Agent",
87  				"Mozilla/5.0 Java/" + System.getProperty("java.version") + " Oversigt/0.9");
88  
89  		if (con instanceof HttpsURLConnection && !isCheckSSL()) {
90  			((HttpsURLConnection) con).setSSLSocketFactory(SSLUtils.getNonCheckingSSLSocketFactory(getEventId()));
91  			((HttpsURLConnection) con).setHostnameVerifier((hostname, session) -> true);
92  		}
93  
94  		if (con instanceof HttpURLConnection) {
95  			for (final HttpHeader httpHeader : getHttpHeaders()) {
96  				con.setRequestProperty(httpHeader.getName(), httpHeader.getValue());
97  			}
98  		}
99  
100 		if (cookie.isPresent()) {
101 			con.setRequestProperty("Cookie", cookie.get());
102 		} else if (loginData.isEmpty() && getCredentials() != Credentials.EMPTY) {
103 			final String encoded = Base64.getEncoder()
104 					.encodeToString((getCredentials().getUsername() + ":" + getCredentials().getPassword())
105 							.getBytes(StandardCharsets.UTF_8));
106 			con.setRequestProperty("Authorization", "Basic " + encoded);
107 		}
108 
109 		if (con instanceof HttpURLConnection && !loginData.isEmpty()) {
110 			final HttpURLConnection hcon = (HttpURLConnection) con;
111 			hcon.setDoOutput(true);
112 			hcon.setInstanceFollowRedirects(false);
113 			hcon.setRequestMethod("POST");
114 			hcon.setRequestProperty("Content-Type", "application/x-www-form-urlencoded");
115 			hcon.setUseCaches(false);
116 			try (BufferedWriter writer = new BufferedWriter(new OutputStreamWriter(hcon.getOutputStream()))) {
117 				final List<String> parts = new ArrayList<>();
118 				for (final LoginData data : loginData) {
119 					parts.add(data.getName());
120 					parts.add(data.getValue()
121 							.replace("${domain}", getCredentials().getDomain())
122 							.replace("${username}", getCredentials().getUsername())
123 							.replace("${password}", getCredentials().getPassword()));
124 				}
125 				final String string = createXWwwUrlEncoded(parts.toArray(new String[0]));
126 				writer.append(string);
127 			}
128 		}
129 
130 		return con;
131 	}

141 	protected URLConnection createConnection(final List<InternetAddress> addresses) throws IOException {
142 		final TextProcessor textProcessor = createTextProcessor();
143 		final List<List<String>> replacements = new ArrayList<>();
144 		Optional<URLConnection> connection = Optional.empty();
145 		Optional<String> cookie = Optional.empty();
146 
147 		for (int i = 0; i < addresses.size(); i += 1) {
148 			final InternetAddress internetAddress = addresses.get(i);
149 			if (connection.isPresent()) {
150 				logDebug(getLogger(), "Reading cookie from old connection");
151 				final String newCookie = connection.get().getHeaderField("Set-Cookie");
152 				logTrace(getLogger(), "New Cookie: %s", cookie);
153 				if (newCookie != null) {
154 					cookie = Optional.ofNullable(newCookie);
155 				}
156 
157 				logDebug(getLogger(), "Closing connection");
158 				connection.get().getInputStream().close();
159 				connection = Optional.empty();
160 			}
161 
162 			String url = internetAddress.getUrlString();
163 			logTrace(getLogger(), "Matching URL: %s", url);
164 			final Matcher repmat = PATTERN_URL_MATCHER_REPLACEMENT.matcher(url);
165 			while (repmat.find()) {
166 				final int major = Integer.parseInt(repmat.group(1));
167 				final int minor = Integer.parseInt(repmat.group(2));
168 				if (replacements.size() < major || replacements.get(major - 1).size() < minor) {
169 					throw new RuntimeException("Unable to find replacement ${" + major + "." + minor + "}");
170 				}
171 				if (replacements.get(major - 1) != null && replacements.get(major - 1).get(minor - 1) != null) {
172 					final String search = "${" + major + "." + minor + "}";
173 					final String replacement = replacements.get(major - 1).get(minor - 1);
174 					url = url.replace(search, replacement);
175 					logTrace(getLogger(), "Replacing %s with %s to %s", search, replacement, url);
176 				}
177 			}
178 			logDebug(getLogger(), "Connecting to URL: %s", url);
179 			if (url != null) {
180 				url = textProcessor.process(url);
181 				connection = Optional.of(createConnection(url, cookie, Arrays.asList(internetAddress.getLoginDatas())));
182 				final Optional<Pattern> pattern = internetAddress.getPattern();
183 				if (pattern.isPresent()) {
184 					final List<String> reps = new ArrayList<>();
185 					logDebug(getLogger(), "Downloading content for matcher: %s", pattern.toString());
186 					final String content = downloadString(connection.get());
187 					final Matcher matcher = pattern.get().matcher(content);
188 					if (matcher.find()) {
189 						for (int j = 1; j <= matcher.groupCount(); j += 1) {
190 							reps.add(matcher.group(j));
191 						}
192 					}
193 					logDebug(getLogger(), "Adding replacements: %s", (Supplier<String>) () -> reps.toString());
194 					replacements.add(reps);
195 
196 					if (i == addresses.size() - 1 && replacements.size() == 1) {
197 						if (reps.isEmpty()) {
198 							throw new RuntimeException("Pattern did not match. Unable to call next URL.");
199 						}
200 						addresses.add(new InternetAddress(reps.get(0)));
201 					}
202 				}
203 			}
204 		}
205 		return connection.orElseThrow(() -> new RuntimeException("Unable to create URLConnection"));
206 	}

224 	private <R> R read(final URLConnection inputConnection,
225 			final ThrowingBiFunction<URLConnection, InputStream, R> inputStreamConsumer) throws IOException {
226 		final URLConnection connectionToRead = handleRedirects(inputConnection);
227 		try (StartedAction action
228 				= getStatisticsCollector().startAction("Download", inputConnection.getURL().toExternalForm());
229 				InputStream in = connectionToRead.getInputStream()) {
230 			return inputStreamConsumer.apply(connectionToRead, in);
231 		} catch (final IOException e) {
232 			throw e;
233 		} catch (final Exception e) {
234 			throw new RuntimeException("Error while reading data from connection.", e);
235 		}
236 	}

238 	private URLConnection handleRedirects(final URLConnection connection) throws IOException {
239 		if (connection instanceof HttpURLConnection) {
240 			return handleRedirects((HttpURLConnection) connection);
241 		}
242 		logWarn(getLogger(),
243 				"Encountered URLConnection of type [%s]. Unable to handle redirect.",
244 				connection.getClass().getName());
245 		return connection;
246 	}

248 	private URLConnection handleRedirects(final HttpURLConnection connection) throws IOException {
249 		final Map<String, List<String>> headerParameters
250 				= Collections.unmodifiableMap(new LinkedHashMap<>(connection.getRequestProperties()));
251 
252 		final int status = connection.getResponseCode();
253 		if (status >= 200 && status < 300) /* status == HttpURLConnection.HTTP_OK */ {
254 			return connection;
255 		}
256 
257 		if (status == HttpURLConnection.HTTP_MOVED_PERM
258 				|| status == HttpURLConnection.HTTP_MOVED_TEMP
259 				|| status == HttpURLConnection.HTTP_SEE_OTHER
260 				|| status == 307 // temporary redirect
261 				|| status == 308 // permanent redirect // TODO safe new URL
262 		) {
263 			// create new connection new new URL
264 			final URI uri;
265 			try {
266 				uri = new URI(connection.getHeaderField("Location"));
267 			} catch (final URISyntaxException e) {
268 				throw new IOException("Invalid redirect", e);
269 			}
270 			final URL url;
271 			if (uri.isAbsolute()) {
272 				url = uri.toURL();
273 			} else {
274 				url = new URL(connection.getURL(), connection.getHeaderField("Location"));
275 			}
276 
277 			// set correct HTTP method
278 			final String method;
279 			final URLConnection newConnection = url.openConnection(getHttpProxy().getProxy());
280 			if (newConnection instanceof HttpURLConnection) {
281 				final HttpURLConnection httpConnection = (HttpURLConnection) newConnection;
282 				if (status == HttpURLConnection.HTTP_SEE_OTHER) {
283 					httpConnection.setRequestMethod("GET");
284 				} else {
285 					httpConnection.setRequestMethod(connection.getRequestMethod());
286 				}
287 				method = httpConnection.getRequestMethod();
288 			} else {
289 				method = null;
290 			}
291 
292 			// copy settings from old connection
293 			newConnection.setAllowUserInteraction(connection.getAllowUserInteraction());
294 			newConnection.setConnectTimeout(connection.getConnectTimeout());
295 			newConnection.setDefaultUseCaches(connection.getDefaultUseCaches());
296 			newConnection.setDoInput(connection.getDoInput());
297 			newConnection.setDoOutput(connection.getDoOutput());
298 			newConnection.setIfModifiedSince(connection.getIfModifiedSince());
299 			newConnection.setReadTimeout(connection.getReadTimeout());
300 			newConnection.setUseCaches(connection.getUseCaches());
301 			headerParameters.entrySet()
302 					.stream()
303 					.filter(e -> !Arrays.asList("Host", "Connection").contains(e.getKey()))
304 					.filter(e -> method == null || !e.getKey().startsWith(method + " "))
305 					.forEach(e -> e.getValue().forEach(v -> newConnection.addRequestProperty(e.getKey(), v)));
306 
307 			// done
308 			return handleRedirects(newConnection);
309 		}
310 
311 		throw new IOException("Return code " + status + " from URL: " + connection.getURL());
312 	}

342 	private static String createXWwwUrlEncoded(final String... strings) {
343 		final StringBuilder builder = new StringBuilder();
344 		for (int index = 0; index < strings.length; index += 2) {
345 			if (index > 0) {
346 				builder.append("&");
347 			}
348 			builder.append(strings[index]).append("=").append(StringConverters.encodeUrl(strings[index + 1]));
349 		}
350 		return builder.toString();
351 	}
