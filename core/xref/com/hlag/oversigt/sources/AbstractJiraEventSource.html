75  	protected Map<DisplayOption, Set<Issue>> getJiraTickets() throws EventSourceException {
76  		// List all issues as defined by JQL query
77  		final List<Issue> issues;
78  		try {
79  			final JiraClient client = JiraClientFactory.createJiraClient(getJiraConnection(), getJiraCredentials());
80  			try (StartedAction action = getStatisticsCollector().startAction("JIRA query", getQuery())) {
81  				issues = client.search(getQuery());
82  			}
83  		} catch (final JiraClientException | TimeoutException e) {
84  			return handleException(e);
85  		}
86  
87  		// Initialize entries per user-defined category to keep the user-defined order
88  		Map<DisplayOption, Set<Issue>> issuesPerCategory = new LinkedHashMap<>();
89  		for (final DisplayOption displayOption : getDisplayOptions()) {
90  			issuesPerCategory.put(displayOption, new HashSet<>());
91  		}
92  
93  		// Find known (categories) and unknown DisplayOptions per issue
94  		final Map<DisplayOption, Set<Issue>> issuesUnknown = new TreeMap<>();
95  		for (final Issue issue : issues) {
96  			boolean containedKnownDislayOption = false;
97  
98  			// Add issues to all known (categories) DisplayOptions,
99  			// following the user-defined option "showUnknownDisplayOptions" for unknown
100 			// DisplayOptions
101 			for (final DisplayOption displayOption : getDisplayOptions(issue)) {
102 				if (issuesPerCategory.containsKey(displayOption) && !Strings.isNullOrEmpty(displayOption.getValue())) {
103 					issuesPerCategory.get(displayOption).add(issue);
104 					containedKnownDislayOption = true;
105 				} else {
106 					final boolean addUnknownIssue = getShowUnknownCategories() == ShowUnknownCategories.ALL
107 							|| getShowUnknownCategories() == ShowUnknownCategories.UNKNOWN_ONLY
108 									&& !containedKnownDislayOption;
109 					if (addUnknownIssue) {
110 						if (!issuesUnknown.containsKey(displayOption)) {
111 							issuesUnknown.put(displayOption, new HashSet<>());
112 						}
113 						issuesUnknown.get(displayOption).add(issue);
114 					}
115 				}
116 			}
117 		}
118 
119 		// Add unknown categories afterwards to have them sorted first (using TreeMap)
120 		issuesPerCategory.putAll(issuesUnknown);
121 
122 		// If not getShowEmptyCategories then remove empty DisplayOption entries
123 		if (!getShowEmptyCategories()) {
124 			issuesPerCategory = issuesPerCategory.entrySet()
125 					.stream()
126 					.filter(e -> !e.getValue().isEmpty())
127 					.collect(de.larssh.utils.Collectors.toLinkedHashMap());
128 		}
129 
130 		return issuesPerCategory;
131 	}

188 	/**
189 	 * Returns all matching DisplayOptions of an issue The returned set is ordered
190 	 * by known categories first and unknown DisplayOptions last. For no further
191 	 * ordering is guaranteed.
192 	 */
193 	private Set<DisplayOption> getDisplayOptions(final Issue issue) {
194 		// Collect attribute values to match
195 		final Map<String, AggregationType> values = new TreeMap<>();
196 		for (final AggregationType aggregationType : getAggregationTypes()) {
197 			for (final String value : aggregationType.getAttributeValues(issue)) {
198 				values.put(value, aggregationType);
199 			}
200 		}
201 
202 		// Collect DisplayOptions per value
203 		final Set<DisplayOption> displayOptions = new LinkedHashSet<>();
204 		final Set<DisplayOption> displayOptionsUnknown = new TreeSet<>();
205 		for (final String value : values.keySet()) {
206 			boolean foundDisplayOption = false;
207 
208 			for (final DisplayOption displayOption : getDisplayOptions()) {
209 				if (AggregationType.matches(displayOption, value)) {
210 					displayOptions.add(displayOption);
211 					foundDisplayOption = true;
212 				}
213 			}
214 
215 			if (!foundDisplayOption) {
216 				displayOptionsUnknown.add(getUnknownDisplayOptions(values.get(value), issue, value));
217 			}
218 		}
219 
220 		displayOptions.addAll(displayOptionsUnknown);
221 		return displayOptions;
222 	}

259 	private <X> X handleException(final Exception original) throws EventSourceException {
260 		Throwable previous = null;
261 		Throwable throwable = original;
262 
263 		while (throwable != null && throwable != previous) {
264 			if (throwable instanceof TimeoutException) {
265 				throw new EventSourceException("JIRA request timed out while waiting for a free connection.");
266 			} else if (throwable instanceof JiraClientException) {
267 				throw new EventSourceException(throwable.getMessage(), throwable.getCause());
268 			} else if (throwable instanceof RestClientException) {
269 				final RestClientException rce = (RestClientException) throwable;
270 				if (rce.getStatusCode().isPresent()
271 						&& rce.getStatusCode().get() == 500
272 						&& rce.getCause() instanceof JSONException) {
273 					final JSONException je = (JSONException) rce.getCause();
274 					final String[] messageParts = HANDLE_AUTHENTICATION_EXCEPTION_PATTERN.split(je.getMessage(), 0);
275 					final List<String> auths = Arrays.stream(messageParts)
276 							.filter(ste -> ste.toLowerCase().contains("authenticator"))
277 							.collect(Collectors.toList());
278 
279 					if (contains(auths, "getUserFromBasicAuthentication") && contains(auths, "checkAuthenticated")) {
280 						if (getJiraCredentials() == Credentials.EMPTY) {
281 							throw new EventSourceException("No credentials configured.");
282 						}
283 						throw new EventSourceException(
284 								"Unable to log in to JIRA. Username: " + getJiraCredentials().getUsername());
285 					}
286 				}
287 			}
288 			if (Throwables.getRootCause(throwable) instanceof SocketTimeoutException) {
289 				throw new EventSourceException("JIRA not available: Timeout");
290 			}
291 			if (Throwables.getRootCause(throwable) instanceof JSONException) {
292 				throw new EventSourceException("JIRA not available: Invalid JSON returned from server");
293 			}
294 			previous = throwable;
295 			throwable = throwable.getCause();
296 		}
297 
298 		throw new RuntimeException(original);
299 	}

306 	public enum AggregationType {
307 		ASSIGNEE("Assignee",
308 				i -> i.getAssignee() == null ? null : i.getAssignee().getName(),
309 				i -> i.getAssignee() == null ? null : i.getAssignee().getDisplayName()),
310 		COMPONENTS("Components",
311 				i -> StreamSupport.stream(i.getComponents().spliterator(), false)
312 						.map(BasicComponent::getName)
313 						.collect(Collectors.toSet()),
314 				(i, v) -> v),
315 		ISSUE_TYPE("Issue Type", i -> i.getIssueType().getName()),
316 		PRIORITY("Priority", i -> i.getPriority().getName()),
317 		PROJECT("Project", i -> i.getProject().getName()),
318 		REPORTER("Reporter", i -> i.getReporter().getName(), i -> i.getReporter().getDisplayName()),
319 		RESPONSIBLE_PERSONS("Responsible Persons",
320 				i -> getUserNamesOfResponsiblePersons(i.getFieldByName("Responsible Persons")),
321 				(i, v) -> getDisplayNameOfResponsiblePerson(i.getFieldByName("Responsible Persons"), v)),
322 		STATUS("Status", i -> i.getStatus().getName()),
323 		WORKER("Worker",
324 				i -> i.getFieldByName("Worker").getValue() == null
325 						? null
326 						: ((User) i.getFieldByName("Worker").getValue()).getName(),
327 				i -> i.getFieldByName("Worker").getValue() == null
328 						? null
329 						: ((User) i.getFieldByName("Worker").getValue()).getDisplayName());

331 		private static String getDisplayNameOfResponsiblePerson(final IssueField field, final String name) {
332 			try {
333 				final JSONArray responsiblePersons = (JSONArray) field.getValue();
334 				for (int i = 0; i < responsiblePersons.length(); i += 1) {
335 					final JSONObject person = responsiblePersons.getJSONObject(i);
336 					if (person.getString("name").equals(name)) {
337 						return person.getString("displayName");
338 					}
339 				}
340 				return name;
341 			} catch (final JSONException e) {
342 				throw new RuntimeException("Unable to extract responsible person.", e);
343 			}
344 		}
345 
346 		private static List<String> getUserNamesOfResponsiblePersons(final IssueField field) {
347 			try {
348 				final List<String> names = new ArrayList<>();
349 				if (field.getValue() instanceof JSONArray) {
350 					final JSONArray responsiblePersons = (JSONArray) field.getValue();
351 					if (responsiblePersons != null) {
352 						for (int i = 0; i < responsiblePersons.length(); i += 1) {
353 							names.add(responsiblePersons.getJSONObject(i).getString("name"));
354 						}
355 					}
356 				}
357 				return names;
358 			} catch (final JSONException e) {
359 				throw new RuntimeException("Unable to extract responsible persons.", e);
360 			}
361 		}

374 		AggregationType(final String displayValue, final Function<Issue, String> attributeValueProducer) {
375 			this(displayValue, i -> {
376 				final String value = attributeValueProducer.apply(i);
377 				return value == null ? Collections.emptyList() : Arrays.asList(value);
378 			}, (i, v) -> v);
379 		}
380 
381 		AggregationType(final String displayValue,
382 				final Function<Issue, String> attributeValueProducer,
383 				final Function<Issue, String> displayValueProducer) {
384 			this(displayValue, i -> {
385 				final String value = attributeValueProducer.apply(i);
386 				return value == null ? Collections.emptyList() : Arrays.asList(value);
387 			}, (i, v) -> displayValueProducer.apply(i));
388 		}
389 
390 		AggregationType(final String displayValue,
391 				final Function<Issue, Collection<String>> attributeValuesProducer,
392 				final BiFunction<Issue, String, String> displayValueProducer) {
393 			this.attributeValuesProducer = attributeValuesProducer;
394 			this.displayValue = displayValue;
395 			this.displayValueProducer = displayValueProducer;
396 		}
