1   package com.hlag.oversigt.sources;
2   
3   import java.time.DayOfWeek;
4   import java.time.LocalDate;
5   import java.time.LocalTime;
6   import java.time.ZoneId;
7   import java.time.ZonedDateTime;
8   import java.time.temporal.ChronoUnit;
9   import java.util.Optional;
10  
11  import com.hlag.oversigt.core.eventsource.DataItemDefault;
12  import com.hlag.oversigt.core.eventsource.EventSource;
13  import com.hlag.oversigt.core.eventsource.ScheduledEventSource;
14  import com.hlag.oversigt.sources.event.MeterEvent;
15  
16  @SuppressWarnings("checkstyle:OperatorWrap")
17  @EventSource(view = "Meter",
18  		displayName = "Motivation",
19  		description = "Compute semi-random numbers depending on time of day and other things.",
20  		hiddenDataItems = { "height", "moreinfo", "updated-at-message", "width" },
21  		dataItemDefaults = {
22  				@DataItemDefault(name = "min", value = "0"),
23  				@DataItemDefault(name = "max", value = "100") })
24  public class MotivationEventSource extends ScheduledEventSource<MeterEvent> {
25  	private static final ZoneId HERE = ZoneId.of("Europe/Berlin");
26  
27  	private static final LocalTime START_ZEIT = LocalTime.of(7, 0);
28  
29  	private static final LocalTime STOP_ZEIT = LocalTime.of(17, 15);
30  
31  	public MotivationEventSource() {
32  		// no fields to be initialized
33  	}
34  
35  	@Override
36  	protected Optional<MeterEvent> produceEvent() {
37  		final ZonedDateTime now = ZonedDateTime.now(HERE);
38  		ZonedDateTime start = getStartingTime();
39  		ZonedDateTime stop = getQuittingTime();
40  
41  		double motivation;
42  
43  		// Consider time of day
44  		if (start.isBefore(now) && stop.isAfter(now)) {
45  			// Motivation goes down...
46  			final double minutesToWork = ChronoUnit.MINUTES.between(start, stop);
47  			final double minutesLeft = ChronoUnit.MINUTES.between(now, stop);
48  
49  			motivation = minutesLeft / minutesToWork;
50  		} else {
51  			// Motivation goes up...
52  			start = start.plusDays(1);
53  			stop = stop.plusDays(1);
54  			final double minutesToWork = ChronoUnit.MINUTES.between(stop, start);
55  			final double minutesLeft = ChronoUnit.MINUTES.between(now, start);
56  
57  			motivation = (minutesToWork - minutesLeft) / minutesToWork;
58  		}
59  
60  		// consider day of week
61  		final DayOfWeek dow = now.getDayOfWeek();
62  		final double offset = dow.ordinal() * 0.05;
63  		final double factor = 1.0 - DayOfWeek.values().length * 0.05;
64  		motivation = offset + motivation * factor;
65  
66  		return Optional.of(new MeterEvent((byte) (motivation * 100 + 0.5)));
67  	}
68  
69  	private static ZonedDateTime getStartingTime() {
70  		return START_ZEIT.atDate(LocalDate.now()).atZone(HERE);
71  	}
72  
73  	private static ZonedDateTime getQuittingTime() {
74  		return STOP_ZEIT.atDate(LocalDate.now()).atZone(HERE);
75  	}
76  }
