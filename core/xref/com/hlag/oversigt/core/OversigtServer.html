43  /**
44   * HTTP Server Controller. Bootstraps server and specifies all needed mappings
45   * and request handlers
46   *
47   * @author avarabyeu
48   * @author Olaf Neumann
49   */
50  @Singleton
51  public class OversigtServer extends AbstractIdleService {
52  	private static final Logger LOGGER = LoggerFactory.getLogger(OversigtServer.class);
53  
54  	@Inject
55  	@Named("listeners")
56  	private List<HttpListenerConfiguration> listeners;
57  
58  	@Inject
59  	private EventBus eventBus;
60  
61  	private Optional<Undertow> server = Optional.empty();
62  
63  	@Inject
64  	private EventSender sender;
65  
66  	@Inject
67  	private HttpHandlers handlers;
68  
69  	@Inject
70  	private EventSourceInstanceController instanceController;
71  
72  	@Inject
73  	private DashboardController dashboardController;
74  
75  	@Inject
76  	@Named("startEventSources")
77  	private boolean startEventSources;
78  
79  	@Inject
80  	@Named("NightlyDashboardReloader")
81  	private Service nightlyDashboardReloader;
82  
83  	@Inject
84  	@Named("NightlyEventSourceRestarter")
85  	private Service nightlyEventSourceRestarter;
86  
87  	@Inject
88  	@Named("foreignEvents.enabled")
89  	private boolean foreignEventsEnabled;
90  
91  	@Inject
92  	public OversigtServer() {
93  		addListener(new OversigtServerListener(), MoreExecutors.directExecutor());
94  	}
95  
96  	@Override
97  	protected void startUp() throws Exception {
98  		// Configure listeners
99  		if (listeners.isEmpty()) {
100 			throw new RuntimeException("No http listeners configured.");
101 		}
102 
103 		LOGGER.info("Initializing controllers");
104 		dashboardController.initialize();
105 
106 		LOGGER.info("Configuring web server");
107 		eventBus.register(new Consumer<OversigtEvent>() {
108 			@Subscribe
109 			@Override
110 			public void accept(@Nullable final OversigtEvent event) {
111 				if (event == null) {
112 					return;
113 				}
114 				handlers.getServerSentEventsHandler()
115 						.getConnections()
116 						.forEach(connection -> sender.sendEventToConnection(event, connection));
117 			}
118 		});
119 
120 		// Create Handlers for dynamic content
121 		final RoutingHandler routingHandler = Handlers.routing()
122 				// dashboard handling
123 				.get("/{dashboard}", handlers.createDashboardHandler())
124 				// send events to dashboards
125 				.get("/events", handlers.getServerSentEventsHandler())
126 				// get widget details
127 				.get("/views/{widget}", handlers.createWidgetHandler())
128 				// JSON Schema output
129 				.get("/schema/{class}", handlers.createJsonSchemaHandler())
130 				// default handler
131 				.get("/", handlers.createWelcomePageHandler());
132 		if (foreignEventsEnabled) {
133 			// get events from outside
134 			routingHandler.post("/events", handlers.createForeignEventHandler());
135 		}
136 
137 		// Create Handlers for static content
138 		final HttpHandler rootHandler = Handlers.path(routingHandler)
139 				.addPrefixPath("/assets", handlers.createAssetsHandler())
140 				.addPrefixPath("/compiled", handlers.createAggregationHandler())
141 				.addPrefixPath("/api/swagger", handlers.createSwaggerUiHandler())
142 				.addPrefixPath(HttpHandlers.MAPPING_API, handlers.createApiHandler())
143 				.addPrefixPath("/config", handlers.createAngularHandler());
144 
145 		// Create Handler for compressing content
146 		final EncodingHandler encodingHandler = new EncodingHandler(new ContentEncodingRepository()
147 				.addEncodingHandler("gzip", new GzipEncodingProvider(), 50, Predicates.requestLargerThan(5))
148 				.addEncodingHandler("deflate", new DeflateEncodingProvider(), 50, Predicates.requestLargerThan(10)))
149 						.setNext(rootHandler);
150 
151 		final Logger accessLogger = LoggerFactory.getLogger("access");
152 
153 		final AccessLogHandler accessHandler = new AccessLogHandler(encodingHandler,
154 				message -> accessLogger.info(message),
155 				"combined",
156 				OversigtServer.class.getClassLoader());
157 
158 		final Builder builder = Undertow.builder();
159 		listeners.stream()
160 				.filter(not(HttpListenerConfiguration::isTls))
161 				.forEach(c -> builder.addHttpListener(c.getPort(), c.getIp()));
162 		listeners.stream()
163 				.filter(HttpListenerConfiguration::isTls)
164 				.forEach(c -> builder.addHttpsListener(c.getPort(),
165 						c.getIp(),
166 						Objects.requireNonNull(c.getTLSConfiguration()).createSSLContext()));
167 		final Undertow server = builder.setHandler(accessHandler).build();
168 		this.server = Optional.of(server);
169 
170 		LOGGER.info("Starting web server");
171 		try {
172 			server.start();
173 		} catch (final Exception e) {
174 			if (e.getCause() instanceof BindException) {
175 				LOGGER.error("Cannot start server", e.getCause());
176 				stopAsync();
177 			}
178 		}
179 
180 		nightlyDashboardReloader.startAsync();
181 		nightlyEventSourceRestarter.startAsync();
182 		if (startEventSources) {
183 			LOGGER.info("Starting event source instances");
184 			instanceController.startAllInstances();
185 		}
186 
187 		LOGGER.info("StartUp finished");
188 	}
189 
190 	@Override
191 	protected void shutDown() throws Exception {
192 		/* stop the service for nightly reloading */
193 		LOGGER.info("Stopping nightly services");
194 		nightlyDashboardReloader.stopAsync();
195 		nightlyDashboardReloader.awaitTerminated();
196 		nightlyEventSourceRestarter.stopAsync();
197 		nightlyEventSourceRestarter.awaitTerminated();
198 
199 		/* stop all event source instances */
200 		LOGGER.info("Stopping event sources");
201 		instanceController.stopAllInstances();
202 
203 		/* close connections */
204 		LOGGER.info("Shutting down server sent event connections");
205 		handlers.getServerSentEventsHandler().getConnections().forEach(ServerSentEventConnection::shutdown);
206 
207 		/* stop the server */
208 		LOGGER.info("Stopping web server");
209 		server.ifPresent(Undertow::stop);
210 	}

212 	private final class OversigtServerListener extends Listener {
213 		private OversigtServerListener() {
214 			// no fields to be initialized
215 		}
216 
217 		@Override
218 		public void running() {
219 			final Undertow server = OversigtServer.this.server.get();
220 			if (server != null) {
221 				LOGGER.info("Embedded Oversigt server has started and is listening on port(s) {}",
222 						server.getListenerInfo()
223 								.stream()
224 								.map(li -> (InetSocketAddress) li.getAddress())
225 								.mapToInt(InetSocketAddress::getPort)
226 								.toArray());
227 			}
228 		}
229 
230 		@Override
231 		public void failed(@Nullable final State from, @Nullable final Throwable failure) {
232 			LOGGER.error("Embedded Oversigt server failed from: " + from, failure);
233 		}
234 
235 		@Override
236 		public void stopping(@SuppressWarnings("unused") @Nullable final State from) {
237 			LOGGER.info("Stopping embedded Oversigt server");
238 		}
239 	}
