34  public class OversigtConfigurationModule extends AbstractModule {
35  	private static final Logger LOGGER = LoggerFactory.getLogger(OversigtConfigurationModule.class);

37  	private static final TypeLiteral<Map<String, String>> MAP_OF_STRING_TO_STRING
38  			= new TypeLiteral<Map<String, String>>() {
39  				// just type literal for generics detection
40  			};
41  
42  	private static final TypeLiteral<List<String>> LIST_OF_STRINGS = new TypeLiteral<List<String>>() {
43  		// just type literal for generics detection
44  	};
45  
46  	private static final TypeLiteral<List<HttpListenerConfiguration>> LIST_OF_HTTP_LISTENER_CONFIGURATION
47  			= new TypeLiteral<List<HttpListenerConfiguration>>() {
48  				// just type literal for generics detection
49  			};

52  
53  	private final boolean debugFallback;
54  
55  	private final String ldapBindPasswordFallback;

57  	public OversigtConfigurationModule(final boolean debugFallback, final String ldapBindPasswordFallback) {
58  		config = OversigtConfiguration.readConfiguration();
59  		this.debugFallback = debugFallback;
60  		this.ldapBindPasswordFallback = ldapBindPasswordFallback;
61  	}
62  
63  	private ConstantBindingBuilder bind(final String name) {
64  		return bindConstant().annotatedWith(Names.named(name));
65  	}
66  
67  	@Override
68  	protected void configure() {
69  		// config itself
70  		bind(OversigtConfiguration.class).toInstance(config);
71  
72  		// debug
73  		bind("debug").to(config.isDebug() || debugFallback);
74  
75  		// template handling
76  		bind(CustomWroConfiguration.class).toInstance(config.getWro());
77  
78  		// ui settings
79  		bind("showOwnersInWelcomePage").to(config.getUi().isShowOwnersInWelcomePage());
80  
81  		// api settings
82  		bind("hostname").to(config.getHostname().orElseThrow(() -> new RuntimeException("No hostname specified")));
83  		bind(SignatureAlgorithm.class).toInstance(config.getApi().getJwtAlgorithm());
84  		bind("api.secret.base64")
85  				.to(Objects.requireNonNull(config.getApi().getJwtSecretBase64(), "api.jwtSecretBase64"));
86  		bind("api.ttl").to(config.getApi().getJwtTimeToLive());
87  		bind("rateLimit").to(config.getEventManager().getRateLimit());
88  		bind(Duration.class).annotatedWith(Names.named("discardEventsAfter"))
89  				.toInstance(config.getEventManager().getDiscardEventsAfter());
90  		bind("templateNumberFormat").to(config.getTemplateNumberFormat());
91  
92  		// database
93  		final DatabaseConfiguration database = config.getDatabase();
94  		bind("databaseLocation").to(database.getLocation());
95  		bind("databaseName").to(database.getName());
96  		bind("databaseUsername").to(database.getUsername());
97  		bind("databasePassword").to(database.getPassword());
98  		bind(SqlDialect.class).to(database.getSqlDialect());
99  
100 		// http server config
101 		bind(LIST_OF_HTTP_LISTENER_CONFIGURATION).annotatedWith(Names.named("listeners"))
102 				.toInstance(config.getListeners());
103 
104 		// foreign events
105 		bind("foreignEvents.enabled").to(config.getSecurity().getForeignEvents().isEnabled());
106 		bind("foreignEvents.needAuthentication").to(config.getSecurity().getForeignEvents().isNeedAuthentication());
107 		bind("foreignEvents.apiKeyHeaderName").to(config.getSecurity().getForeignEvents().getApiKeyHeaderName());
108 		bind(LIST_OF_STRINGS).annotatedWith(Names.named("foreignEvents.allowedApiKeys"))
109 				.toInstance(config.getSecurity().getForeignEvents().getAllowedApiKeys());
110 
111 		// event sources
112 		bindNamedStringArray("additionalPackages", config.getEventSources().getPackages());
113 		bindNamedStringArray("widgetsPaths", config.getEventSources().getWidgetsPaths());
114 		bindNamedArray(Path[].class, "addonFolders", Paths::get, config.getEventSources().getAddonFolders());
115 
116 		// Mail Settings
117 		bind("mailSenderHost").to(
118 				config.getMail().getHostname().orElseThrow(() -> new RuntimeException("No mail hostname configured.")));
119 		bind("mailSenderPort").to(config.getMail().getPort());
120 		bind("mailSenderStartTls").to(config.getMail().isStartTls());
121 		bind("mailSenderUsername").to(config.getMail().getUsername().orElse(""));
122 		bind("mailSenderPassword").to(config.getMail().getPassword().orElse(""));
123 		bind("mailSenderAddress").to(config.getMail().getSenderAddress());
124 
125 		// Security
126 		boolean boundAuthenticator = false;
127 		if (config.getSecurity().getLdap().isPresent()) {
128 			final LdapConfiguration ldapConfiguration = config.getSecurity().getLdap().get();
129 			if (!ldapConfiguration.isBindPasswordSet()) {
130 				ldapConfiguration.setBindPassword(ldapBindPasswordFallback);
131 			}
132 			if (ldapConfiguration.isBindPasswordSet()) {
133 				bind(LdapConfiguration.class).toInstance(ldapConfiguration);
134 				bind(Authenticator.class).to(LdapAuthenticator.class);
135 				boundAuthenticator = true;
136 			}
137 		}
138 		if (!boundAuthenticator && !config.getSecurity().getUsers().isEmpty()) {
139 			bind(MAP_OF_STRING_TO_STRING).annotatedWith(Names.named("UsernamesAndPasswords"))
140 					.toInstance(config.getSecurity().getUsers());
141 			bind(Authenticator.class).to(MapAuthenticator.class);
142 			boundAuthenticator = true;
143 		}
144 		final List<String> admins = config.getSecurity().getServerAdmins().orElseGet(ArrayList::new);
145 		bind(LIST_OF_STRINGS).annotatedWith(Names.named("serverAdmins")).toInstance(admins);
146 		if (admins.isEmpty()) {
147 			LOGGER.warn("No server admins configured. Please check configuration security.serverAdmins");
148 		}
149 
150 		// JIRA
151 		bind("jira.socketTimeout").to(config.getJira().getSocketTimeout());
152 	}

154 	private void bindNamedStringArray(final String name, final String[] inputs) {
155 		bindNamedArray(String[].class, name, Function.identity(), inputs);
156 	}
157 
158 	@SuppressWarnings("unchecked")
159 	private <T, I> void bindNamedArray(final Class<T[]> targetClass,
160 			final String name,
161 			final Function<I, T> converter,
162 			final I[] inputs) {
163 		bind(targetClass).annotatedWith(Names.named(name))
164 				.toInstance((T[]) Stream.of(inputs)
165 						.map(converter)
166 						.collect(Collectors.toList())
167 						.toArray(createArray(targetClass.getComponentType(), 0)));
168 	}
