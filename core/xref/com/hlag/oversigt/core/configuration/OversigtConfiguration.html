51  	private static ObjectMapper createObjectMapper() {
52  		final SimpleModule module = new SimpleModule("Oversigt-API");
53  		module.addSerializer(Duration.class, serializer(Duration.class, Duration::toString));
54  		module.addDeserializer(Duration.class, deserializer(Duration.class, Duration::parse));
55  
56  		final ObjectMapper objectMapper = new ObjectMapper();
57  		objectMapper.registerModule(module);
58  		objectMapper.registerModule(new Jdk8Module());
59  		objectMapper.configure(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS, false);
60  		objectMapper.setVisibility(PropertyAccessor.ALL, Visibility.NONE);
61  		objectMapper.setVisibility(PropertyAccessor.FIELD, Visibility.ANY);
62  		return objectMapper;
63  	}

65  	static OversigtConfiguration readConfiguration() {
66  		try {
67  			final URL configUrl = Resources.getResource(Oversigt.APPLICATION_CONFIG_FILE);
68  			Preconditions.checkState(configUrl != null,
69  					"Main application config [%s] not found",
70  					Oversigt.APPLICATION_CONFIG_FILE);
71  			LOGGER.info("Reading Oversigt configuration: " + configUrl);
72  			final String configString = Resources.toString(configUrl, Charsets.UTF_8);
73  			return createObjectMapper().readValue(configString, OversigtConfiguration.class);
74  		} catch (final IOException e) {
75  			throw new IllegalStateException("Unable to read configuration", e);
76  		}
77  	}
