1   package com.hlag.oversigt.core.eventsource;
2   
3   import java.time.Duration;
4   import java.time.LocalDateTime;
5   import java.util.concurrent.TimeUnit;
6   
7   import com.google.common.util.concurrent.AbstractScheduledService;
8   
9   /**
10   * Scheduled EventSource - produces events with specified time period.
11   * Basically, based on Guava's
12   * {@link com.google.common.util.concurrent.AbstractScheduledService}
13   *
14   * @author neumaol
15   */
16  public abstract class NightlyService extends AbstractScheduledService {
17  	protected NightlyService() {
18  		// no fields to be initialized manually, some will be injected
19  	}
20  
21  	@Override
22  	protected final Scheduler scheduler() {
23  		final LocalDateTime tomorrowMidnight
24  				= LocalDateTime.now().withHour(0).withMinute(0).withSecond(0).withNano(0).plusDays(1);
25  		final Duration durationUntilMidnight = Duration.between(LocalDateTime.now(), tomorrowMidnight).abs();
26  		return Scheduler.newFixedDelaySchedule(durationUntilMidnight.getSeconds(),
27  				Duration.ofDays(1).getSeconds(),
28  				TimeUnit.SECONDS);
29  	}
30  }
