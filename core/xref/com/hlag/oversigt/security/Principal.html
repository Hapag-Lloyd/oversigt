16  public class Principal implements java.security.Principal {
17  	private static final Map<String, Principal> PRINCIPALS = Collections.synchronizedMap(new WeakHashMap<>());
18  
19  	static Optional<Principal> getPrincipal(final String username) {
20  		return Optional.ofNullable(PRINCIPALS.get(username));
21  	}
22  
23  	/**
24  	 * Gets the principal with the given username from the given
25  	 * {@link Authenticator}.
26  	 *
27  	 * @param authenticator the {@link Authenticator} to use for reading the
28  	 *                      principal information
29  	 * @param username      the name of the {@link Principal} to read
30  	 * @return the read {@link Principal} object. This method does not return
31  	 *         <code>null</code>.
32  	 * @throws java.util.NoSuchElementException if the principal could not be
33  	 *                                          loaded.
34  	 */
35  	public static Optional<Principal> loadPrincipal(final Authenticator authenticator, final String username) {
36  		PRINCIPALS.computeIfAbsent(username, name -> {
37  			final Optional<Principal> principal = authenticator.readPrincipal(name);
38  			authenticator.reloadRoles(name);
39  			return principal.orElse(null);
40  		});
41  		return getPrincipal(username);
42  	}

99  	public synchronized boolean hasRole(final Role role) {
100 		for (Optional<Role> currentRole = Optional.of(role);
101 				currentRole.isPresent();
102 				currentRole = currentRole.get().getParent()) {
103 			if (roles.contains(currentRole.get())) {
104 				return true;
105 			}
106 		}
107 		return false;
108 	}
109 
110 	public boolean hasRole(final String roleName) {
111 		final Optional<Roles> roles = Roles.fromString(roleName);
112 		if (roles.isPresent()) {
113 			return hasRole(roles.get().getRole());
114 		}
115 
116 		final String[] parts = Ascii.toLowerCase(roleName).split("\\.", 3);
117 		if (parts.length == 3 && parts[0].equals("dashboard")) {
118 			switch (parts[2]) {
119 			case "owner":
120 				return hasRole(Role.DASHBOARD_OWNER.getDashboardSpecificRole(parts[1]));
121 			case "editor":
122 				return hasRole(Role.DASHBOARD_EDITOR.getDashboardSpecificRole(parts[1]));
123 			default:
124 				// unknown part
125 				return false;
126 			}
127 		}
128 		// does not fit current role names
129 		return false;
130 	}
