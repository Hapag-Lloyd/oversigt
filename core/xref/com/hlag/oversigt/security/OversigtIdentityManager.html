3   import static java.util.stream.Collectors.toSet;
4   
5   import java.util.Objects;
6   import java.util.Optional;
7   import java.util.Set;
8   
9   import com.google.inject.Inject;
10  import com.google.inject.Singleton;
11  
12  import edu.umd.cs.findbugs.annotations.Nullable;
13  import io.undertow.security.idm.Account;
14  import io.undertow.security.idm.Credential;
15  import io.undertow.security.idm.IdentityManager;
16  import io.undertow.security.idm.PasswordCredential;
17  
18  /**
19   * @author neumaol
20   * @see <a href="https://www.baeldung.com/jboss-undertow">Baeldung</a>
21   */
22  @Singleton
23  public class OversigtIdentityManager implements IdentityManager {
24  	@Inject
25  	private Authenticator authenticator;
26  
27  	OversigtIdentityManager() {
28  		// nothing to do
29  	}
30  
31  	@Override
32  	@Nullable
33  	public Account verify(@Nullable final Account account) {
34  		return account;
35  	}
36  
37  	@Override
38  	@Nullable
39  	public Account verify(@SuppressWarnings("unused") @Nullable final Credential credential) {
40  		return null;
41  	}
42  
43  	@Override
44  	@Nullable
45  	public Account verify(@Nullable final String id, @Nullable final Credential credential) {
46  		if (credential instanceof PasswordCredential) {
47  			final Optional<Principal> principal = authenticator.login(Objects.requireNonNull(id),
48  					new String(((PasswordCredential) credential).getPassword()));
49  			return principal.map(OversigtAccount::new).orElse(null);
50  		}
51  		return null;
52  	}
53  
54  	private static final class OversigtAccount implements Account {
55  		private final Principal principal;
56  
57  		OversigtAccount(final Principal principal) {
58  			this.principal = principal;
59  		}
60  
61  		@Override
62  		public Principal getPrincipal() {
63  			return principal;
64  		}
65  
66  		@Override
67  		public Set<String> getRoles() {
68  			return principal.getRoles().stream().map(Object::toString).collect(toSet());
69  		}
70  	}
71  }
