3   import java.lang.annotation.ElementType;
4   import java.lang.annotation.Retention;
5   import java.lang.annotation.RetentionPolicy;
6   import java.lang.annotation.Target;
7   
8   import javax.ws.rs.NameBinding;
9   
10  /**
11   * @author Olaf Neumann
12   * @see <a href=
13   *      "https://stackoverflow.com/questions/26777083/best-practice-for-rest-token-based-authentication-with-jax-rs-and-jersey">https://stackoverflow.com/questions/26777083/best-practice-for-rest-token-based-authentication-with-jax-rs-and-jersey</a>
14   */
15  @NameBinding
16  @Retention(RetentionPolicy.RUNTIME)
17  @Target({ ElementType.TYPE, ElementType.METHOD })
18  public @interface JwtSecured {
19  	/**
20  	 * Defines whether a call to the annotated method must be authenticated or not.
21  	 * Setting this value to <code>true</code> will authentication is required.
22  	 *
23  	 * @return <code>true</code> if authentication is required, else
24  	 *         <code>false</code>
25  	 */
26  	boolean mustBeAuthenticated() default true;
27  }
