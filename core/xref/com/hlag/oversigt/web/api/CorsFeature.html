3   import javax.ws.rs.core.Feature;
4   import javax.ws.rs.core.FeatureContext;
5   import javax.ws.rs.ext.Provider;

7   import org.jboss.resteasy.plugins.interceptors.CorsFilter;
8   import org.slf4j.Logger;
9   import org.slf4j.LoggerFactory;
10  
11  import com.google.inject.Inject;
12  import com.google.inject.name.Named;
13  
14  import edu.umd.cs.findbugs.annotations.Nullable;
15  
16  /**
17   * Feature to enable CORS in debug mode to enable development of web UIs
18   *
19   * @see <a href=
20   *      "https://stackoverflow.com/questions/36511330/resteasy-and-corsfilter-implementation">StackOverflow</a>
21   */
22  @Provider
23  public class CorsFeature implements Feature {
24  	private static final Logger LOGGER = LoggerFactory.getLogger(CorsFeature.class);
25  
26  	@Inject
27  	@Named("debug")
28  	private boolean debug;
29  
30  	public CorsFeature() {
31  		// no fields to be initialized manually, some will be injected
32  	}
33  
34  	/** {@inheritDoc} */
35  	@Override
36  	public boolean configure(@Nullable final FeatureContext context) {
37  		if (debug && context != null) {
38  			LOGGER.warn("Debug mode: Enabling CORS to permit calls from other hosts.");
39  			final CorsFilter corsFilter = new CorsFilter();
40  			corsFilter.getAllowedOrigins().add("*");
41  			context.register(corsFilter);
42  		} else {
43  			LOGGER.info("CORS will not be enabled in order to prevent calls from other servers.");
44  		}
45  		return true;
46  	}
47  }
