1   package com.hlag.oversigt.web.api;

3   import java.util.LinkedHashSet;
4   import java.util.Objects;
5   import java.util.Set;
6   
7   import javax.validation.ConstraintViolation;
8   import javax.validation.Validator;
9   
10  import org.aopalliance.intercept.MethodInterceptor;
11  import org.aopalliance.intercept.MethodInvocation;
12  
13  import com.google.inject.Inject;
14  
15  import edu.umd.cs.findbugs.annotations.Nullable;
16  
17  /**
18   * Interceptor that checks parameters of calls to public API methods. The javax
19   * annotations will be used to determine valid values for the parameters.
20   *
21   * @author Olaf Neumann
22   *
23   */
24  public class ApiValidationInterceptor implements MethodInterceptor {
25  	@Inject
26  	private Validator validator;
27  
28  	public ApiValidationInterceptor() {
29  		// no fields to be initialized manually, some will be injected
30  	}
31  
32  	/** {@inheritDoc} */
33  	@Override
34  	public Object invoke(@Nullable final MethodInvocation nullableInvocation) throws Throwable {
35  		final MethodInvocation invocation = Objects.requireNonNull(nullableInvocation);
36  
37  		// save parameters for logging
38  		LoggingInterceptor.PARAMETERS.set(invocation.getArguments());
39  
40  		// Validate call
41  		final Set<ConstraintViolation<?>> violations = new LinkedHashSet<>();
42  
43  		violations.addAll(validator.forExecutables()
44  				.validateParameters(invocation.getThis(), invocation.getMethod(), invocation.getArguments()));
45  
46  		final Class<?>[] types = invocation.getMethod().getParameterTypes();
47  		for (int i = 0; i < types.length; i += 1) {
48  			if (!(types[i].isPrimitive()
49  					|| Number.class.isAssignableFrom(types[i])
50  					|| types[i] == String.class
51  					|| types[i] == Boolean.class)) {
52  				violations.addAll(validator.validate(invocation.getArguments()[i]));
53  			}
54  		}
55  
56  		if (violations.isEmpty()) {
57  			return invocation.proceed();
58  		}
59  		throw new ApiValidationException(invocation.getMethod(), violations);
60  	}
61  }
