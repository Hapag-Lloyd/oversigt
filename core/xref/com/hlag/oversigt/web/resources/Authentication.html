86  	@POST
87  	@Consumes(MediaType.APPLICATION_FORM_URLENCODED)
88  	@Produces(MediaType.APPLICATION_JSON)
89  	@Path("/login")
90  	@ApiResponses({
91  			@ApiResponse(code = 200, message = "User successfully logged in", response = AuthData.class),
92  			@ApiResponse(code = 403, message = "Log in failed") })
93  	@ApiOperation("Log in a user")
94  	public Response authenticateUser(@FormParam("username") final String username,
95  			@FormParam("password") @ApiParam(format = "password") final String password) {
96  		try {
97  			// Authenticate the user using the credentials provided
98  			final Principal principal = authentication.authenticate(username, password);
99  
100 			// Issue a token for the user
101 			final String token = authentication.issueToken(principal);
102 			Utils.logChange(principal, "logged in");
103 
104 			// Return the token on the response
105 			return ok(new AuthData(principal.getUsername(), principal.getName(), token, findRolesForUser(principal)),
106 					MediaType.APPLICATION_JSON).build();
107 		} catch (final Exception e) {
108 			LOGGER.warn("MAYBE " + username + " - tried to authenticate", e);
109 			return ErrorResponse.forbidden("The user and password combination is unknown.");
110 		}
111 	}

113 	@GET
114 	@Consumes(MediaType.APPLICATION_JSON)
115 	@Produces(MediaType.APPLICATION_JSON)
116 	@Path("/renew")
117 	@ApiResponses({
118 			@ApiResponse(code = 200, message = "Token successfully renewed", response = AuthData.class),
119 			@ApiResponse(code = 403, message = "Token renewal failed") })
120 	@ApiOperation("Renew the authentication token")
121 	@NoChangeLog
122 	public Response renewToken(@HeaderParam("token") @NotBlank @ApiParam(allowEmptyValue = false,
123 			value = "The JWT to renew") final String token) {
124 		String newToken = null;
125 		try {
126 			final Principal principal = authentication.validateToken(token);
127 			newToken = authentication.issueToken(principal);
128 
129 			return ok(new AuthData(principal.getUsername(), principal.getName(), newToken, findRolesForUser(principal)),
130 					MediaType.APPLICATION_JSON).build();
131 		} catch (@SuppressWarnings("unused") final Exception ignore) {
132 			// take any exception of the token check as login failure
133 		}
134 		if (newToken == null) {
135 			return Response.status(Status.FORBIDDEN).build();
136 		}
137 		return ok(createTokenMap(newToken)).build();
138 	}

140 	@GET
141 	@Path("/roles")
142 	@Produces(MediaType.APPLICATION_JSON)
143 	@ApiResponses({
144 			@ApiResponse(code = 200, message = "A list of roles the current user has.", response = AuthData.class) })
145 	@ApiOperation(value = "Get user roles",
146 			authorizations = { @Authorization(value = ApiAuthenticationFilter.API_OPERATION_AUTHENTICATION) })
147 	@NoChangeLog
148 	@JwtSecured
149 	public Response getRoles() {
150 		return ok(new AuthData(findRolesForUser((Principal) getSecurityContext().getUserPrincipal()))).build();
151 	}
152 
153 	private Set<String> findRolesForUser(final Principal principal) {
154 		return roleProvider.getRoles(principal.getUsername()).stream().map(Role::getName).collect(toSet());
155 	}

162 			@ApiResponse(code = 200,
163 					message = "The check was successful and the result can be found in the response body",
164 					response = boolean.class) })
165 	@NoChangeLog
166 	public boolean checkToken(@HeaderParam("token") @NotBlank @ApiParam(allowEmptyValue = false,
167 			value = "The JWT to check") final String token) {
168 		try {
169 			authentication.validateToken(token);
170 			return true; // ok(true).build();
171 		} catch (@SuppressWarnings("unused") final Exception e) {
172 			return false; // ok(false).build();
173 		}

176 	public static class AuthData {
177 		private final Optional<String> userId;
178 
179 		private final Optional<String> displayName;
180 
181 		private final Optional<String> token;
182 
183 		@NotNull
184 		private final Set<@NotBlank String> roles;
185 
186 		AuthData(final String userId, final String displayName, final String token, final Set<@NotBlank String> roles) {
187 			this.userId = Optional.of(userId);
188 			this.displayName = Optional.of(displayName);
189 			this.token = Optional.of(token);
190 			this.roles = new LinkedHashSet<>(roles);
191 		}
192 
193 		AuthData(final Set<String> roles) {
194 			userId = Optional.empty();
195 			displayName = Optional.empty();
196 			token = Optional.empty();
197 			this.roles = new LinkedHashSet<>(roles);
198 		}
199 
200 		public Optional<String> getUserId() {
201 			return userId;
202 		}
203 
204 		public Optional<String> getDisplayName() {
205 			return displayName;
206 		}
207 
208 		public Optional<String> getToken() {
209 			return token;
210 		}
211 
212 		public Set<String> getRoles() {
213 			return roles;
214 		}
