112 	@GET
113 	@ApiResponses({
114 			@ApiResponse(code = 200,
115 					message = "Returns a list of existing event source instances",
116 					response = EventSourceInstanceInfo.class,
117 					responseContainer = "List") })
118 	@JwtSecured
119 	@ApiOperation(value = "List existing event source instances")
120 	@NoChangeLog
121 	public Response listInstances(@QueryParam("containing") @ApiParam(required = false,
122 			value = "Filter to reduce the number of listed instances") @Nullable @edu.umd.cs.findbugs.annotations.Nullable final String containing,
123 			@QueryParam("limit") @ApiParam(required = false,
124 					value = "Maximum number of instances to be returned") @Nullable @edu.umd.cs.findbugs.annotations.Nullable final Integer limit,
125 			@QueryParam("onlyStartable") @ApiParam(required = false,
126 					value = "Only return instances that can be started") @Nullable @edu.umd.cs.findbugs.annotations.Nullable final Boolean onlyStartable) {
127 		Predicate<EventSourceInstance> containingFilter = i -> true;
128 		Predicate<EventSourceInstance> startableFilter = i -> true;
129 
130 		if (!Strings.isNullOrEmpty(containing)) {
131 			final String searchedContent = Strings.nullToEmpty(containing).toLowerCase();
132 			containingFilter = EventSourceInstance.createFilter(searchedContent);
133 		}
134 
135 		if (onlyStartable != null && onlyStartable) {
136 			startableFilter = i -> i.getDescriptor().getEventClass() != null;
137 		}
138 
139 		Stream<EventSourceInstanceInfo> stream = eventSourceInstanceController.getEventSourceInstances()
140 				.stream()
141 				.filter(containingFilter)
142 				.filter(startableFilter)
143 				.map(instance -> new EventSourceInstanceInfo(eventSourceInstanceController,
144 						dashboardController,
145 						statisticsManager,
146 						instance));
147 		if (limit != null && limit > 0) {
148 			stream = stream.limit(limit);
149 		}
150 		final Set<EventSourceInstanceInfo> infos
151 				= new TreeSet<>((a, b) -> String.CASE_INSENSITIVE_ORDER.compare(a.getName(), b.getName()));
152 		infos.addAll(stream.collect(toList()));
153 		return ok(infos).build();
154 	}

159 			@ApiResponse(code = 201,
160 					message = "Event source instance created",
161 					response = EventSourceInstanceDetails.class),
162 			@ApiResponse(code = 400,
163 					message = "The data provided by the client is invalid",
164 					response = ErrorResponse.class),
165 			@ApiResponse(code = 404,
166 					message = "There is no event source descriptor named by the given key",
167 					response = ErrorResponse.class) })
168 	@JwtSecured
169 	@ApiOperation(value = "Create event source instance")
170 	@RolesAllowed(Role.ROLE_NAME_GENERAL_DASHBOARD_OWNER)
171 	public Response createInstance(@QueryParam("key") @ApiParam(
172 			value = "The key of the event source descriptor to be used") @NotBlank final String keyString) {
173 		final EventSourceKey key;
174 		try {
175 			key = EventSourceKey.fromKeyString(keyString);
176 		} catch (@SuppressWarnings("unused") final InvalidKeyException e) {
177 			return ErrorResponse.badRequest("The key '" + keyString + "' is invalid.");
178 		}
179 
180 		final EventSourceDescriptor descriptor;
181 		try {
182 			descriptor = eventSourceDescriptorController.getEventSourceDescriptor(key);
183 		} catch (@SuppressWarnings("unused") final NoSuchElementException e) {
184 			return ErrorResponse.notFound("No descriptor found for key: " + keyString);
185 		}
186 
187 		final EventSourceInstance instance = eventSourceInstanceController.createEventSourceInstance(descriptor,
188 				(Principal) getSecurityContext().getUserPrincipal());
189 		return created(URI.create(getUriInfo().getAbsolutePath() + "/" + instance.getId()))
190 				.entity(new EventSourceInstanceDetails(instance))
191 				.build();
192 	}

243 	@Path("/{id}")
244 	@ApiResponses({
245 			@ApiResponse(code = 200,
246 					message = "The event source instance has been updated",
247 					response = FullEventSourceInstanceInfo.class),
248 			@ApiResponse(code = 400, message = "The data is invalid", response = ErrorResponse.class),
249 			@ApiResponse(code = 404,
250 					message = "The event source instance with the given id does not exist",
251 					response = ErrorResponse.class) })
252 	@JwtSecured
253 	@ApiOperation(value = "Update event source instance")
254 	@RolesAllowed(Role.ROLE_NAME_GENERAL_DASHBOARD_OWNER)
255 	public Response updateInstance(@PathParam("id") @NotBlank final String instanceId,
256 			final EventSourceInstanceDetails details) {
257 		final EventSourceInstance originalInstance;
258 		try {
259 			originalInstance = eventSourceInstanceController.getEventSourceInstance(instanceId);
260 		} catch (@SuppressWarnings("unused") final NoSuchElementException e) {
261 			return ErrorResponse.notFound("Event source instance does not exist.");
262 		}
263 
264 		if (!originalInstance.getId().equals(details.getId()) || !details.getId().equals(instanceId)) {
265 			return ErrorResponse.badRequest("The ID does not match");
266 		}
267 		if (!originalInstance.getDescriptor().getKey().getKey().equals(details.getEventSourceDescriptor())) {
268 			return ErrorResponse.badRequest("The event source descriptor key does not match");
269 		}
270 		// TODO this needs to be checked when the UI is fixed
271 		// if (!instance.getDescriptor().isScheduledService()&& details.frequency !=
272 		// Duration.ZERO) {
273 		// return ErrorResponse.badRequest("The event source does not take a
274 		// frequency");
275 		// }
276 		final Set<String> unnessaccaryDataItems = new HashSet<>(details.dataItems.keySet());
277 		unnessaccaryDataItems.removeAll(originalInstance.getDescriptor()
278 				.getDataItems()
279 				.stream()
280 				.map(EventSourceProperty::getName)
281 				.collect(Collectors.toSet()));
282 		if (!unnessaccaryDataItems.isEmpty()) {
283 			return ErrorResponse.badRequest("Some data items are unknown", unnessaccaryDataItems);
284 		}
285 
286 		final EventSourceInstance newInstance;
287 		try {
288 			newInstance = new EventSourceInstance(originalInstance.getDescriptor(),
289 					originalInstance.getId(),
290 					details.getName(),
291 					details.isEnabled(),
292 					details.getFrequency(),
293 					originalInstance.getCreatedBy(),
294 					((Principal) getSecurityContext().getUserPrincipal()).getUsername());
295 			newInstance.setEnabled(details.isEnabled());
296 			newInstance.setName(details.getName());
297 			newInstance.setFrequency(details.getFrequency());
298 			newInstance.getDescriptor()
299 					.getProperties()
300 					.forEach(p -> newInstance.setPropertyString(p, details.properties.get(p.getName())));
301 			adoptPasswordsForProperties(newInstance, originalInstance);
302 			newInstance.getDescriptor().getDataItems().forEach(p -> {
303 				if (details.dataItems.containsKey(p.getName())) {
304 					newInstance.setPropertyString(p, details.dataItems.get(p.getName()));
305 				} else {
306 					newInstance.removeProperty(p);
307 				}
308 			});
309 		} catch (final Exception e) {
310 			LOGGER.error("Unable to update event source instance", e);
311 			return ErrorResponse.badRequest("Invalid event source instance values", e);
312 		}
313 
314 		eventSourceInstanceController.updateEventSourceInstance(newInstance);
315 		return ok(getInstanceInfo(instanceId)).build();
316 	}

376 	static Map<String, String> getValueMap(final Stream<EventSourceProperty> propertyStream,
377 			final Function<EventSourceProperty, String> getValue,
378 			final Predicate<EventSourceProperty> hasValue,
379 			final boolean removeEmpty) {
380 		final Map<String, String> map = propertyStream.filter(p -> !(removeEmpty && !hasValue.test(p)))
381 				.collect(Collectors.toMap(EventSourceProperty::getName, p -> {
382 					String string = getValue.apply(p);
383 					if (p.isJson()) {
384 						string = JsonUtils.removePasswordsFromJson(string);
385 					}
386 					return string;
387 				}));
388 		return removePasswords(map);
389 	}

441 			usedBy = new ArrayList<>(dashboardController.getDashboardsWhereEventSourceIsUsed(instance)
