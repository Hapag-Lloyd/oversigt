126 	@Path("/value/{type}")
127 	@ApiResponses({
128 			@ApiResponse(code = 201, message = "Serializable property has been created", response = Map.class),
129 			@ApiResponse(code = 404,
130 					message = "Serializable property type does not exist",
131 					response = ErrorResponse.class) })
132 	@JwtSecured
133 	@ApiOperation(value = "Create serializable property values",
134 			authorizations = { @Authorization(value = ApiAuthenticationFilter.API_OPERATION_AUTHENTICATION) })
135 	@RolesAllowed(Role.ROLE_NAME_GENERAL_DASHBOARD_OWNER)
136 	public Response createProperty(@Context final UriInfo uri,
137 			@PathParam("type") @NotBlank final String className,
138 			@NotEmpty final Map<@NotBlank String, @NotNull Object> map) {
139 		final Class<? extends SerializableProperty> clazz;
140 		try {
141 			clazz = spController.getClass(className);
142 		} catch (@SuppressWarnings("unused") final NoSuchElementException e) {
143 			return ErrorResponse.notFound("Serializable property type '" + className + "' does not exist.");
144 		}
145 
146 		final List<String> errors = checkMembers(clazz, map);
147 		if (!errors.isEmpty()) {
148 			return ErrorResponse.badRequest("Cannot create serializable property", errors);
149 		}
150 
151 		try {
152 			final SerializableProperty prop = spController.createProperty(clazz, (String) map.get("name"), map);
153 			// TODO change to proper Location-header returning response
154 			return status(Status.CREATED).entity(toMapWithoutPassword(prop))
155 					.type(MediaType.APPLICATION_JSON_TYPE)
156 					.link(uri.getAbsolutePath() + "/" + prop.getId(), "self")
157 					.build();
158 		} catch (@SuppressWarnings("unused") final NoSuchElementException e) {
159 			return ErrorResponse.notFound("Serializable property '" + className + "' does not exist.");
160 		}

190 	@PUT
191 	@Path("/value/{type}/{id}")
192 	@ApiResponses({
193 			@ApiResponse(code = 200,
194 					message = "Returns a list of all values of the requested serializable property",
195 					response = Map.class),
196 			@ApiResponse(code = 404, message = "Property does not exist", response = ErrorResponse.class) })
197 	@JwtSecured
198 	@ApiOperation(value = "Update serializable property values",
199 			authorizations = { @Authorization(value = ApiAuthenticationFilter.API_OPERATION_AUTHENTICATION) })
200 	@RolesAllowed(Role.ROLE_NAME_GENERAL_DASHBOARD_OWNER)
201 	public Response updateProperty(@Context final UriInfo uri,
202 			@PathParam("type") @NotBlank final String className,
203 			@PathParam("id") @Positive final int id,
204 			@NotNull final Map<@NotBlank String, @NotNull Object> map) {
205 		final Class<? extends SerializableProperty> clazz;
206 		try {
207 			clazz = spController.getClass(className);
208 		} catch (@SuppressWarnings("unused") final NoSuchElementException e) {
209 			return ErrorResponse.notFound("Serializable property type '" + className + "' does not exist.");
210 		}
211 		final SerializableProperty prop = spController.getProperty(clazz, id);
212 		if (prop.getId() != id) {
213 			return ErrorResponse
214 					.notFound("Serializable property of type '" + className + "' with id " + id + " does not exist.");
215 		}
216 
217 		final List<String> errors = checkMembers(clazz, map);
218 		if (!errors.isEmpty()) {
219 			return ErrorResponse.badRequest("Cannot update serializable property", errors);
220 		}
221 
222 		// TODO make change in a copy of the property and then write changes to real
223 		// object
224 		for (final SerializablePropertyMember member : spController.getMembers(clazz)) {
225 			try {
226 				member.set(prop, map.get(member.getName()).toString());
227 			} catch (final MemberMissingException e) {
228 				throw new RuntimeException("Cannot find member: " + member.getName(), e);
229 			}
230 		}
231 		spController.updateProperty(prop);
232 		esiController.restartInstancesUsingSerializableProperty(prop);
233 
234 		return ok(toMapWithoutPassword(prop)).link(uri.getAbsolutePath() + "/" + prop.getId(), "self").build();
235 	}

266 	private List<String> checkMembers(final Class<? extends SerializableProperty> clazz,
267 			final Map<String, Object> map) {
268 		final List<String> errors = new ArrayList<>();
269 		// Are all members in the map?
270 		final Collection<SerializablePropertyMember> members = spController.getMembers(clazz);
271 		for (final SerializablePropertyMember m : members) {
272 			if (!map.containsKey(m.getName())) {
273 				errors.add("Member '" + m.getName() + "' is missing");
274 			}
275 		}
276 		// Are only members in the map?
277 		if (map.size() != members.size()) {
278 			members.stream().map(SerializablePropertyMember::getName).forEach(map::remove);
279 			for (final String n : map.keySet()) {
280 				errors.add("'" + n + "' is not a member of " + clazz.getSimpleName());
281 			}
282 		}
283 		return errors;
284 	}
