114 	@ApiResponses({
115 			@ApiResponse(code = 202, message = "Server shutdown initiated."),
116 			@ApiResponse(code = 401, message = "If the user is not authorized.") })
117 	@JwtSecured
118 	@ApiOperation(value = "Shut down the server",
119 			authorizations = { @Authorization(value = ApiAuthenticationFilter.API_OPERATION_AUTHENTICATION) },
120 			notes = "Shuts down the dashboard server. Once the request has been accepted the shut down will be initiated after two seconds.")
121 	@RolesAllowed(Role.ROLE_NAME_SERVER_ADMIN)
122 	public Response shutdown() {
123 		ForkJoinPool.commonPool().execute(() -> {
124 			try {
125 				Thread.sleep(1000);
126 			} catch (@SuppressWarnings("unused") final Exception ignore) {
127 				// on interruption continue
128 			}
129 			shutdownRunnable.run();
130 		});
131 		return Response.status(Status.ACCEPTED).build();
132 	}

170 	@GET
171 	@Path("/logfiles/{filename}")
172 	@ApiResponses({
173 			@ApiResponse(code = 200,
174 					message = "The requested lines of the log file will be transferred",
175 					response = String.class,
176 					responseContainer = "list") })
177 	@JwtSecured
178 	@ApiOperation(value = "Retreive log file content",
179 			authorizations = { @Authorization(value = ApiAuthenticationFilter.API_OPERATION_AUTHENTICATION) })
180 	@NoChangeLog
181 	@RolesAllowed(Role.ROLE_NAME_SERVER_ADMIN)
182 	public Response getLogFileContent(@PathParam("filename") @NotBlank final String filename,
183 			@QueryParam("lines") @ApiParam(required = false,
184 					defaultValue = "0",
185 					value = "Number of lines to read from the log file. Negative value to read from the end of the file.") final Integer nullableLineCount)
186 			throws IOException {
187 		final int lineCount = Nullables.orElse(nullableLineCount, 0);
188 
189 		final java.nio.file.Path logfile = Paths.get("log", filename);
190 		if (!Files.exists(logfile)) {
191 			return notFound("The log file '" + filename + "' does not exist.");
192 		}
193 
194 		final Collection<String> lines;
195 		try (Stream<String> lineStream = Files.lines(logfile)) {
196 			if (lineCount == 0) {
197 				lines = lineStream.collect(Collectors.toList());
198 			} else if (lineCount > 0) {
199 				lines = lineStream.limit(lineCount).collect(Collectors.toList());
200 			} else {
201 				lines = lineStream.collect(Collectors.toCollection(() -> EvictingQueue.create(-lineCount)));
202 			}
203 		}
204 		return ok(lines).build();
205 	}

296 	@JwtSecured
297 	@ApiOperation(value = "Retrieve the cached events",
298 			authorizations = { @Authorization(value = ApiAuthenticationFilter.API_OPERATION_AUTHENTICATION) })
299 	@RolesAllowed(Role.ROLE_NAME_GENERAL_DASHBOARD_OWNER)
300 	@NoChangeLog
301 	public Response getCachedEvents(
302 			@QueryParam("eventSourceId") @ApiParam(value = "Optional filter to get only one cached event",
303 					required = false) final String filter) {
304 		final Collection<OversigtEvent> events = new ArrayList<>(eventSender.getCachedEvents());
305 		if (Strings.isNullOrEmpty(filter)) {
306 			return ok(events.stream()
307 					.map(OversigtEvent::toJson)
308 					.map(json -> JsonUtils.fromJson(json, Map.class))
309 					.collect(toList())).build();
310 		}
311 
312 		final Optional<OversigtEvent> event
313 				= events.stream().filter(e -> Objects.requireNonNull(e.getId()).equals(filter)).findFirst();
314 		if (!event.isPresent()) {
315 			return ErrorResponse.notFound("The event source does not exist", filter);
316 		}
317 
318 		final List<OversigtEvent> eventList
319 				= Arrays.asList(event.orElseThrow(() -> new RuntimeException("The event is not present")));
320 		final String json = JsonUtils.toJson(eventList);
321 		return ok(json).build();
322 	}

368 	@Path("/server/all-objects")
369 	@ApiResponses({ @ApiResponse(code = 200, message = "Search all server objects", response = ServerInfo.class) })
370 	@ApiOperation(value = "Server wide search for objects")
371 	@NoChangeLog
372 	public Response searchAllObjects(@QueryParam("query") @ApiParam(value = "Text to search for",
373 			required = true) final String rawSearchString) {
374 		if (Strings.nullToEmpty(rawSearchString).trim().length() == 0) {
375 			return ErrorResponse.badRequest("The search text must not be empty.");
376 		}
377 
378 		// prepare search
379 		final String searchString = rawSearchString.toLowerCase();
380 		final List<SearchResult> results = new ArrayList<>();
381 
382 		// TODO implement dynamic search registration
383 
384 		// search for dashboards
385 		results.addAll(dashboardController.getDashboardIds()
386 				.stream()
387 				.map(dashboardController::getDashboard)
388 				.map(Optional::get)
389 				.filter(d -> d.getTitle().toLowerCase().contains(searchString)
390 						|| d.getId().toLowerCase().contains(searchString))
391 				.map(d -> new SearchResult(d.getTitle(), d.getId(), "dashboard"))
392 				.collect(toList()));
393 
394 		// search for event sources
395 		results.addAll(eventSourceInstanceController.getEventSourceInstances()
396 				.stream()
397 				.filter(EventSourceInstance.createFilter(searchString))
398 				.map(i -> new SearchResult(i.getName(), i.getId(), "event-source"))
399 				.collect(toList()));
400 
401 		// search for properties
402 		spController.getClasses()
403 				.stream()
404 				.flatMap(spController::streamProperties)
405 				.filter(p -> SerializablePropertyResource.toMapWithoutPassword(p)
406 						.values()
407 						.stream()
408 						.filter(v -> v != null)
409 						.map(Object::toString)
410 						.map(String::toLowerCase)
411 						.anyMatch(s -> s.contains(searchString)))
412 				.map(p -> new SearchResult(p.getName(),
413 						Integer.toString(p.getId()),
414 						"serializable-property",
415 						p.getClass().getSimpleName()))
416 				.collect(Collectors.toList());
417 
418 		// search for loggers
419 		results.addAll(getLoggerStream().map(Logger::getName)
420 				.map(String::toLowerCase)
421 				.filter(name -> name.contains(searchString))
422 				.map(name -> new SearchResult(name, name, "logger"))
423 				.collect(toList()));
424 
425 		return ok(results).build();
426 	}
