113 	@ApiResponses({
114 			@ApiResponse(code = 201,
115 					message = "A new dashboard has been created. The body contains a detailed representation of the newly created dashboard",
116 					response = Dashboard.class),
117 			@ApiResponse(code = 303,
118 					message = "The dashboard already exists. Sending redirect to dashboard information.") })
119 	@JwtSecured
120 	@ApiOperation(value = "Create new dashboard",
121 			authorizations = { @Authorization(value = ApiAuthenticationFilter.API_OPERATION_AUTHENTICATION) })
122 	public Response createDashboard(@Context final SecurityContext securityContext,
123 			@QueryParam("dashboardId") @NotNull final String id,
124 			@QueryParam("owner") @NotNull @UserId final String ownerUserId,
125 			@QueryParam("enabled") @ApiParam(defaultValue = "false") final boolean enabled) {
126 		final Optional<Dashboard> existingDashboard = dashboardController.getDashboard(id);
127 		if (existingDashboard.isPresent()) {
128 			return Response.seeOther(URI.create(getUriInfo().getAbsolutePath() + "/" + id)).build();
129 		}
130 
131 		final Optional<Dashboard> createdDashboard = dashboardController.createDashboard(id,
132 				Principal.loadPrincipal(authenticator, ownerUserId)
133 						.orElseThrow(() -> new RuntimeException("Unknown principal for: " + ownerUserId)),
134 				enabled && securityContext.isUserInRole(Role.ROLE_NAME_SERVER_ADMIN));
135 		if (!createdDashboard.isPresent()) {
136 			final UUID uuid = UUID.randomUUID();
137 			LOGGER.error("Dashboard {} does not exist, but creation failed, too.", id);
138 			return internalServerError(uuid, "Unable to create dashboard. Details are written to log file.");
139 		}
140 		return created(URI.create(getUriInfo().getAbsolutePath() + "/" + createdDashboard.get().getId()))
141 				.entity(createdDashboard.get())
142 				.type(MediaType.APPLICATION_JSON_TYPE)
143 				.build();
144 
145 		// TODO write a mail to server admins if the dashboard needs to be enabled
146 	}

157 	@JwtSecured(mustBeAuthenticated = false)
158 	@ApiOperation(value = "Read dashboard details",
159 			authorizations = { @Authorization(value = ApiAuthenticationFilter.API_OPERATION_AUTHENTICATION) })
160 	@NoChangeLog
161 	@PermitAll
162 	public Response readDashboard(@Context final SecurityContext secu, @PathParam("dashboardId") final String id) {
163 		final Optional<Dashboard> dashboard = dashboardController.getDashboard(id);
164 		if (!dashboard.isPresent()) {
165 			return notFound("A dashboard with id '" + id + "' does not exist.");
166 		}
167 
168 		final Map<String, Object> dashboardMap = dashboard.map(TypeUtils::toMemberMap)
169 				.orElseThrow(() -> new RuntimeException("The dashboard is not present."));
170 		if (secu.getUserPrincipal() == null) {
171 			dashboardMap.remove("owner");
172 			dashboardMap.remove("editors");
173 		}
174 		return ok(dashboardMap).build();
175 	}

179 	@ApiResponses({
180 			@ApiResponse(code = 200, message = "The dashboard has been updated", response = Dashboard.class),
181 			@ApiResponse(code = 400, message = "The provided information are invalid", response = ErrorResponse.class),
182 			@ApiResponse(code = 403,
183 					message = "The user is either not permitted to edit this dashboard or he wants to perform a change that he is not allowed to",
184 					response = ErrorResponse.class),
185 			@ApiResponse(code = 404, message = "The dashboard does not exist") })
186 	@ApiOperation(value = "Partially update dashboard details",
187 			authorizations = { @Authorization(value = ApiAuthenticationFilter.API_OPERATION_AUTHENTICATION) })
188 	@JwtSecured
189 	@RolesAllowed("dashboard.{dashboardId}.editor")
190 	public Response updateDashboardPartially(@Context final SecurityContext securityContext,
191 			@PathParam("dashboardId") @NotNull final String id,
192 			final Map<String, Object> newDashboardData) {
193 		// load current dashboard from storage
194 		final Optional<Dashboard> dashboard = dashboardController.getDashboard(id);
195 		if (!dashboard.isPresent()) {
196 			return notFound("A dashboard with id '" + id + "' does not exist.");
197 		}
198 
199 		// clone it to work on it
200 		final Dashboard newDashboard = dashboard.get().copy();
201 
202 		// check dashboard id
203 		if (newDashboardData.containsKey("id")
204 				&& (!id.equals(newDashboardData.get("id")) || !id.equals(newDashboard.getId()))) {
205 			return badRequest("The dashboard ID does not match");
206 		}
207 
208 		// check if the change is allowed
209 		final List<String> allowedChangeElements = Arrays.asList("id", "enabled");
210 		if (newDashboardData.keySet().stream().anyMatch(key -> !allowedChangeElements.contains(key))) {
211 			return badRequest("Currently the only allowed attribute for change is: 'enabled'");
212 		}
213 
214 		// Check if the user changed the enabled state of the dashboard. Only server
215 		// admins may perform that action
216 		if (newDashboardData.containsKey("enabled")
217 				&& Boolean.parseBoolean(newDashboardData.get("enabled").toString()) != newDashboard.isEnabled()
218 				&& !securityContext.isUserInRole(Role.ROLE_NAME_SERVER_ADMIN)) {
219 			return forbidden(
220 					"Only server admins are allowed to change a dashboard's enabled state. Please contact a server admin.");
221 		}
222 
223 		// // if the owner has changed the user needs to be at least dashboard owner
224 		// if (!newDashboardData.getOwners().equals(dashboard.getOwners())
225 		// &&
226 		// !securityContext.isUserInRole(Role.getDashboardOwnerRole(dashboard.getId()).getName()))
227 		// {
228 		// return forbidden("To change the owner of a dashboard you need to be at least
229 		// the owner of the dashboard.");
230 		// }
231 
232 		// update dashboard data
233 		if (newDashboardData.containsKey("enabled")) {
234 			newDashboard.setEnabled(Boolean.parseBoolean(newDashboardData.get("enabled").toString()));
235 		}
236 		if (!dashboardController.updateDashboard(newDashboard)) {
237 			return Response.status(Status.INTERNAL_SERVER_ERROR).build();
238 		}
239 		return ok(dashboardController.getDashboard(id)).build();
240 	}

242 	@PUT
243 	@Path("/{dashboardId}")
244 	@ApiResponses({
245 			@ApiResponse(code = 200, message = "The dashboard has been updated", response = Dashboard.class),
246 			@ApiResponse(code = 400, message = "The provided information are invalid", response = ErrorResponse.class),
247 			@ApiResponse(code = 403,
248 					message = "The user is either not permitted to edit this dashboard or he wants to perform a change that he is not allowed to",
249 					response = ErrorResponse.class),
250 			@ApiResponse(code = 404, message = "The dashboard does not exist") })
251 	@ApiOperation(value = "Update dashboard details",
252 			authorizations = { @Authorization(value = ApiAuthenticationFilter.API_OPERATION_AUTHENTICATION) })
253 	@JwtSecured
254 	@RolesAllowed("dashboard.{dashboardId}.editor")
255 	public Response updateDashboard(@Context final SecurityContext securityContext,
256 			@PathParam("dashboardId") @NotNull final String id,
257 			final Dashboard newDashboardData) {
258 		// load current dashboard from storage
259 		final Optional<Dashboard> maybeOriginalDashboard = dashboardController.getDashboard(id);
260 
261 		// Checks
262 		// ======
263 
264 		// does it exist?
265 		if (!maybeOriginalDashboard.isPresent()) {
266 			return notFound("A dashboard with id '" + id + "' does not exist.");
267 		}
268 		final Dashboard originalDashboard = maybeOriginalDashboard.get();
269 
270 		// check dashboard id
271 		if (!id.equals(newDashboardData.getId()) || !id.equals(originalDashboard.getId())) {
272 			return badRequest("The dashboard ID does not match");
273 		}
274 
275 		// Check if the user changed the enabled state of the dashboard. Only server
276 		// admins may perform that action
277 		if (newDashboardData.isEnabled() != originalDashboard.isEnabled()
278 				&& !securityContext.isUserInRole(Role.ROLE_NAME_SERVER_ADMIN)) {
279 			return forbidden(
280 					"Only server admins are allowed to change a dashboard's enabled state. Please contact a server admin.");
281 		}
282 
283 		// if the owner has changed the user needs to be at least dashboard owner
284 		if (!newDashboardData.getOwners().equals(originalDashboard.getOwners())
285 				&& !securityContext.isUserInRole(Role.getDashboardOwnerRole(originalDashboard.getId()).getName())) {
286 			return forbidden("To change the owner of a dashboard you need to be at least the owner of the dashboard.");
287 		}
288 
289 		// check usernames
290 		newDashboardData.setOwners(
291 				newDashboardData.getOwners().stream().filter(authenticator::isUsernameValid).collect(toList()));
292 		newDashboardData.setEditors(
293 				newDashboardData.getEditors().stream().filter(authenticator::isUsernameValid).collect(toList()));
294 		// TODO notify changed users
295 
296 		// Prepare changes
297 		// ===============
298 		final Set<String> users
299 				= Stream.concat(originalDashboard.getOwners().stream(), originalDashboard.getEditors().stream())
300 						.collect(toSet());
301 
302 		// update dashboard data
303 		// =====================
304 		if (!dashboardController.updateDashboard(newDashboardData)) {
305 			return Response.status(Status.INTERNAL_SERVER_ERROR).build();
306 		}
307 
308 		// return dashboard info
309 		// =====================
310 		final Optional<Dashboard> changedDashboard = dashboardController.getDashboard(id);
311 
312 		// reload user rights
313 		users.addAll(
314 				Stream.concat(changedDashboard.get().getOwners().stream(), changedDashboard.get().getEditors().stream())
315 						.collect(toSet()));
316 		users.forEach(authenticator::reloadRoles);
317 
318 		return ok(changedDashboard).build();
319 	}

321 	@PUT
322 	@Path("/{dashboardId}/positions")
323 	@ApiResponses({
324 			@ApiResponse(code = 200, message = "The widget positions has been updated"),
325 			@ApiResponse(code = 400, message = "The provided information are invalid", response = ErrorResponse.class),
326 			@ApiResponse(code = 404, message = "The dashboard does not exist") })
327 	@ApiOperation(value = "Update dashboard widget positions",
328 			authorizations = { @Authorization(value = ApiAuthenticationFilter.API_OPERATION_AUTHENTICATION) })
329 	@JwtSecured
330 	@RolesAllowed("dashboard.{dashboardId}.editor")
331 	public Response updateWidgetPositions(@PathParam("dashboardId") @NotNull final String id,
332 			final List<WidgetPosition> widgetPositions) {
333 		// load current dashboard from storage
334 		final Optional<Dashboard> dashboard = dashboardController.getDashboard(id);
335 
336 		// does it exist?
337 		if (!dashboard.isPresent()) {
338 			return notFound("Dashboard does not exist.");
339 		}
340 
341 		// Do all given widget positions match a dashboard widget?
342 		final Set<String> unmatchedWidgetIds = widgetPositions.stream()
343 				.mapToInt(WidgetPosition::getWidgetId)
344 				.filter(i -> !dashboard.get().getWidgets().stream().mapToInt(Widget::getId).anyMatch(w -> i == w))
345 				.mapToObj(Integer::toString)
346 				.collect(Collectors.toSet());
347 		if (!unmatchedWidgetIds.isEmpty()) {
348 			return badRequest("Some widgets ids do not exist in the dashboard", unmatchedWidgetIds);
349 		}
350 
351 		// Update widget positions
352 		widgetPositions.forEach(t -> {
353 			final Widget widget = dashboard.get().getWidget(t.getWidgetId());
354 			widget.setSizeX(t.getSizeX());
355 			widget.setSizeY(t.getSizeY());
356 			widget.setPosX(t.getPosX());
357 			widget.setPosY(t.getPosY());
358 			dashboardController.updateWidget(widget);
359 		});
360 
361 		return ok().build();
362 	}
