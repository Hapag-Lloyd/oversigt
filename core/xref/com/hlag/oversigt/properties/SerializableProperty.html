3   import java.lang.annotation.ElementType;
4   import java.lang.annotation.Retention;
5   import java.lang.annotation.RetentionPolicy;
6   import java.lang.annotation.Target;
7   import java.util.Objects;
8   import java.util.Optional;
9   
10  import org.apache.commons.lang3.builder.ToStringBuilder;
11  import org.apache.commons.lang3.builder.ToStringStyle;
12  
13  import com.fasterxml.jackson.annotation.JsonTypeInfo;
14  
15  import edu.umd.cs.findbugs.annotations.Nullable;
16  
17  /**
18   * This interface marks classes that will be stored in the database in a
19   * separate tables.
20   *
21   * @author neumaol
22   */
23  @JsonTypeInfo(use = JsonTypeInfo.Id.CLASS, include = JsonTypeInfo.As.PROPERTY, property = "$class")
24  public abstract class SerializableProperty implements Comparable<SerializableProperty> {
25  	private final int id;
26  
27  	private String name;
28  
29  	protected SerializableProperty(final int id, final String name) {
30  		this.id = id;
31  		this.name = name;
32  	}
33  
34  	public int getId() {
35  		return id;
36  	}
37  
38  	public String getName() {
39  		return name;
40  	}
41  
42  	public void setName(final String name) {
43  		this.name = name;
44  	}
45  
46  	@Override
47  	public int compareTo(@Nullable final SerializableProperty that) {
48  		return String.CASE_INSENSITIVE_ORDER.compare(getName(),
49  				Optional.ofNullable(that).map(SerializableProperty::getName).orElse(""));
50  	}
51  
52  	@Override
53  	public String toString() {
54  		return getClass().getSimpleName() + ToStringBuilder.reflectionToString(this, ToStringStyle.JSON_STYLE);
55  	}
56  
57  	@Override
58  	public final boolean equals(@Nullable final Object that) {
59  		return this == that;
60  	}
61  
62  	@Override
63  	public final int hashCode() {
64  		return Objects.hashCode(this);
65  	}
66  
67  	@Retention(RetentionPolicy.RUNTIME)
68  	@Target(ElementType.FIELD)
69  	public static @interface Member {
70  		String icon() default "tag";
71  
72  		int size() default 2;
73  
74  		boolean mayBeEmpty() default false;
75  	}
76  
77  	/**
78  	 * Use this annotation to add an description to the annotated object. Unlike
79  	 * JavaDoc this description can be inspected at runtime.
80  	 *
81  	 * @author Olaf Neumann
82  	 */
83  	@Retention(RetentionPolicy.RUNTIME)
84  	@Target({ ElementType.TYPE, ElementType.FIELD })
85  	public static @interface Description {
86  		/**
87  		 * The description of the annotated object
88  		 *
89  		 * @return the description
90  		 */
91  		String value();
92  	}
