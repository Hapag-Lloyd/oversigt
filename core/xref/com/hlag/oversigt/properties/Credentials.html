1   package com.hlag.oversigt.properties;
2   
3   import org.apache.commons.lang3.builder.ToStringBuilder;
4   import org.apache.commons.lang3.builder.ToStringStyle;
5   
6   import com.fasterxml.jackson.annotation.JsonCreator;
7   import com.fasterxml.jackson.annotation.JsonProperty;
8   import com.hlag.oversigt.properties.SerializableProperty.Description;
9   
10  /**
11   * This class stores the credentials for a given use case. Credentials will be
12   * stored separately within the dashboard and can be changed without the need to
13   * change all event source configurations.
14   *
15   * <p>
16   * This class is also useful because when configuring an event source you don't
17   * need to know passwords because they can be entered somewhere else.
18   *
19   * @author neumaol
20   */
21  @Description("Store credentials to access a resource using them. The password will be stored in a way that others can use it without being able to read it.")
22  public class Credentials extends SerializableProperty {
23  	public static final Credentials EMPTY = new Credentials(0, "", "", "", "");
24  
25  	@Member(icon = "user", size = 3, mayBeEmpty = true)
26  	private String username;
27  
28  	@Member(icon = "lock", size = 3)
29  	private String password;
30  
31  	@Member(icon = "cloud", size = 2, mayBeEmpty = true)
32  	private String domain;
33  
34  	@JsonCreator
35  	public Credentials(@JsonProperty("id") final int id,
36  			@JsonProperty("name") final String name,
37  			@JsonProperty("username") final String username,
38  			@JsonProperty("password") final String password,
39  			@JsonProperty("domain") final String domain) {
40  		super(id, name);
41  		this.domain = domain;
42  		this.username = username;
43  		this.password = password;
44  	}
45  
46  	/**
47  	 * The domain of these credentials
48  	 *
49  	 * @return The domain of these credentials
50  	 */
51  	public String getDomain() {
52  		return domain;
53  	}
54  
55  	/**
56  	 * The password belonging to these credentials
57  	 *
58  	 * @return The password belonging to these credentials
59  	 */
60  	public String getPassword() {
61  		return password;
62  	}
63  
64  	/**
65  	 * The username belonging to these credentials
66  	 *
67  	 * @return The username belonging to these credentials
68  	 */
69  	public String getUsername() {
70  		return username;
71  	}
72  
73  	public void setDomain(final String domain) {
74  		this.domain = domain;
75  	}
76  
77  	public void setPassword(final String password) {
78  		this.password = password;
79  	}
80  
81  	public void setUsername(final String username) {
82  		this.username = username;
83  	}
84  
85  	@Override
86  	public String toString() {
87  		return ToStringBuilder.reflectionToString(new Credentials(getId(), getName(), username, "********", domain),
88  				ToStringStyle.JSON_STYLE);
89  	}
90  }
