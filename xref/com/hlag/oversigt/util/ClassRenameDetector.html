28  public final class ClassRenameDetector {
29  	private static final Logger LOGGER = LoggerFactory.getLogger(ClassRenameDetector.class);
30  
31  	private static final String RENAME_FILENAME = "data/classRenames.properties";
32  
33  	private static final Properties RENAME_PROPERTIES = new Properties();
34  
35  	static {
36  		final Path path = Paths.get(RENAME_FILENAME);
37  		if (Files.exists(path)) {
38  			LOGGER.info("Class renaming properties exist. Loading them...");
39  			try (InputStream in = new BufferedInputStream(Files.newInputStream(path))) {
40  				RENAME_PROPERTIES.load(in);
41  				LOGGER.info("Found {} class renaming instructions.", RENAME_PROPERTIES.size());
42  				LOGGER.debug("Class renaming properties: {}", RENAME_PROPERTIES);
43  			} catch (final IOException e) {
44  				LOGGER.error("Unable to load class renaming properties", e);
45  			}
46  		}
47  	}

49  	private static String getSimpleName(final String className) {
50  		return className.contains(".") ? className.substring(className.lastIndexOf(".") + 1) : className;
51  	}
52  
53  	public static Optional<EventSourceKey> detectPackageMove(final Map<String, EventSourceKey> keys,
54  			final String className) {
55  		final String simpleClassNameWithDot = "." + getSimpleName(className);
56  		final Set<Entry<String, EventSourceKey>> candidates = keys.entrySet()
57  				.stream()
58  				.filter(s -> s.getKey().startsWith("class:"))
59  				.filter(s -> s.getKey().endsWith(simpleClassNameWithDot))
60  				.collect(toSet());
61  		if (candidates.size() == 1) {
62  			return Optional.of(candidates.iterator().next().getValue());
63  		}
64  		if (candidates.isEmpty()) {
65  			LOGGER.warn("No moved class found for name: {}", className);
66  		} else if (candidates.size() > 1) {
67  			LOGGER.warn("Multiple moved candidates found for class [{}]: {}",
68  					className,
69  					candidates.stream().map(Entry::getKey).collect(joining(", ")));
70  		}
71  		return Optional.empty();
72  	}

74  	private static final MetricStringDistance DISTANCE = new Levenshtein();
75  
76  	public static Optional<EventSourceKey> detectSimpleRename(final Map<String, EventSourceKey> keys,
77  			final String className) {
78  		final String simpleClassName = getSimpleName(className);
79  		final List<KeyDistance> candidates = keys.entrySet()//
80  				.stream()
81  				.filter(s -> s.getKey().startsWith("class:"))
82  				.map(s -> new KeyDistance(s.getValue(), DISTANCE.distance(simpleClassName, getSimpleName(s.getKey()))))
83  				.filter(t -> t.getDistance() <= 2.0)
84  				.sorted((a, b) -> Double.compare(a.getDistance(), b.getDistance()))
85  				.collect(toList());
86  		if (candidates.size() == 1) {
87  			return Optional.of(candidates.iterator().next().getKey());
88  		}
89  		if (candidates.isEmpty()) {
90  			LOGGER.warn("No renamed class found for name: {}", className);
91  		} else if (candidates.size() > 1) {
92  			LOGGER.warn("Multiple renamed candidates found for class [{}]: {}",
93  					className,
94  					candidates.stream().map(KeyDistance::getKey).map(EventSourceKey::getKey).collect(joining(", ")));
95  		}
96  		return Optional.empty();
97  	}

99  	public static Optional<EventSourceKey> detectComplexRename(final Map<String, EventSourceKey> keys,
100 			final String className) {
101 		final String newClassName = RENAME_PROPERTIES.getProperty(className);
102 		if (newClassName != null) {
103 			try {
104 				Class.forName(newClassName);
105 				LOGGER.info("Class {} replaced by {}", className, newClassName);
106 				return Optional.of(keys.get("class:" + newClassName));
107 			} catch (final Exception e) {
108 				LOGGER.error("Unable to load class {} in order to replace class {}", newClassName, className, e);
109 			}
110 		}
111 		return Optional.empty();
112 	}
