8   @FunctionalInterface
9   public interface ThrowingFunction<T, R> extends Function<T, R> {
10  	static <T, R> Function<T, R> sneaky(final ThrowingFunction<T, R> function) {
11  		return function;
12  	}
13  
14  	@Override
15  	default R apply(@Nullable final T t) {
16  		try {
17  			return applyThrowing(t);
18  		} catch (final Exception e) {
19  			throw e instanceof RuntimeException ? (RuntimeException) e : new SneakyException(e);
20  		}
21  	}
22  
23  	R applyThrowing(@Nullable T t) throws Exception;
24  }
