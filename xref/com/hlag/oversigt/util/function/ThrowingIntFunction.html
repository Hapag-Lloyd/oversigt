3   import java.util.function.IntFunction;

5   import de.larssh.utils.SneakyException;

7   @FunctionalInterface
8   public interface ThrowingIntFunction<T> extends IntFunction<T> {
9   	static <T> IntFunction<T> sneakyInt(final ThrowingIntFunction<T> function) {
10  		return function;
11  	}
12  
13  	@Override
14  	default T apply(final int value) {
15  		try {
16  			return applyThrowing(value);
17  		} catch (final Exception e) {
18  			throw e instanceof RuntimeException ? (RuntimeException) e : new SneakyException(e);
19  		}
20  	}
21  
22  	T applyThrowing(int value) throws Exception;
23  }
