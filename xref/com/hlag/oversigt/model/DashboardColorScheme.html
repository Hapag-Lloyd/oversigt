7   public enum DashboardColorScheme {
8   	COLORED("colored", "default"),
9   	SINGLE_COLOR("singleColor"),
10  	TILED_VERTICAL_GRADIENT("tiledVerticalGradient"),
11  	TILED_HORIZONTAL_GRADIENT("tiledHorizontalGradient"),
12  	VERTICAL_GRADIENT("verticalGradient"),
13  	HORIZONTAL_GRADIENT("horizontalGradient");
14  
15  	private final Set<String> names;
16  
17  	DashboardColorScheme(final String... names) {
18  		this.names = new HashSet<>(Arrays.asList(names));
19  	}
20  
21  	public static DashboardColorScheme fromString(final String string) {
22  		for (final DashboardColorScheme scheme : values()) {
23  			if (scheme.name().equals(string) || scheme.names.stream().anyMatch(string::equals)) {
24  				return scheme;
25  			}
26  		}
27  		throw new RuntimeException("No ColorScheme found for: " + string);
28  	}
29  }
