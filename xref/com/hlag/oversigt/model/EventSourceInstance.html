182 	public static Predicate<EventSourceInstance> createFilter(final String filter) {
183 		return i -> i.getName().toLowerCase().contains(filter)
184 				|| i.getId().toLowerCase().contains(filter)
185 				|| Optional.ofNullable(i.getFrequency()).map(Object::toString).orElse("").toLowerCase().contains(filter)
186 				|| i.getDescriptor()
187 						.getProperties()
188 						.stream()
189 						.filter(i::hasPropertyValue)
190 						.anyMatch(p -> i.getPropertyValueString(p).toLowerCase().contains(filter))
191 				|| i.getDescriptor()
192 						.getDataItems()
193 						.stream()
194 						.filter(i::hasPropertyValue)
195 						.anyMatch(p -> i.getPropertyValueString(p).toLowerCase().contains(filter));
