181 	private static Color parseString(final String string) {
182 		if (string.length() == 0) {
183 			return BLACK;
184 		}
185 		switch (string.charAt(0)) {
186 		case '#':
187 			return parseHexColor(string.substring(1));
188 		case 'r':
189 		case 'h':
190 			return parseFunctionColor(string);
191 		default:
192 			throw new RuntimeException("Unable to parse color: " + string);
193 		}
194 	}

196 	private static Color parseHexColor(final String string) {
197 		if (string.length() == 3) {
198 			final String r = string.substring(0, 1);
199 			final String g = string.substring(1, 2);
200 			final String b = string.substring(2, 3);
201 			return new Color(Integer.parseInt(r + r, 16),
202 					Integer.parseInt(g + g, 16),
203 					Integer.parseInt(b + b, 16),
204 					1.0);
205 		} else if (string.length() == 6) {
206 			final String r = string.substring(0, 2);
207 			final String g = string.substring(2, 4);
208 			final String b = string.substring(4, 6);
209 			return new Color(Integer.parseInt(r, 16), Integer.parseInt(g, 16), Integer.parseInt(b, 16), 1.0);
210 		} else {
211 			throw new RuntimeException("Unable to parse color: " + string);
212 		}
213 	}

219 	@SuppressFBWarnings(value = "IMPROPER_UNICODE",
220 			justification = "comparing lower cased string with ASCII characters only")
221 	private static Color parseFunctionColor(final String string) {
222 		final Matcher matcher = FUNCTION_PATTERN.matcher(string);
223 		if (matcher.find()) {
224 			final String function = matcher.group(1);
225 			final String r = matcher.group(2);
226 			final String g = matcher.group(3);
227 			final String b = matcher.group(4);
228 			final String a = matcher.group(5);
229 
230 			switch (function.toLowerCase()) {
231 			case "rgb":
232 				return new Color(Integer.parseInt(r), Integer.parseInt(g), Integer.parseInt(b), 1.0);
233 			case "rgba":
234 				return new Color(Integer.parseInt(r), Integer.parseInt(g), Integer.parseInt(b), Double.parseDouble(a));
235 			case "hsl":
236 			case "hsla":
237 				final int[] rgb = hslToRgb(Integer.parseInt(r), Integer.parseInt(r), Integer.parseInt(b));
238 				return new Color(rgb[0], rgb[1], rgb[2], a != null ? Double.parseDouble(a) : 1.0);
239 			default:
240 				// throw below exception
241 			}
242 		}
243 		throw new RuntimeException("Unable to parse color: " + string);
244 	}

250 	/**
251 	 * Converts an HSL color value to RGB. Conversion formula adapted from
252 	 * http://en.wikipedia.org/wiki/HSL_color_space. Assumes h, s, and l are
253 	 * contained in the set [0, 1] and returns r, g, and b in the set [0, 255].
254 	 *
255 	 * @param h The hue
256 	 * @param s The saturation
257 	 * @param l The lightness
258 	 * @return int array, the RGB representation
259 	 */
260 	public static int[] hslToRgb(final float h, final float s, final float l) {
261 		final float r;
262 		final float g;
263 		final float b;
264 
265 		// achromatic
266 		if (s == 0f) {
267 			r = l;
268 			g = l;
269 			b = l;
270 		} else {
271 			final float q = l < 0.5f ? l * (1 + s) : l + s - l * s;
272 			final float p = 2 * l - q;
273 			r = hueToRgb(p, q, h + 1f / 3f);
274 			g = hueToRgb(p, q, h);
275 			b = hueToRgb(p, q, h - 1f / 3f);
276 		}
277 		return new int[] { (int) (r * 255), (int) (g * 255), (int) (b * 255) };
278 	}
279 
280 	private static float hueToRgb(final float p, final float q, final float t) {
281 		float tt = t;
282 		if (tt < 0f) {
283 			tt += 1f;
284 		} else if (tt > 1f) {
285 			tt -= 1f;
286 		}
287 		if (tt < 1f / 6f) {
288 			return p + (q - p) * 6f * tt;
289 		}
290 		if (tt < 1f / 2f) {
291 			return q;
292 		}
293 		if (tt < 2f / 3f) {
294 			return p + (q - p) * (2f / 3f - tt) * 6f;
295 		}
296 		return p;
297 	}
