18  import com.hlag.oversigt.sources.event.TwoColumnListEvent.ListEventItem;

20  @EventSource(displayName = "GIT Committers", view = "List", hiddenDataItems = { "moreinfo", "updated-at-message" })
21  public class GitCommittersEventSource extends AbstractGitCommitEventSource<OversigtEvent> {
22  	public GitCommittersEventSource() {
23  		// no fields to be initialized
24  	}
25  
26  	@Override
27  	protected Optional<OversigtEvent> produceEvent() throws EventSourceException {
28  		// Count commits per user
29  		final Map<String, Integer> committers;
30  		try {
31  			committers = streamLogWithoutFilteredUsers(s -> s.map(r -> r.getCommitterIdent().getName())
32  					.map(createNameMapper())
33  					.collect(toMap(Function.identity(), x -> 1, (x, y) -> x + y)));
34  		} catch (final GitAPIException | IOException e) {
35  			throw new EventSourceException("Unable to retreive GIT information", e);
36  		}
37  
38  		// Order by commit count
39  		final List<ListEventItem<Integer>> list = committers.entrySet()
40  				.stream()
41  				.sorted((a, b) -> -Integer.compare(a.getValue(), b.getValue()))
42  				.limit(getLimit() > 0 ? getLimit() : committers.size())
43  				.map(ListEventItem::new)
44  				.collect(toList());
45  
46  		// create event
47  		return Optional.of(new TwoColumnListEvent<>(list));
48  	}
49  }
