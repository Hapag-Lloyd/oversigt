32  @EventSource(view = "Rickshawgraph", displayName = "CPU Usage")
33  public class CpuUsageGraphEventSource extends ScheduledEventSource<ComplexGraphEvent> {
34  	private final Map<ZonedDateTime, Map<Server, Integer>> values = new TreeMap<>();
35  
36  	private Server[] servers = new Server[] { new Server() };
37  
38  	private int historyLength = 10;
39  
40  	public CpuUsageGraphEventSource() {
41  		// no fields to be initialized
42  	}
43  
44  	@Override
45  	protected Optional<ComplexGraphEvent> produceEvent() {
46  		final ZonedDateTime now = ZonedDateTime.now(ZoneOffset.UTC);
47  
48  		// get usages from hosts
49  		final Map<Server, Integer> usages = Arrays.stream(getServers())
50  				.parallel()
51  				.collect(Collectors.toMap(Function.identity(), this::getCpuUsage));
52  		values.put(now, usages);
53  
54  		// remove old values
55  		values.keySet().removeIf(now.minusMinutes(getHistoryLength())::isAfter);
56  
57  		// graphen berechnen
58  		final long secondsOffset = now.withHour(0).withMinute(0).withSecond(0).toEpochSecond();
59  		final Map<String, Series> series = new TreeMap<>(String.CASE_INSENSITIVE_ORDER.reversed());
60  		for (final Server server : getServers()) {
61  			final List<Point> points = values.entrySet()
62  					.stream()
63  					.map(e -> new Point(e.getKey().toEpochSecond() - secondsOffset,
64  							e.getValue().get(server).longValue()))
65  					.collect(Collectors.toList());
66  			series.put(server.hostname, new Series(server.getDisplayName(), points));
67  		}
68  
69  		return Optional.of(new ComplexGraphEvent(new ArrayList<>(series.values())));
70  	}
71  
72  	private int getCpuUsage(final Server server) {
73  		final int usage;
74  		try (StartedAction action = getStatisticsCollector().startAction("Retreive CPU usage", server.hostname)) {
75  			switch (server.operatingSystem) {
76  			case Linux:
77  			case Aix:
78  				usage = getUnixCpuUsage(server);
79  				break;
80  			// case Windows:
81  			// usage = getWindowsCpuUsage(server);
82  			// break;
83  			default:
84  				throw new RuntimeException("Unknown operating system: " + server.operatingSystem.name());
85  			}
86  		}
87  		Utils.logInfo(getSpecialLogger(), "%s: %s", server.hostname, usage);
88  		return usage;
89  	}
90  
91  	private int getUnixCpuUsage(final Server server) {
92  		final double usage = SshConnection.getConnection(server.hostname, server.port, server.username, server.password)
93  				.getCpuUsage();
94  		if (Double.isNaN(usage)) {
95  			return 0;
96  		}
97  		int percent = (int) (100 * usage + 0.5);
98  		if (percent < 0) {
99  			percent = 0;
100 		}
101 		if (percent > 100) {
102 			percent = 100;
103 		}
104 		return percent;
105 	}
