24  public abstract class AbstractGitCommitEventSource<E extends OversigtEvent> extends AbstractGitEventSource<E> {
25  	private int limit = 5;
26  
27  	private String[] usersToSuppress = new String[0];
28  
29  	private NameMapping[] nameMappings = new NameMapping[0];
30  
31  	@Property(name = "Limit",
32  			description = "The maximum number of committers to display. Values below 1 display all committers.")
33  	public int getLimit() {
34  		return limit;
35  	}
36  
37  	public void setLimit(final int limit) {
38  		this.limit = limit;
39  	}
40  
41  	@Property(name = "Users to suppress",
42  			description = "If you want to suppress certain names, enter them here. This may be useful to prevent technical users from being shown in the list of committers.")
43  	@JsonHint(arrayStyle = ArrayStyle.TABLE)
44  	public String[] getUsersToSuppress() {
45  		return usersToSuppress;
46  	}
47  
48  	public void setUsersToSuppress(final String[] usersToSuppress) {
49  		this.usersToSuppress = usersToSuppress;
50  	}
51  
52  	@Property(name = "Name mappings",
53  			description = "Some users have different user names. Use these mappings to map all the known user names to one single name.")
54  	public NameMapping[] getNameMappings() {
55  		return nameMappings;
56  	}
57  
58  	public void setNameMappings(final NameMapping[] nameMappings) {
59  		this.nameMappings = nameMappings;
60  	}
61  
62  	protected <R> R streamLogWithoutFilteredUsers(final Function<Stream<RevCommit>, R> function)
63  			throws NoHeadException, GitAPIException, IOException {
64  		final Set<String> usersToSuppress = new HashSet<>(Arrays.asList(getUsersToSuppress()));
65  		return streamLog(
66  				s -> function.apply(s.filter(rv -> !usersToSuppress.contains(rv.getCommitterIdent().getName()))));
67  	}
68  
69  	protected Function<String, String> createNameMapper() {
70  		final Map<String, String> nameMap
71  				= Arrays.stream(getNameMappings()).collect(toMap(NameMapping::getName, NameMapping::getMapTo));
72  		return x -> nameMap.computeIfAbsent(x, Function.identity());
73  	}
74  
75  	@JsonHint(headerTemplate = "{{self.name}}", arrayStyle = ArrayStyle.GRID)
76  	public static class NameMapping {
77  		private final String name;
78  
79  		private final String mapTo;
80  
81  		@JsonCreator
82  		public NameMapping(@JsonProperty("name") final String name, @JsonProperty("mapTo") final String mapTo) {
83  			this.name = name;
84  			this.mapTo = mapTo;
85  		}
86  
87  		public String getName() {
88  			return name;
89  		}
90  
91  		public String getMapTo() {
92  			return mapTo;
93  		}
94  	}
95  }
