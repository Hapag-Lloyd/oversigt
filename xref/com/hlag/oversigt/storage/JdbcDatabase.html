87  			@Named("databaseLocation") final String databaseLocation,
88  			@Named("databaseName") final String schema,
89  			@Named("databaseUsername") final String username,
90  			@Named("databasePassword") final String password) {
91  		sqlDialect = dialect;
92  		try {
93  			LOGGER.info("Loading JDBC driver class: " + dialect.getDriverClassName());
94  			Class.forName(dialect.getDriverClassName());
95  			LOGGER.info("Creating JDBC connection for location '{}' and schema '{}'", databaseLocation, schema);
96  			connection = DriverManager
97  					.getConnection(dialect.getJdbcConnectionUrl(databaseLocation, schema, username, password));
98  			checkTable(TABLE_EVENT_SOURCE, COLUMN_OPTIONS_EVENT_SOURCE);
99  			checkTable(TABLE_EVENT_SOURCE_PROPERTY, COLUMN_OPTIONS_EVENT_SOURCE_PROPERTY);
100 			checkTable(TABLE_DASHBOARD, COLUMN_OPTIONS_DASHBOARD);
101 			checkTable(TABLE_WIDGET, COLUMN_OPTIONS_WIDGET);
102 			checkTable(TABLE_WIDGET_DATA, COLUMN_OPTIONS_WIDGET_DATA);
103 			checkTable(TABLE_VALUES, COLUMN_OPTIONS_VALUES);
104 		} catch (final SQLException | ClassNotFoundException e) {
105 			throw new RuntimeException("Unable to create database connection", e);
106 		}

114 	private void checkTable(final String tableName, final ColumnOptions[] options) throws SQLException {
115 		LOGGER.debug("Checking table " + tableName);
116 		final DatabaseMetaData databaseMetaData = getConnection().getMetaData();
117 		try (ResultSet tables = databaseMetaData.getTables(null, null, tableName, new String[] { "TABLE" })) {
118 			final List<Object> tableList = getList(tables, rs -> rs.getString("TABLE_NAME"));
119 			if (tableList.isEmpty()) {
120 				// Table does not exist. we need to create it
121 				LOGGER.info("Creating table " + tableName);
122 				final String sql = getDialect().createTable(tableName, options);
123 				executeUpdate(sql);
124 			} else {
125 				final List<Map<String, Object>> columnProperties;
126 				try (ResultSet columns = databaseMetaData.getColumns(null, null, tableName, "%")) {
127 					columnProperties = load(columns,
128 							rs -> readColumnValues(rs,
129 									new String[] {
130 											"COLUMN_NAME",
131 											"TYPE_NAME",
132 											"COLUMN_SIZE",
133 											"IS_NULLABLE",
134 											"IS_AUTOINCREMENT" }));
135 				}
136 				for (final ColumnOptions option : options) {
137 					final Optional<Map<String, Object>> foundColumnProperties = columnProperties.stream()
138 							.filter(cp -> option.getName().equals(cp.get("COLUMN_NAME")))
139 							.findFirst();
140 					if (!foundColumnProperties.isPresent()) {
141 						LOGGER.info("Altering table " + tableName + ". Adding column " + option.getName());
142 						final String sql = getDialect().alterTableAddColumn(tableName, option);
143 						executeUpdate(sql);
144 					} else {
145 						columnProperties.remove(foundColumnProperties.get());
146 					}
147 				}
148 				final List<String> columnsToDrop = columnProperties.stream()
149 						.map(m -> (String) m.get("COLUMN_NAME"))
150 						.collect(Collectors.toList());
151 				for (final String columnToDrop : columnsToDrop) {
152 					LOGGER.info("Altering table " + tableName + ". Dropping column " + columnToDrop);
153 					final String sql = getDialect().alterTableDropColumn(tableName, columnToDrop);
154 					executeUpdate(sql);
155 				}
156 			}
157 		} catch (final Exception e) {
158 			throw new RuntimeException("Unable to check table: " + tableName, e);
159 		}
160 	}

298 						"NAME",

302 			} else {
303 				// create
304 				insert(TABLE_EVENT_SOURCE_PROPERTY,
305 						Optional.empty(),
306 						"EVENT_SOURCE_ID",
307 						instance.getId(),
308 						"TYPE",
309 						e.getKey().getType(),
310 						"NAME",
311 						e.getKey().getName(),

522 	private Widget loadWidget(final Map<String, Object> map,
523 			final List<Map<String, Object>> widgetDataMapList,
524 			final Function<String, EventSourceInstance> instanceProvider) {
525 		final int id = (int) map.get("ID");
526 		final EventSourceInstance eventSource = instanceProvider.apply((String) map.get("EVENT_SOURCE_ID"));
527 		final String title = (String) map.get("TITLE");
528 		final String name = (String) map.get("NAME");
529 		final boolean enabled = is(map.get("ENABLED"));
530 		final int posX = (int) map.get("POS_X");
531 		final int posY = (int) map.get("POS_Y");
532 		final int sizeX = (int) map.get("SIZE_X");
533 		final int sizeY = (int) map.get("SIZE_Y");
534 		final Color backgroundColor = Color.parse((String) map.get("BACKGROUND_COLOR"));
535 		final String style = (String) map.get("STYLE");
536 
537 		final Widget widget
538 				= new Widget(id, eventSource, title, name, enabled, posX, posY, sizeX, sizeY, backgroundColor, style);
539 
540 		final Map<String, String> datas = widgetDataMapList.stream()
541 				.filter(d -> (int) d.get("WIDGET_ID") == widget.getId())
542 				.collect(Collectors.toMap(d -> (String) d.get("NAME"), d -> (String) d.get("VALUE")));
543 
544 		eventSource.getDescriptor()

663 	@Override
664 	public <T extends SerializableProperty> T createProperty(final Class<T> clazz,
665 			final String name,
666 			final Object... parameters) {
667 		final OptionalInt id = insert(TABLE_VALUES, RETRIEVE_ID, "CLASS", clazz.getName(), "NAME", name, "JSON", "{}")
668 				.map(OptionalInt::of)
669 				.orElse(OptionalInt.empty());
670 		if (!id.isPresent()) {
671 			throw new RuntimeException("Unable to create property: " + name);
672 		}
673 		final Object[] params = new Object[2 + parameters.length];
674 		params[0] = id.getAsInt();
675 		params[1] = name;
676 		System.arraycopy(parameters, 0, params, 2, parameters.length);
677 		final T value;
678 		try {
679 			value = TypeUtils.getAppropriateConstructor(clazz, params).newInstance(params);
680 		} catch (final InstantiationException
681 				| IllegalAccessException
682 				| IllegalArgumentException
683 				| InvocationTargetException e) {
684 			throw new RuntimeException(
685 					"Class " + clazz.getName() + " does not contain a constructor to be used in the database class",
686 					e);
687 		}
688 		updateProperty(value);
689 		return value;
690 	}

699 
700 	private <T, I> void persist(final String table,
701 			final String idColumn,
702 			final T item,
703 			final Function<T, I> getId,
704 			final Map<String, Object> values,
705 			final Map<String, Object> creationValues) {
706 		final List<String> found
707 				= load(table, idColumn, Optional.of(idColumn), getId.apply(item), rs -> rs.getString(idColumn));
708 		if (found.isEmpty()) {
709 			values.put(idColumn, getId.apply(item));
710 			values.putAll(creationValues);
711 			insert(table, Optional.empty(), values);
712 		} else if (found.size() == 1) {
713 			// update existing entry
714 			update(table,
715 					Maps.<String, Object>builder() //
716 							.put(idColumn, getId.apply(item))
717 							.get(),
718 					values);
719 		} else {
720 			throw new RuntimeException("More than one line found for ID: " + getId.apply(item));
721 		}
722 	}
