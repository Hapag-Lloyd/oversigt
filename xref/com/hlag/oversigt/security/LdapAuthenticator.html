27  @Singleton
28  public class LdapAuthenticator implements Authenticator {
29  	private static final Logger LOGGER = LoggerFactory.getLogger(LdapAuthenticator.class);

31  	private final String url;

41  	private Optional<DirContext> serviceCtx = Optional.empty();

44  	private RoleProvider roleProvider;

46  	@Inject
47  	public LdapAuthenticator(final LdapConfiguration configuration) {
48  		this(notNullOrEmpty(configuration.getUrl(), "The LDAP URL must not be null or empty"),
49  				notNullOrEmpty(configuration.getBaseDn(), "The Base-DN must not be null or empty"),
50  				notNullOrEmpty(configuration.getBindUser(), "The LDAP identifying attribute must not be null or empty"),
51  				notNullOrEmpty(configuration.getBindPassword(), "The LDAP Bind user must not be null or empty"),
52  				notNullOrEmpty(configuration.getUidAttribute(), "The LDAP Bind password must not be null or empty"));
53  	}
54  
55  	private LdapAuthenticator(final String url,
56  			final String baseDn,
57  			final String bindUser,
58  			final String bindPassword,
59  			final String uidAttribute) {
60  		this.url = url;
61  		this.baseDn = baseDn;
62  		identifyingAttribute = uidAttribute;
63  		this.bindUser = bindUser;
64  		this.bindPassword = bindPassword;
65  
66  		try {
67  			setDirContext();
68  		} catch (final NamingException e) {
69  			LOGGER.error("Unable to create Directory Context", e);
70  			throw new RuntimeException("Unable to create Directory Context", e);
71  		}
72  	}
73  
74  	private DirContext setDirContext() throws NamingException {
75  		if (serviceCtx.isPresent()) {
76  			try {
77  				serviceCtx.get().close();
78  			} catch (final Exception e) {
79  				LOGGER.warn("Exception while closing old Service Context", e);
80  			}
81  		}
82  
83  		final Properties serviceEnv = new Properties();
84  		serviceEnv.put(Context.INITIAL_CONTEXT_FACTORY, "com.sun.jndi.ldap.LdapCtxFactory");
85  		serviceEnv.put(Context.PROVIDER_URL, url);
86  		serviceEnv.put(Context.SECURITY_AUTHENTICATION, "simple");
87  		serviceEnv.put(Context.SECURITY_PRINCIPAL, bindUser);
88  		serviceEnv.put(Context.SECURITY_CREDENTIALS, bindPassword);
89  
90  		serviceCtx = Optional.of(new InitialDirContext(serviceEnv));
91  		return serviceCtx.get();
92  	}

94  	@Override
95  	public Optional<Principal> login(final String username, final String password) {
96  		Objects.requireNonNull(username, "Username must not be null");
97  		Objects.requireNonNull(password, "Password must not be null");
98  
99  		try {
100 			final Optional<Principal> distinguishedName = readDistinguishedName(username);
101 			if (distinguishedName.isPresent()) {
102 				if (authorize(distinguishedName.get().getDistinguishedName(), password)) {
103 					return distinguishedName;
104 				}
105 			}
106 		} catch (final NamingException e) {
107 			LOGGER.error("Unable to log in user", e);
108 		}
109 		return Optional.empty();
110 	}

112 	@Override
113 	public Optional<Principal> readPrincipal(final String username) {
114 		try {
115 			return readDistinguishedName(username);
116 		} catch (final NamingException e) {
117 			throw new RuntimeException("Unable to read user information for username: " + username, e);
118 		}
119 	}
120 
121 	@Override
122 	public boolean isUsernameValid(final String username) {
123 		if (Strings.isNullOrEmpty(username)) {
124 			return false;
125 		}
126 		try {
127 			return readDistinguishedName(username).isPresent();
128 		} catch (final NamingException e) {
129 			LOGGER.error("Unable to check username validity", e);
130 			return false;
131 		}
132 	}

134 	private Optional<Principal> readDistinguishedName(final String username) throws NamingException {
135 		final String[] attributeFilter = { identifyingAttribute, "distinguishedName", "sn", "givenname", "mail" };
136 		final SearchControls sc = new SearchControls();
137 		sc.setReturningAttributes(attributeFilter);
138 		sc.setSearchScope(SearchControls.SUBTREE_SCOPE);
139 
140 		// use a search filter to find only the user we want to authenticate
141 		final String searchFilter = "(" + identifyingAttribute + "=" + username + ")";
142 		NamingEnumeration<SearchResult> results;
143 		if (serviceCtx.isPresent()) {
144 			try {
145 				results = serviceCtx.get().search(baseDn, searchFilter, sc);
146 			} catch (final NamingException e) {
147 				if (e.getCause() instanceof IOException && "connection closed".equals(e.getCause().getMessage())) {
148 					LOGGER.info("Connection broken. Trying to create a new connection.");
149 					try {
150 						results = setDirContext().search(baseDn, searchFilter, sc);
151 					} catch (final NamingException namingException) {
152 						throw new RuntimeException("Connection was closed and now unable to create a new context",
153 								namingException);
154 					}
155 				} else {
156 					LOGGER.warn("Unknown exception type. Unable to handle it.");
157 					throw e;
158 				}
159 			}
160 		} else {
161 			return Optional.empty();
162 		}
163 
164 		if (!results.hasMore()) {
165 			return Optional.empty();
166 		}
167 
168 		// get the users DN (distinguishedName) from the result
169 		final SearchResult result = results.next();
170 		final Attributes attributes = result.getAttributes();
171 
172 		return Optional.of(new Principal(result.getNameInNamespace(),
173 				username,
174 				attributes.get("givenname").get().toString() + " " + attributes.get("sn").get().toString(),
175 				attributes.get("mail").get().toString(),
176 				roleProvider.getRoles(username)));
177 	}

179 	private boolean authorize(final String distinguishedName, final String password) {
180 		try {
181 			// attempt another authentication, now with the user
182 			final Properties authEnv = new Properties();
183 			authEnv.put(Context.INITIAL_CONTEXT_FACTORY, "com.sun.jndi.ldap.LdapCtxFactory");
184 			authEnv.put(Context.PROVIDER_URL, url);
185 			authEnv.put(Context.SECURITY_PRINCIPAL, distinguishedName);
186 			authEnv.put(Context.SECURITY_CREDENTIALS, password);
187 			new InitialDirContext(authEnv).close();
188 			return true;
189 		} catch (@SuppressWarnings("unused") final Exception e) {
190 			try {
191 				// Sleep to slow down responses for brute force attacks
192 				Thread.sleep(1000);
193 			} catch (@SuppressWarnings("unused") final InterruptedException ignore) {
194 				// on interruption continue
195 			}
196 			return false;
197 		}
198 	}

200 	@Override
201 	public void reloadRoles(final String username) {
202 		Objects.requireNonNull(username);
203 		Principal.getPrincipal(username).ifPresent(p -> p.changeRoles(roleProvider.getRoles(username)));
204 	}

206 	@Override
207 	public void close() {
208 		if (serviceCtx.isPresent()) {
209 			try {
210 				serviceCtx.get().close();
211 			} catch (final NamingException e) {
212 				throw new RuntimeException("Unable to close directory context", e);
213 			}
214 		}
