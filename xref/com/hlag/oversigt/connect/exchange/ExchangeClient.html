3   import java.io.Closeable;
4   import java.time.LocalDate;
5   import java.time.ZoneId;
6   import java.time.ZonedDateTime;
7   import java.util.List;
8   import java.util.Map;
9   
10  import microsoft.exchange.webservices.data.core.service.item.Appointment;
11  import microsoft.exchange.webservices.data.core.service.item.Task;
12  
13  /**
14   * A client allowing to read simple information from Exchange client. This
15   * client simply wraps a library from Microsoft for easier usage.
16   *
17   * @author neumaol
18   *
19   */
20  public interface ExchangeClient extends Closeable {
21  	/**
22  	 * Read all mails from the folder of the logged in user
23  	 *
24  	 * @param folderName the name of the folder to read
25  	 * @return a list of all {@link Mail}s from this folder
26  	 * @throws Exception if something fails
27  	 */
28  	List<Mail> loadMails(String folderName) throws Exception;
29  
30  	/**
31  	 * Read all tasks from 'Dashboard' the folder of the logged in user
32  	 *
33  	 * @return all {@link Task}s from the dashboard folder
34  	 * @throws Exception if something fails
35  	 */
36  	List<Task> loadTasks() throws Exception;
37  
38  	/**
39  	 * Read all appointments of the currently logged in user in the given time range
40  	 *
41  	 * @param from  when to start looking for appointments
42  	 * @param until when to stop looking for appointments
43  	 * @return a list of all appointments
44  	 */
45  	List<Appointment> loadAppointments(ZonedDateTime from, ZonedDateTime until);
46  
47  	/**
48  	 * Read meetings of the given resource for the specified day.
49  	 *
50  	 * @param rooms  the resources for which to retreive the meetings
51  	 * @param day    the day when to look for meetings
52  	 * @param zoneId the zoneId of the server
53  	 * @return a map of room to meetings
54  	 * @throws Exception if something fails
55  	 */
56  	Map<Room, List<Meeting>> getMeetings(List<Room> rooms, LocalDate day, ZoneId zoneId) throws Exception;
57  }
