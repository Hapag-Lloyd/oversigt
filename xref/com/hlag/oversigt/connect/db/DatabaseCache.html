3   import java.sql.Connection;
4   import java.sql.SQLException;
5   import java.time.Duration;
6   import java.time.LocalDateTime;
7   import java.util.ArrayList;
8   import java.util.Collection;
9   import java.util.List;
10  import java.util.concurrent.atomic.AtomicReference;
11  import java.util.stream.Stream;
12  
13  import org.slf4j.Logger;
14  import org.slf4j.LoggerFactory;
15  
16  import com.hlag.oversigt.util.function.ThrowingFunction;
17  
18  public abstract class DatabaseCache<T> {
19  	public static <X> DatabaseCache<X> createCache(final ThrowingFunction<Connection, Collection<X>> readFunction) {
20  		return new DatabaseCache<X>() {
21  			@Override
22  			protected Collection<X> readItems(final Connection connection) throws SQLException {
23  				return readFunction.apply(connection);
24  			}
25  		};
26  	}
27  
28  	private static final Logger LOGGER = LoggerFactory.getLogger(DatabaseCache.class);
29  
30  	private DatabaseCache() {}
31  
32  	private final AtomicReference<List<T>> items = new AtomicReference<>(new ArrayList<>());
33  
34  	private final AtomicReference<LocalDateTime> lastAccess = new AtomicReference<>();
35  
36  	private Duration durationBetweenReads = Duration.ofHours(1);
37  
38  	public Duration getDurationBetweenReads() {
39  		return durationBetweenReads;
40  	}
41  
42  	public void setDurationBetweenReads(final Duration durationBetweenReads) {
43  		if (LOGGER.isTraceEnabled()) {
44  			LOGGER.trace("Set durationBetweenReads=" + durationBetweenReads);
45  		}
46  		this.durationBetweenReads = durationBetweenReads;
47  	}
48  
49  	public Stream<T> stream() {
50  		if (LOGGER.isTraceEnabled()) {
51  			LOGGER.trace("stream called");
52  		}
53  		return getItems().stream();
54  	}
55  
56  	public List<T> getItems() {
57  		final List<T> items = this.items.get();
58  		if (items.isEmpty()) {
59  			LOGGER.warn("Items have not been loaded yet.");
60  		}
61  		return items;
62  	}
63  
64  	public List<T> getItems(final Connection connection) throws SQLException {
65  		if (shouldReadItems()) {
66  			if (LOGGER.isDebugEnabled()) {
67  				LOGGER.debug("Reading items");
68  			}
69  			final List<T> items = new ArrayList<>(readItems(connection));
70  			if (LOGGER.isDebugEnabled()) {
71  				LOGGER.debug("Found " + items.size() + " items");
72  			}
73  			this.items.set(items);
74  		}
75  		return items.get();
76  	}
77  
78  	private synchronized boolean shouldReadItems() {
79  		final LocalDateTime now = LocalDateTime.now();
80  		final boolean yes = lastAccess.get() == null || lastAccess.get().plus(getDurationBetweenReads()).isBefore(now);
81  		if (yes) {
82  			lastAccess.set(now);
83  		}
84  		if (LOGGER.isDebugEnabled()) {
85  			LOGGER.debug("Should read items: " + yes);
86  		}
87  		return yes;
88  	}
89  
90  	protected abstract Collection<T> readItems(Connection connection) throws SQLException;
91  }
