16  /**
17   * JIRA client with no restrictions. This class is not intended to be used
18   * directly by consuming code. This class should always be an invisible
19   * implementation detail.
20   */
21  final class UnlimitedJiraClient implements JiraClient {
22  	private final ServerConnection connection;
23  
24  	private final Credentials credentials;
25  
26  	private Optional<JiraRestClient> jiraClient;
27  
28  	/**
29  	 * Create a new JIRA client without limitations. This constructor should NOT be
30  	 * called from outside of the package of the declaring class.
31  	 *
32  	 * @param connection  the server details to connect to
33  	 * @param credentials the credentials to be used for the connection
34  	 * @throws JiraClientException if something fails while creating the connection
35  	 */
36  	UnlimitedJiraClient(final ServerConnection connection, final Credentials credentials) throws JiraClientException {
37  		if (connection == ServerConnection.EMPTY) {
38  			throw new JiraClientException("No Jira hostname configured.");
39  		}
40  		if (credentials == Credentials.EMPTY) {
41  			throw new JiraClientException("No Jira credentials configured.");
42  		}
43  
44  		this.connection = connection;
45  		this.credentials = credentials;
46  		jiraClient = Optional.empty();
47  	}
48  
49  	private JiraRestClient getJiraRestClient() throws JiraClientException {
50  		if (!jiraClient.isPresent()) {
51  			try {
52  				jiraClient = Optional.of(JiraConfigurationProvider.createClientFactory()
53  						.createWithBasicHttpAuthentication(new URI(connection.getUrl()),
54  								credentials.getUsername(),
55  								credentials.getPassword()));
56  			} catch (final URISyntaxException e) {
57  				throw new JiraClientException("Jira URI is invalid.", e);
58  			}
59  		}
60  		return jiraClient.get();
61  	}
62  
63  	private void resetJiraRestClient() {
64  		try {
65  			if (jiraClient.isPresent()) {
66  				jiraClient.get().close();
67  			}
68  		} catch (@SuppressWarnings("unused") final IOException ignore) {
69  			// ignore exception while closing the connection
70  		}
71  		jiraClient = Optional.empty();
72  	}
73  
74  	/** {@inheritDoc} */
75  	@Override
76  	public List<Issue> search(final String jql, final int maxResults, final int startAt) throws JiraClientException {
77  		try {
78  			return Lists.newArrayList(//
79  					getJiraRestClient()//
80  							.getSearchClient()
81  							.searchJql(jql, maxResults, startAt, null)
82  							.get()
83  							.getIssues());
84  		} catch (final Exception e) {
85  			resetJiraRestClient();
86  			throw new JiraClientException("Failed searching Jira.", e);
87  		}
88  	}
89  }
