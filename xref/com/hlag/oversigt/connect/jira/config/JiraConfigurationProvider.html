8   /**
9    * Provide jira properties
10   *
11   * @author neumaol
12   */
13  @SuppressWarnings("PMD.ClassNamingConventions")
14  public final class JiraConfigurationProvider {
15  	/**
16  	 * Socket Timeout
17  	 *
18  	 * <p>
19  	 * The value of this field will be injected using Google Guice and is named
20  	 * {@code jiraSocketTimeout}.
21  	 */
22  	@SuppressWarnings("checkstyle:MagicNumber")
23  	private static int socketTimeout = 60;
24  
25  	/**
26  	 * Return the timeout for a jira request
27  	 *
28  	 * @return the timeout for a jira request
29  	 */
30  	@SuppressFBWarnings(value = "MRC_METHOD_RETURNS_CONSTANT", justification = "socketTimeout is not really a constant")
31  	public static int getSocketTimeout() {
32  		return socketTimeout;
33  	}
34  
35  	/**
36  	 * Sets the timeout to be used by jira requests
37  	 *
38  	 * @param timeoutInSeconds the timeout in seconds to be used for jira requests
39  	 */
40  	public static void setSocketTimeout(final int timeoutInSeconds) {
41  		socketTimeout = timeoutInSeconds;
42  	}
43  
44  	/**
45  	 * Create a new {@link JiraRestClientFactory}
46  	 *
47  	 * @return a new instanceof {@link JiraRestClientFactory}
48  	 */
49  	public static JiraRestClientFactory createClientFactory() {
50  		return new AsynchronousJiraRestClientFactory();
51  	}
52  
53  	@SuppressWarnings({ "checkstyle:MissingJavadocMethod", "PMD.AvoidThrowingRawExceptionTypes" })
54  	private JiraConfigurationProvider() {
55  		throw new RuntimeException();
56  	}
57  }
