1   package com.hlag.oversigt.connect.jira;
2   
3   import java.util.List;
4   import java.util.concurrent.TimeoutException;
5   
6   import com.atlassian.jira.rest.client.api.domain.Issue;
7   
8   /**
9    * A client enabling consumer code to retrieve information from a JIRA backend
10   *
11   * @author Lars Knickrehm
12   * @author Olaf Neumann
13   */
14  public interface JiraClient {
15  	/**
16  	 * The maximum number of issues to return from the JIRA backend
17  	 */
18  	int MAX_RESULTS_DEFAULT = 500;
19  
20  	/**
21  	 * Search the JIRA backend using the given query
22  	 *
23  	 * @param jql the query to execute on the JIRA backend
24  	 * @return a possibly empty list of issues found in jira
25  	 * @throws JiraClientException if something fails while searching for issues
26  	 * @throws TimeoutException    if the query while waiting before its execution
27  	 */
28  	default List<Issue> search(final String jql) throws JiraClientException, TimeoutException {
29  		return search(jql, MAX_RESULTS_DEFAULT, 0);
30  	}
31  
32  	/**
33  	 * Perform the Jira search with the given parameters
34  	 *
35  	 * @param jql        the query to execute
36  	 * @param maxResults the maximum number of results to return
37  	 * @param startAt    <em>don't know</em>
38  	 * @return the list of found issues
39  	 * @throws JiraClientException if something fails
40  	 * @throws TimeoutException    if the query while waiting before its execution
41  	 */
42  	List<Issue> search(String jql, int maxResults, int startAt) throws JiraClientException, TimeoutException;
43  }
