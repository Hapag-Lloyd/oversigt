1   package com.hlag.oversigt.core.event;

3   import java.util.Optional;
4   
5   import com.google.common.base.Throwables;
6   import com.hlag.oversigt.core.eventsource.EventSourceException;
7   
8   public class ErrorEvent extends OversigtEvent {
9   	private final Optional<String> errorMessage;
10  
11  	private final boolean error = true;
12  
13  	public ErrorEvent(final Throwable t) {
14  		if (t instanceof EventSourceException) {
15  			errorMessage = Optional.of(t.getMessage()); // the EventSourceException always has a message!
16  		} else {
17  			errorMessage = Optional.of(Throwables.getStackTraceAsString(t));
18  		}
19  	}
20  
21  	public ErrorEvent(final Optional<String> errorMessage) {
22  		this.errorMessage = errorMessage;
23  	}
24  
25  	public ErrorEvent(final String errorMessage) {
26  		this(Optional.of(errorMessage));
27  	}
28  
29  	public ErrorEvent(final String errorMessage, final Optional<Throwable> throwable) {
30  		this(errorMessage + throwable.map(Throwables::getStackTraceAsString).map(s -> "<br>" + s).orElse(""));
31  	}
32  
33  	public Optional<String> getErrorMessage() {
34  		return errorMessage;
35  	}
36  
37  	public boolean isError() {
38  		return error;
39  	}
40  }
