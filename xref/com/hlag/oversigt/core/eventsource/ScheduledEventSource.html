27  /**
28   * Scheduled EventSource - produces events with specified time period.
29   * Basically, based on Guava's
30   * {@link com.google.common.util.concurrent.AbstractScheduledService}
31   *
32   * @author avarabyeu
33   */
34  public abstract class ScheduledEventSource<T extends OversigtEvent> extends AbstractScheduledService {
35  	private static final int ALLOWED_NUMBER_OF_FAILED_CALLS = 5;
36  
37  	private static final Map<String, Logger> LOGGERS = Collections.synchronizedMap(new HashMap<>());
38  
39  	/**
40  	 * Time period between sending events
41  	 */
42  	private Duration frequency = Duration.ofMinutes(1);
43  
44  	/**
45  	 * ID of event this event source bound to
46  	 */
47  	@Inject
48  	@EventId
49  	private String eventId;
50  
51  	/**
52  	 * EventBus to send events
53  	 */
54  	@Inject
55  	private EventBus eventBus;
56  
57  	@Inject
58  	private EventSourceStatisticsManager statisticsManager;
59  
60  	private final AtomicInteger numberOfFailedRuns = new AtomicInteger(0);
61  
62  	private final AtomicBoolean stoppedBecauseOfError = new AtomicBoolean(false);
63  
64  	public boolean isStoppedBecauseOfError() {
65  		return stoppedBecauseOfError.get();
66  	}
67  
68  	protected String getEventId() {
69  		return eventId;
70  	}
71  
72  	private Optional<StatisticsCollector> collector = Optional.empty();
73  
74  	protected StatisticsCollector getStatisticsCollector() {
75  		return collector.orElseThrow(() -> new RuntimeException("The statistics collector has not been created."));
76  	}
77  
78  	private EventSourceStatistics getEventSourceStatistics() {
79  		return statisticsManager.getEventSourceStatistics(getEventId());
80  	}
81  
82  	/**
83  	 * Executes one iteration of
84  	 * {@link com.google.common.util.concurrent.AbstractScheduledService} Sends
85  	 * event to event bus
86  	 */
87  	@Override
88  	@SuppressWarnings("checkstyle:XIllegalCatchDefault")
89  	protected final void runOneIteration() {
90  		stoppedBecauseOfError.set(false);
91  		logTrace(getLogger(), "Run one iteration");
92  
93  		try {
94  			logTrace(getLogger(), "Produce event");
95  			collector = Optional.of(statisticsManager.createCollector(getEventId()));
96  			final OversigtEvent event = produceEvent()
97  					.orElseThrow(() -> new EventSourceException("Event source was unable to produce event."));
98  			getStatisticsCollector().success();
99  
100 			logTrace(getLogger(), "Send event");
101 			sendEvent(event);
102 			logTrace(getLogger(), "Done with iteration");
103 			numberOfFailedRuns.set(0);
104 		} catch (final Throwable e) {
105 			logError(getLogger(), "Error while producing event.", e);
106 			numberOfFailedRuns.incrementAndGet();
107 
108 			// prepare error messages
109 			final String message;
110 			final Optional<Throwable> cause;
111 			if (e instanceof EventSourceException) {
112 				message = e.getMessage(); // an EventSourceException always has an exception
113 				cause = Optional.ofNullable(e.getCause());
114 			} else {
115 				message = "Event source did not catch exception properly.";
116 				cause = Optional.of(e);
117 			}
118 			// collect stats
119 			getStatisticsCollector().failure(message, cause);
120 			logTrace(getLogger(), "Send error event");
121 			sendEvent(new ErrorEvent(message, cause));
122 
123 			// clear cache
124 			logTrace(getLogger(), "Remove event from cache: " + eventId);
125 			removeLastEvent();
126 
127 			// count failures
128 			if (getEventSourceStatistics().isAutomticallyStarted()) {
129 				logWarn(getLogger(), "Nightly restart resulted in error. Stopping again.");
130 				stopAsync();
131 			} else if (numberOfFailedRuns.get() > ALLOWED_NUMBER_OF_FAILED_CALLS) {
132 				logWarn(getLogger(),
133 						"Running the event source resulted in %s errors in a row. Maximum allowed errors in a row is %s. Stopping service.",
134 						numberOfFailedRuns.get(),
135 						ALLOWED_NUMBER_OF_FAILED_CALLS);
136 				stoppedBecauseOfError.set(true);
137 				stopAsync();
138 			} else if (!(e instanceof Exception)) /* Throwable is some kind of Error or similar - stop immediately */ {
139 				logError(getLogger(), "Error occurred. Stopping service immediately.");
140 				stoppedBecauseOfError.set(true);
141 				stopAsync();
142 			}
143 		}
144 	}
145 
146 	protected abstract Optional<T> produceEvent() throws EventSourceException;
147 
148 	protected final void sendEvent(final OversigtEvent event) {
149 		event.setId(eventId);
150 		try {
151 			event.setLifetime(getEventLifetime());
152 		} catch (final Exception e) {
153 			logWarn(getLogger(), "Unable to compute event life time", e);
154 		}
155 		this.eventBus.post(event);
156 	}
157 
158 	protected final void removeLastEvent() {
159 		this.eventBus.post(eventId);
160 	}
161 
162 	@Override
163 	protected final Scheduler scheduler() {
164 		return Scheduler.newFixedDelaySchedule(1, getFrequency().toMillis(), TimeUnit.MILLISECONDS);
165 	}
166 
167 	@Override
168 	protected final String serviceName() {
169 		return getClass().getSimpleName() + "[eventID=" + eventId + "]";
170 	}
171 
172 	protected final Logger getLogger() {
173 		return LOGGERS.computeIfAbsent(getClass().getName(), LoggerFactory::getLogger);
174 	}
175 
176 	protected Logger getSpecialLogger() {
177 		return LOGGERS.computeIfAbsent("special." + eventId, LoggerFactory::getLogger);
178 	}
179 
180 	protected Duration getEventLifetime() {
181 		return getFrequency().multipliedBy(3);
182 	}
183 
184 	public final Duration getFrequency() {
185 		return frequency;
186 	}
187 
188 	public final void setFrequency(final Duration frequency) {
189 		this.frequency = frequency;
190 	}
191 }
