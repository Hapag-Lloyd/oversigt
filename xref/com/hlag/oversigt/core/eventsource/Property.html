1   package com.hlag.oversigt.core.eventsource;
2   
3   import java.lang.annotation.Documented;
4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Retention;
6   import java.lang.annotation.RetentionPolicy;
7   import java.lang.annotation.Target;
8   
9   /**
10   * Marks properties if event sources and determines how they should be handled
11   *
12   * @author Olaf Neumann
13   */
14  @Retention(RetentionPolicy.RUNTIME)
15  @Target(ElementType.METHOD)
16  @Documented
17  public @interface Property {
18  	/**
19  	 * The name of the property. If this annotation is not used the camel case name
20  	 * of the getter method is used.
21  	 *
22  	 * @return the name of the property
23  	 */
24  	String name();
25  
26  	/**
27  	 * A short description of the property. This is a text presented to the end
28  	 * user.
29  	 *
30  	 * @return a description of the property.
31  	 */
32  	String description() default "";
33  
34  	/**
35  	 * The input type of the property. This is used for the type attribute of input
36  	 * fields. Leave empty for auto detection based on the properties type.
37  	 *
38  	 * @return input type of the property.
39  	 */
40  	String type() default "";
41  
42  	/**
43  	 * Determines the allowed values for the current property. If the list is empty
44  	 * any value is valid
45  	 *
46  	 * @return a list of allowed values
47  	 */
48  	String[] allowedValues() default {};
49  }
