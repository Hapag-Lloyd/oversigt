11  /**
12   * Service restarting erroneous event sources once a day.
13   *
14   * @author neumaol
15   */
16  @Singleton
17  public class NightlyEventSourceRestarterService extends NightlyService {
18  	@Inject
19  	private EventSourceInstanceController eventSourceInstanceController;
20  
21  	public NightlyEventSourceRestarterService() {
22  		// no fields to be initialized manually, some will be injected
23  	}
24  
25  	/**
26  	 * Executes one iteration of
27  	 * {@link com.google.common.util.concurrent.AbstractScheduledService} Sends
28  	 * event to event bus
29  	 */
30  	@Override
31  	protected final void runOneIteration() {
32  		eventSourceInstanceController.getEventSourceInstances()
33  				.stream()
34  				.filter(not(eventSourceInstanceController::isRunning))
35  				.filter(EventSourceInstance::isEnabled)
36  				.map(EventSourceInstance::getId)
37  				.peek(x -> Utils.sleep((long) (10000 * Math.random())))
38  				.forEach(id -> eventSourceInstanceController.startInstance(id, true));
39  	}
40  
41  	@Override
42  	protected final String serviceName() {
43  		return getClass().getSimpleName();
44  	}
45  }
