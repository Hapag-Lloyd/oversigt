1   package com.hlag.oversigt.core.eventsource;
2   
3   import java.lang.annotation.ElementType;
4   import java.lang.annotation.Retention;
5   import java.lang.annotation.RetentionPolicy;
6   import java.lang.annotation.Target;
7   
8   import com.google.inject.Module;
9   
10  /**
11   * Marks class as event source
12   *
13   * @author Andrei Varabyeu
14   * @author Olaf Neumann
15   */
16  @Target(value = ElementType.TYPE)
17  @Retention(value = RetentionPolicy.RUNTIME)
18  public @interface EventSource {
19  	/**
20  	 * Defines the display name of the event source. If this field is left out the
21  	 * event source's class name will be used.
22  	 *
23  	 * @return the display name of the event source
24  	 */
25  	String displayName() default "";
26  
27  	String description() default "";
28  
29  	/**
30  	 * Defines the name of the widget that will be used to render the data produced
31  	 * by this event source
32  	 *
33  	 * @return the name of the widget to be used
34  	 */
35  	String view();
36  
37  	/**
38  	 * Lists data items that users may change to configure the appearance of the
39  	 * rendered widget
40  	 *
41  	 * @return a list of data item names
42  	 */
43  	String[] dataItems() default {};
44  
45  	/**
46  	 * Widgets may bring a lot of data items. Sometimes it is not meaningful to show
47  	 * those data items to the user so with this attribute you have the possibility
48  	 * to hide these data items form the user.
49  	 *
50  	 * @return a list of data item names
51  	 */
52  	String[] hiddenDataItems() default {};
53  
54  	/**
55  	 * Provide default values for data items of widgets
56  	 *
57  	 * @return a list of default values
58  	 */
59  	DataItemDefault[] dataItemDefaults() default {};
60  
61  	/**
62  	 * Provides custom Guice configuration for event source instances of the current
63  	 * event source
64  	 *
65  	 * @return a {@link Module} class object describing the module to use
66  	 */
67  	Class<? extends Module> explicitConfiguration() default NOP.class;
68  
69  	/**
70  	 * Empty Module to provide an empty Guice configuration if the event source does
71  	 * not need custom configuration
72  	 */
73  	interface NOP extends Module {
74  		/* Empty module to have a default value */
75  	}
76  }
